 16  * @version 1.0 Beta 2
 17  *
 18  *
 19  * Adding options to one of the setting screens isn't hard.
 20  * Call prepareDBSettingsContext;
 21  * The basic format for a checkbox is:
 22  *    array('check', 'nameInModSettingsAndSQL'),
 23  * And for a text box:
 24  *    array('text', 'nameInModSettingsAndSQL')
 25  * (NOTE: You have to add an entry for this at the bottom!)
 26  *
 27  * In the above examples, it will look for $txt['nameInModSettingsAndSQL'] as the description,
 28  * and $helptxt['nameInModSettingsAndSQL'] as the help popup description.
 29  *
 30  * Here's a quick explanation of how to add a new item:
 31  *
 32  * - A text input box.  For textual values.
 33  *     array('text', 'nameInModSettingsAndSQL', 'OptionalInputBoxWidth'),
 34  * - A text input box.  For numerical values.
 35  *     array('int', 'nameInModSettingsAndSQL', 'OptionalInputBoxWidth'),
 36  * - A text input box.  For floating point values.
 37  *     array('float', 'nameInModSettingsAndSQL', 'OptionalInputBoxWidth'),
 38  * - A large text input box. Used for textual values spanning multiple lines.
 39  *     array('large_text', 'nameInModSettingsAndSQL', 'OptionalNumberOfRows'),
 40  * - A check box.  Either one or zero. (boolean)
 41  *     array('check', 'nameInModSettingsAndSQL'),
 42  * - A selection box.  Used for the selection of something from a list.
 43  *     array('select', 'nameInModSettingsAndSQL', array('valueForSQL' => $txt['displayedValue'])),
 44  *     Note that just saying array('first', 'second') will put 0 in the SQL for 'first'.
 45  * - A password input box. Used for passwords, no less!
 46  *     array('password', 'nameInModSettingsAndSQL', 'OptionalInputBoxWidth'),
 47  * - A permission - for picking groups who have a permission.
 48  *     array('permissions', 'manage_groups'),
 49  * - A BBC selection box.
 50  *     array('bbc', 'sig_bbc'),
 51  *
 52  * For each option:
 53  *  - type (see above), variable name, size/possible values.
 54  *    OR make type '' for an empty string for a horizontal rule.
 55  *  - SET preinput - to put some HTML prior to the input box.
 56  *  - SET postinput - to put some HTML following the input box.
 57  *  - SET invalid - to mark the data as invalid.
 58  *  - PLUS you can override label and help parameters by forcing their keys in the array, for example:
 59  *    array('text', 'invalidlabel', 3, 'label' => 'Actual Label')
 60  */
 61 
 62 if (!defined('ELK'))
 63     die('No access...');

 65 /**
 66  * Settings Form class.
 67  * This class handles display, edit, save, of forum settings.
 68  * It is used by the various admin areas which set their own settings,
 69  * and it is available for addons administration screens.
 70  *
 71  */
 72 class Settings_Form
 73 {
 74     /**
 75      * Configuration variables and values, for this settings form.
 76      * @var array
 77      */
 78     protected $_config_vars;
 79 
 80     /**
 81      * Helper method, it sets up the context for the settings.
 82      * - The basic usage of the six numbered key fields are
 83      * - array(0 ,1, 2, 3, 4, 5
 84      *    0 variable name - the name of the saved variable
 85      *    1 label - the text to show on the settings page
 86      *    2 saveto - file or db, where to save the variable name - value pair
 87      *    3 type - type of data to save, int, float, text, check
 88      *    4 size - false or field size
 89      *    5 help - '' or helptxt variable name
 90      *  )
 91      *
 92      * the following named keys are also permitted
 93      *  'disabled' =>
 94      *  'postinput' =>
 95      *  'preinput' =>
 96      */
 97     public function prepare_file()
 98     {
 99         global $context, $modSettings;
100 
101         $context['config_vars'] = array();
102         $defines = array(
103             'boarddir',
104             'sourcedir',
105             'cachedir',
106         );
107 
108         foreach ($this->_config_vars as $identifier => $config_var)
109         {
110             if (!is_array($config_var) || !isset($config_var[1]))
111                 $context['config_vars'][] = $config_var;
112             else
113             {
114                 $varname = $config_var[0];
115                 global $$varname;
116 
117                 // Set the subtext in case it's part of the label.
118                 // @todo Temporary. Preventing divs inside label tags.
119                 $divPos = strpos($config_var[1], '<div');
120                 $subtext = '';
121 
122                 if ($divPos !== false)
123                 {
124                     $subtext = preg_replace('~</?div[^>]*>~', '', substr($config_var[1], $divPos));
125                     $config_var[1] = substr($config_var[1], 0, $divPos);
126                 }
127 
128                 $context['config_vars'][$config_var[0]] = array(
129                     'label' => $config_var[1],
130                     'help' => isset($config_var[5]) ? $config_var[5] : '',
131                     'type' => $config_var[3],
132                     'size' => empty($config_var[4]) ? 0 : $config_var[4],
133                     'data' => isset($config_var[4]) && is_array($config_var[4]) && $config_var[3] != 'select' ? $config_var[4] : array(),
134                     'name' => $config_var[0],
135                     'value' => $config_var[2] == 'file' ? (in_array($varname, $defines) ? constant(strtoupper($varname)) : htmlspecialchars($$varname, ENT_COMPAT, 'UTF-8')) : (isset($modSettings[$config_var[0]]) ? htmlspecialchars($modSettings[$config_var[0]], ENT_COMPAT, 'UTF-8') : (in_array($config_var[3], array('int', 'float')) ? 0 : '')),
136                     'disabled' => !empty($context['settings_not_writable']) || !empty($config_var['disabled']),
137                     'invalid' => false,
138                     'subtext' => !empty($config_var['subtext']) ? $config_var['subtext'] : $subtext,
139                     'javascript' => '',
140                     'preinput' => !empty($config_var['preinput']) ? $config_var['preinput'] : '',
141                     'postinput' => !empty($config_var['postinput']) ? $config_var['postinput'] : '',
142                 );

580     /**
581      * Update the Settings.php file.
582      * Typically this method is used from admin screens, just like this entire class.
583      * They're also available for addons and integrations.
584      *
585      * - updates the Settings.php file with the changes supplied in config_vars.
586      * - expects config_vars to be an associative array, with the keys as the
587      *   variable names in Settings.php, and the values the variable values.
588      * - does not escape or quote values.
589      * - preserves case, formatting, and additional options in file.
590      * - writes nothing if the resulting file would be less than 10 lines
591      *   in length (sanity check for read lock.)
592      * - check for changes to db_last_error and passes those off to a separate handler
593      * - attempts to create a backup file and will use it should the writing of the
594      *   new settings file fail
595      *
596      * @param mixed[] $config_vars
597      */
598     public static function save_file($config_vars)
599     {
600         global $context;
601 
602         // Some older code is trying to updating the db_last_error,
603         // then don't mess around with Settings.php
604         if (count($config_vars) === 1 && isset($config_vars['db_last_error']))
605         {
606             require_once(SUBSDIR . '/Admin.subs.php');
607 
608             updateDbLastError($config_vars['db_last_error']);
609             return;
610         }
611 
612         // When was Settings.php last changed?
613         $last_settings_change = filemtime(BOARDDIR . '/Settings.php');
614 
615         // Load the settings file.
616         $settingsArray = trim(file_get_contents(BOARDDIR . '/Settings.php'));
617 
618         // Break it up based on \r or \n, and then clean out extra characters.
619         if (strpos($settingsArray, "\n") !== false)
620             $settingsArray = explode("\n", $settingsArray);
621         elseif (strpos($settingsArray, "\r") !== false)
622             $settingsArray = explode("\r", $settingsArray);
623         else
624             return;
625 
626         // Presumably, the file has to have stuff in it for this function to be called :P.
627         if (count($settingsArray) < 10)
628             return;
629 
630         // remove any /r's that made there way in here
631         foreach ($settingsArray as $k => $dummy)
632             $settingsArray[$k] = strtr($dummy, array("\r" => '')) . "\n";
633 
634         // go line by line and see whats changing
635         for ($i = 0, $n = count($settingsArray); $i < $n; $i++)
636         {
637             // Don't trim or bother with it if it's not a variable.
638             if (substr($settingsArray[$i], 0, 1) != '$')
639                 continue;
640 
641             $settingsArray[$i] = trim($settingsArray[$i]) . "\n";
642 
643             // Look through the variables to set....
644             foreach ($config_vars as $var => $val)
645             {
646                 // be sure someone is not updating db_last_error this with a group
647                 if ($var === 'db_last_error')
648                 {
649                     updateDbLastError($val);
650                     unset($config_vars[$var]);
651                 }
652                 elseif (strncasecmp($settingsArray[$i], '$' . $var, 1 + strlen($var)) == 0)
653                 {
654                     $comment = strstr(substr($settingsArray[$i], strpos($settingsArray[$i], ';')), '#');
655                     $settingsArray[$i] = '$' . $var . ' = ' . $val . ';' . ($comment == '' ? '' : "\t\t" . rtrim($comment)) . "\n";
656 
657                     // This one's been 'used', so to speak.
658                     unset($config_vars[$var]);
659                 }
660             }
661 
662             // End of the file ... maybe
663             if (substr(trim($settingsArray[$i]), 0, 2) == '?' . '>')
664                 $end = $i;
665         }
666 
667         // This should never happen, but apparently it is happening.
668         if (empty($end) || $end < 10)
669             $end = count($settingsArray) - 1;
670 
671         // Still more variables to go?  Then lets add them at the end.
672         if (!empty($config_vars))
673         {
674             if (trim($settingsArray[$end]) == '?' . '>')
675                 $settingsArray[$end++] = '';
676             else
677                 $end++;
678 
679             // Add in any newly defined vars that were passed
680             foreach ($config_vars as $var => $val)
681                 $settingsArray[$end++] = '$' . $var . ' = ' . $val . ';' . "\n";
682         }
683         else
684             $settingsArray[$end] = trim($settingsArray[$end]);
685 
686         // Sanity error checking: the file needs to be at least 12 lines.
687         if (count($settingsArray) < 12)
688             return;
689 
690         // Try to avoid a few pitfalls:
691         //  - like a possible race condition,
692         //  - or a failure to write at low diskspace
693         //
694         // Check before you act: if cache is enabled, we can do a simple write test
695         // to validate that we even write things on this filesystem.
696         if ((!defined('CACHEDIR') || !file_exists(CACHEDIR)) && file_exists(BOARDDIR . '/cache'))
697             $tmp_cache = BOARDDIR . '/cache';
698         else
699             $tmp_cache = CACHEDIR;
700 
701         $test_fp = @fopen($tmp_cache . '/settings_update.tmp', "w+");
702         if ($test_fp)
703         {
704             fclose($test_fp);
705             $written_bytes = file_put_contents($tmp_cache . '/settings_update.tmp', 'test', LOCK_EX);
706             @unlink($tmp_cache . '/settings_update.tmp');
707 
708             if ($written_bytes !== 4)
709             {
710                 // Oops. Low disk space, perhaps. Don't mess with Settings.php then.
711                 // No means no. :P
712                 return;
713             }
714         }
715 
716         // Protect me from what I want! :P
717         clearstatcache();
718         if (filemtime(BOARDDIR . '/Settings.php') === $last_settings_change)
719         {
720             // save the old before we do anything
721             $settings_backup_fail = !@is_writable(BOARDDIR . '/Settings_bak.php') || !@copy(BOARDDIR . '/Settings.php', BOARDDIR . '/Settings_bak.php');
722             $settings_backup_fail = !$settings_backup_fail ? (!file_exists(BOARDDIR . '/Settings_bak.php') || filesize(BOARDDIR . '/Settings_bak.php') === 0) : $settings_backup_fail;
723 
724             // write out the new
725             $write_settings = implode('', $settingsArray);
726             $written_bytes = file_put_contents(BOARDDIR . '/Settings.php', $write_settings, LOCK_EX);
727 
728             // survey says ...
729             if ($written_bytes !== strlen($write_settings) && !$settings_backup_fail)
730             {
731                 // Well this is not good at all, lets see if we can save this
732                 $context['settings_message'] = 'settings_error';
733 
734                 if (file_exists(BOARDDIR . '/Settings_bak.php'))
735                     @copy(BOARDDIR . '/Settings_bak.php', BOARDDIR . '/Settings.php');
