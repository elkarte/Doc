123     /**
124      * callback function for usort used to sort the results.
125      * the order of sorting is: large words, small words, large words that
126      * are excluded from the search, small words that are excluded.
127      *
128      * @param string $a Word A
129      * @param string $b Word B
130      * @return int
131      */
132     public function searchSort($a, $b)
133     {
134         global $excludedWords;
135 
136         $x = strlen($a) - (in_array($a, $excludedWords) ? 1000 : 0);
137         $y = strlen($b) - (in_array($b, $excludedWords) ? 1000 : 0);
138 
139         return $x < $y ? 1 : ($x > $y ? -1 : 0);
140     }

160     /**
161      * This has it's own custom search.
162      *
163      * @param mixed[] $search_params
164      * @param mixed[] $search_words
165      * @param string[] $excluded_words
166      * @param int[] $participants
167      * @param string[] $search_results
168      */
169     public function searchQuery($search_params, $search_words, $excluded_words, &$participants, &$search_results)
170     {
171         global $user_info, $context, $modSettings;
172 
173         // Only request the results if they haven't been cached yet.
174         if (($cached_results = cache_get_data('searchql_results_' . md5($user_info['query_see_board'] . '_' . $context['params']))) === null)
175         {
176             // Create an instance of the sphinx client and set a few options.
177             $mySphinx = @mysql_connect(($modSettings['sphinx_searchd_server'] === 'localhost' ? '127.0.0.1' : $modSettings['sphinx_searchd_server']) . ':' . (int) $modSettings['sphinxql_searchd_port']);
178 
179             // Compile different options for our query
180             $query = 'SELECT *' . (empty($search_params['topic']) ? ', COUNT(*) num' : '') . ', (weight() + (relevance/1000)) rank FROM elkarte_index';
181 
182             // Construct the (binary mode & |) query.
183             $where_match = $this->_constructQuery($search_params['search']);
184 
185             // Nothing to search, return zero results
186             if (trim($where_match) === '')
187                 return 0;
188 
189             if ($search_params['subject_only'])
190                 $where_match = '@subject ' . $where_match;
191 
192             $query .= ' WHERE MATCH(\'' . $where_match . '\')';
193 
194             // Set the limits based on the search parameters.
195             $extra_where = array();
196             if (!empty($search_params['min_msg_id']) || !empty($search_params['max_msg_id']))
197                 $extra_where[] = 'id >= ' . $search_params['min_msg_id'] . ' AND id <= ' . (empty($search_params['max_msg_id']) ? (int) $modSettings['maxMsgID'] : $search_params['max_msg_id']);
198             if (!empty($search_params['topic']))
199                 $extra_where[] = 'id_topic = ' . (int) $search_params['topic'];
200             if (!empty($search_params['brd']))
201                 $extra_where[] = 'id_board IN (' . implode(',', $search_params['brd']) . ')';
202             if (!empty($search_params['memberlist']))
203                 $extra_where[] = 'id_member IN (' . implode(',', $search_params['memberlist']) . ')';
204             if (!empty($extra_where))
205                 $query .= ' AND ' . implode(' AND ', $extra_where);
206 
207             // Put together a sort string; besides the main column sort (relevance, id_topic, or num_replies)
208             $search_params['sort_dir'] = strtoupper($search_params['sort_dir']);
209             $sphinx_sort = $search_params['sort'] === 'id_msg' ? 'id_topic' : $search_params['sort'];
210 
211             // Add secondary sorting based on relevance value (if not the main sort method) and age
212             $sphinx_sort .= ' ' . $search_params['sort_dir'] . ($search_params['sort'] === 'relevance' ? '' : ', relevance DESC') . ', poster_time DESC';
213 
214             // Replace relevance with the returned rank value, rank uses sphinx weight + our own computed field weight relevance
215             $sphinx_sort = str_replace('relevance ', 'rank ', $sphinx_sort);
216 
217             // Grouping by topic id makes it return only one result per topic, so don't set that for in-topic searches
218             if (empty($search_params['topic']))
219                 $query .= ' GROUP BY id_topic WITHIN GROUP ORDER BY ' . $sphinx_sort;
220 
221             $query .= ' ORDER BY ' . $sphinx_sort;
222             $query .= ' LIMIT 0,' . (int) $modSettings['sphinx_max_results'];
223 
224             // Set any options needed, like field weights
225             $query .= ' OPTION field_weights=(subject=' . (!empty($modSettings['search_weight_subject']) ? $modSettings['search_weight_subject'] * 200 : 1000) . ',body=1000)';
226 
227             // Execute the search query.
228             $request = mysql_query($query, $mySphinx);
229 
230             // Can a connection to the daemon be made?
231             if ($request === false)
232             {
233                 // Just log the error.
234                 if (mysql_error($mySphinx))
235                     log_error(mysql_error($mySphinx));
236 
237                 fatal_lang_error('error_no_search_daemon');
238             }
239 
240             // Get the relevant information from the search results.
241             $cached_results = array(
242                 'num_results' => 0,
243                 'matches' => array(),
244             );
245 
246             if (mysql_num_rows($request) != 0)
247             {
248                 while ($match = mysql_fetch_assoc($request))
249                 {
250                     $cached_results['matches'][$match['id']] = array(
251                         'id' => $match['id_topic'],
252                         'relevance' => round($match['relevance'] / 10000, 1) . '%',
253                         'num_matches' => empty($search_params['topic']) ? $match['num'] : 0,
254                         'matches' => array(),
255                     );
256                 }
257             }
258             mysql_free_result($request);
259             mysql_close($mySphinx);
260 
261             $cached_results['num_results'] = count($cached_results['matches']);
262 
263             // Store the search results in the cache.
264             cache_put_data('searchql_results_' . md5($user_info['query_see_board'] . '_' . $context['params']), $cached_results, 600);
265         }
266 
267         $participants = array();
268         foreach (array_slice(array_keys($cached_results['matches']), $_REQUEST['start'], $modSettings['search_results_per_page']) as $msgID)
269         {
270             $context['topics'][$msgID] = $cached_results['matches'][$msgID];
271             $participants[$cached_results['matches'][$msgID]['id']] = false;
272         }
273 
274         // Sentences need to be broken up in words for proper highlighting.
275         $search_results = array();
276         foreach ($search_words as $orIndex => $words)
277             $search_results = array_merge($search_results, $search_words[$orIndex]['subject_words']);
278 
279         return $cached_results['num_results'];
280     }

282     /**
283      * Constructs a binary mode query to pass back to sphinx
284      *
285      * @param string $string The user entered query to construct with
286      * @return string A binary mode query
287      */
288     private function _constructQuery($string)
289     {
290         $keywords = array('include' => array(), 'exclude' => array());
291 
292         // Split our search string and return an empty string if no matches
293         if (!preg_match_all('~ (-?)("[^"]+"|[^" ]+)~', ' ' . $string, $tokens, PREG_SET_ORDER))
294             return '';
295 
296         // First we split our string into included and excluded words and phrases
297         $or_part = false;
298         foreach ($tokens as $token)
299         {
300             // Strip the quotes off of a phrase
301             if ($token[2][0] == '"')
302             {
303                 $token[2] = substr($token[2], 1, -1);
304                 $phrase = true;
305             }
306             else
307                 $phrase = false;
308 
309             // Prepare this token
310             $cleanWords = $this->_cleanString($token[2]);
311 
312             // Explode the cleanWords again incase the cleaning puts more spaces into it
313             $addWords = $phrase ? array('"' . $cleanWords . '"') : preg_split('~ ~u', $cleanWords, null, PREG_SPLIT_NO_EMPTY);
314 
315             // Excluding this word?
316             if ($token[1] == '-')
317                 $keywords['exclude'] = array_merge($keywords['exclude'], $addWords);
318             // OR'd keywords (we only do this if we have something to OR with)
319             elseif (($token[2] == 'OR' || $token[2] == '|') && count($keywords['include']))
320             {
321                 $last = array_pop($keywords['include']);
322                 if (!is_array($last))
323                     $last = array($last);
324                 $keywords['include'][] = $last;
325                 $or_part = true;
326                 continue;
327             }
328             // AND is implied in a Sphinx Search
329             elseif ($token[2] == 'AND' || $token[2] == '&')
330                 continue;
331             // If this part of the query ended up being blank, skip it
332             elseif (trim($cleanWords) == '')
333                 continue;
334             // Must be something they want to search for!
335             else
336             {
337                 // If this was part of an OR branch, add it to the proper section
338                 if ($or_part)
339                     $keywords['include'][count($keywords['include']) - 1] = array_merge($keywords['include'][count($keywords['include']) - 1], $addWords);
340                 else
341                     $keywords['include'] = array_merge($keywords['include'], $addWords);
342             }
343 
344             // Start fresh on this...
345             $or_part = false;
346         }
347 
348         // Let's make sure they're not canceling each other out
349         if (!count(array_diff($keywords['include'], $keywords['exclude'])))
350             return '';
351 
352         // Now we compile our arrays into a valid search string
353         $query_parts = array();
354         foreach ($keywords['include'] as $keyword)
355             $query_parts[] = is_array($keyword) ? '(' . implode(' | ', $keyword) . ')' : $keyword;
356 
357         foreach ($keywords['exclude'] as $keyword)
358             $query_parts[] = '-' . $keyword;
359 
360         return implode(' ', $query_parts);
361     }

363     /**
364      * Cleans a string of everything but alphanumeric characters
365      *
366      * @param string $string A string to clean
367      * @return string A cleaned up string
368      */
369     private function _cleanString($string)
370     {
371         // Decode the entities first
372         $string = html_entity_decode($string, ENT_QUOTES, 'UTF-8');
373 
374         // Lowercase string
375         $string = Util::strtolower($string);
376 
377         // Fix numbers so they search easier (phone numbers, SSN, dates, etc)
378         $string = preg_replace('~([[:digit:]]+)\pP+(?=[[:digit:]])~u', '', $string);
379 
380         // Last but not least, strip everything out that's not alphanumeric
381         $string = preg_replace('~[^\pL\pN]+~u', ' ', $string);
382 
383         return $string;
384     }
