251 /**
252  * Recounts all approved messages
253  *
254  * @package Maintenance
255  * @param int $start
256  * @param int $increment
257  */
258 function recountApprovedMessages($start, $increment)
259 {
260     $db = database();
261 
262     // Recount approved messages
263     $request = $db->query('', '
264         SELECT /*!40001 SQL_NO_CACHE */ t.id_topic, MAX(t.num_replies) AS num_replies,
265             CASE WHEN COUNT(ma.id_msg) >= 1 THEN COUNT(ma.id_msg) - 1 ELSE 0 END AS real_num_replies
266         FROM {db_prefix}topics AS t
267             LEFT JOIN {db_prefix}messages AS ma ON (ma.id_topic = t.id_topic AND ma.approved = {int:is_approved})
268         WHERE t.id_topic > {int:start}
269             AND t.id_topic <= {int:max_id}
270         GROUP BY t.id_topic
271         HAVING CASE WHEN COUNT(ma.id_msg) >= 1 THEN COUNT(ma.id_msg) - 1 ELSE 0 END != MAX(t.num_replies)',
272         array(
273             'is_approved' => 1,
274             'start' => $start,
275             'max_id' => $start + $increment,
276         )
277     );
278     while ($row = $db->fetch_assoc($request))
279         $db->query('', '
280             UPDATE {db_prefix}topics
281             SET num_replies = {int:num_replies}
282             WHERE id_topic = {int:id_topic}',
283             array(
284                 'num_replies' => $row['real_num_replies'],
285                 'id_topic' => $row['id_topic'],
286             )
287         );
288     $db->free_result($request);
289 }

360 /**
361  * Recalculates the boards table's counter
362  *
363  * @package Maintenance
364  * @param string $type - can be 'posts', 'topic', 'unapproved_posts', 'unapproved_topics'
365  * @param int $start
366  * @param int $increment
367  */
368 function updateBoardsCounter($type, $start, $increment)
369 {
370     $db = database();
371 
372     switch ($type)
373     {
374     case 'posts':
375         $request = $db->query('', '
376             SELECT /*!40001 SQL_NO_CACHE */ m.id_board, COUNT(*) AS real_num_posts
377             FROM {db_prefix}messages AS m
378             WHERE m.id_topic > {int:id_topic_min}
379                 AND m.id_topic <= {int:id_topic_max}
380                 AND m.approved = {int:is_approved}
381             GROUP BY m.id_board',
382             array(
383                 'id_topic_min' => $start,
384                 'id_topic_max' => $start + $increment,
385                 'is_approved' => 1,
386             )
387         );
388         while ($row = $db->fetch_assoc($request))
389             $db->query('', '
390                 UPDATE {db_prefix}boards
391                 SET num_posts = num_posts + {int:real_num_posts}
392                 WHERE id_board = {int:id_board}',
393                 array(
394                     'id_board' => $row['id_board'],
395                     'real_num_posts' => $row['real_num_posts'],
396                 )
397             );
398         $db->free_result($request);
399         break;
400 
401     case 'topics':
402         $request = $db->query('', '
403             SELECT /*!40001 SQL_NO_CACHE */ t.id_board, COUNT(*) AS real_num_topics
404             FROM {db_prefix}topics AS t
405             WHERE t.approved = {int:is_approved}
406                 AND t.id_topic > {int:id_topic_min}
407                 AND t.id_topic <= {int:id_topic_max}
408             GROUP BY t.id_board',
409             array(
410                 'is_approved' => 1,
411                 'id_topic_min' => $start,
412                 'id_topic_max' => $start + $increment,
413             )
414         );
415         while ($row = $db->fetch_assoc($request))
416             $db->query('', '
417                 UPDATE {db_prefix}boards
418                 SET num_topics = num_topics + {int:real_num_topics}
419                 WHERE id_board = {int:id_board}',
420                 array(
421                     'id_board' => $row['id_board'],
422                     'real_num_topics' => $row['real_num_topics'],
423                 )
424             );
425         $db->free_result($request);
426         break;
427 
428     case 'unapproved_posts':
429         $request = $db->query('', '
430             SELECT /*!40001 SQL_NO_CACHE */ m.id_board, COUNT(*) AS real_unapproved_posts
431             FROM {db_prefix}messages AS m
432             WHERE m.id_topic > {int:id_topic_min}
433                 AND m.id_topic <= {int:id_topic_max}
434                 AND m.approved = {int:is_approved}
435             GROUP BY m.id_board',
436             array(
437                 'id_topic_min' => $start,
438                 'id_topic_max' => $start + $increment,
439                 'is_approved' => 0,
440             )
441         );
442         while ($row = $db->fetch_assoc($request))
443             $db->query('', '
444                 UPDATE {db_prefix}boards
445                 SET unapproved_posts = unapproved_posts + {int:unapproved_posts}
446                 WHERE id_board = {int:id_board}',
447                 array(
448                     'id_board' => $row['id_board'],
449                     'unapproved_posts' => $row['real_unapproved_posts'],
450                 )
451             );
452         $db->free_result($request);
453         break;
454 
455     case 'unapproved_topics':
456         $request = $db->query('', '
457             SELECT /*!40001 SQL_NO_CACHE */ t.id_board, COUNT(*) AS real_unapproved_topics
458             FROM {db_prefix}topics AS t
459             WHERE t.approved = {int:is_approved}
460                 AND t.id_topic > {int:id_topic_min}
461                 AND t.id_topic <= {int:id_topic_max}
462             GROUP BY t.id_board',
463             array(
464                 'is_approved' => 0,
465                 'id_topic_min' => $start,
466                 'id_topic_max' => $start + $increment,
467             )
468         );
469         while ($row = $db->fetch_assoc($request))
470             $db->query('', '
471                 UPDATE {db_prefix}boards
472                 SET unapproved_topics = unapproved_topics + {int:real_unapproved_topics}
473                 WHERE id_board = {int:id_board}',
474                 array(
475                     'id_board' => $row['id_board'],
476                     'real_unapproved_topics' => $row['real_unapproved_topics'],
477                 )
478             );
479         $db->free_result($request);
480         break;
481 
482     default:
483         trigger_error('updateBoardsCounter(): Invalid counter type \'' . $type . '\'', E_USER_NOTICE);
484     }
485 }

568 /**
569  * Updates the latest message of each board.
570  *
571  * @package Maintenance
572  */
573 function updateBoardsLastMessage()
574 {
575     $db = database();
576 
577     // Update the latest message of each board.
578     $request = $db->query('', '
579         SELECT m.id_board, MAX(m.id_msg) AS local_last_msg
580         FROM {db_prefix}messages AS m
581         WHERE m.approved = {int:is_approved}
582         GROUP BY m.id_board',
583         array(
584             'is_approved' => 1,
585         )
586     );
587     $realBoardCounts = array();
588     while ($row = $db->fetch_assoc($request))
589         $realBoardCounts[$row['id_board']] = $row['local_last_msg'];
590     $db->free_result($request);
591 
592     $request = $db->query('', '
593         SELECT /*!40001 SQL_NO_CACHE */ id_board, id_parent, id_last_msg, child_level, id_msg_updated
594         FROM {db_prefix}boards',
595         array(
596         )
597     );
598     $resort_me = array();
599     while ($row = $db->fetch_assoc($request))
600     {
601         $row['local_last_msg'] = isset($realBoardCounts[$row['id_board']]) ? $realBoardCounts[$row['id_board']] : 0;
602         $resort_me[$row['child_level']][] = $row;
603     }
604     $db->free_result($request);
605 
606     krsort($resort_me);
607 
608     $lastModifiedMsg = array();
609     foreach ($resort_me as $rows)
610         foreach ($rows as $row)
611         {
612             // The latest message is the latest of the current board and its children.
613             if (isset($lastModifiedMsg[$row['id_board']]))
614                 $curLastModifiedMsg = max($row['local_last_msg'], $lastModifiedMsg[$row['id_board']]);
615             else
616                 $curLastModifiedMsg = $row['local_last_msg'];
617                 // If what is and what should be the latest message differ, an update is necessary.
618             if ($row['local_last_msg'] != $row['id_last_msg'] || $curLastModifiedMsg != $row['id_msg_updated'])
619                 $db->query('', '
620                     UPDATE {db_prefix}boards
621                     SET id_last_msg = {int:id_last_msg}, id_msg_updated = {int:id_msg_updated}
622                     WHERE id_board = {int:id_board}',
623                     array(
624                         'id_last_msg' => $row['local_last_msg'],
625                         'id_msg_updated' => $curLastModifiedMsg,
626                         'id_board' => $row['id_board'],
627                     )
628                 );
629 
630             // Parent boards inherit the latest modified message of their children.
631             if (isset($lastModifiedMsg[$row['id_parent']]))
632                 $lastModifiedMsg[$row['id_parent']] = max($row['local_last_msg'], $lastModifiedMsg[$row['id_parent']]);
633             else
634                 $lastModifiedMsg[$row['id_parent']] = $row['local_last_msg'];
635         }
636 }

669 function getTopicsToMove($id_board)
670 {
671     $db = database();
672

674
