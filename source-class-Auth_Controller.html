 24 /**
 25  * Auth_Controller class, deals with logging in and out members,
 26  * and the validation of them
 27  *
 28  * @pachage Authorization
 29  */
 30 class Auth_Controller extends Action_Controller
 31 {
 32     /**
 33      * Entry point in Auth controller
 34      *
 35      * - (well no, not really. We route directly to the rest.)
 36      *
 37      * @see Action_Controller::action_index()
 38      */
 39     public function action_index()
 40     {
 41         // What can we do? login page!
 42         $this->action_login();
 43     }
 44 
 45     /**
 46      * Ask them for their login information.
 47      *
 48      * What it does:
 49      *  - shows a page for the user to type in their username and password.
 50      *  - It caches the referring URL in $_SESSION['login_url'].
 51      *  - It is accessed from ?action=login.
 52      *  @uses Login template and language file with the login sub-template.
 53      */
 54     public function action_login()
 55     {
 56         global $txt, $context, $scripturl, $user_info;
 57 
 58         // You are already logged in, go take a tour of the boards
 59         if (!empty($user_info['id']))
 60             redirectexit();
 61 
 62         // Load the Login template/language file.
 63         loadLanguage('Login');
 64         loadTemplate('Login');
 65         $context['sub_template'] = 'login';
 66 
 67         // Get the template ready.... not really much else to do.
 68         $context['page_title'] = $txt['login'];
 69         $context['default_username'] = &$_REQUEST['u'];
 70         $context['using_openid'] = isset($_GET['openid']);
 71         $context['default_password'] = '';
 72         $context['never_expire'] = false;
 73 
 74         // Add the login chain to the link tree.
 75         $context['linktree'][] = array(
 76             'url' => $scripturl . '?action=login',
 77             'name' => $txt['login'],
 78         );
 79 
 80         // Set the login URL - will be used when the login process is done (but careful not to send us to an attachment).
 81         if (isset($_SESSION['old_url']) && strpos($_SESSION['old_url'], 'dlattach') === false && preg_match('~(board|topic)[=,]~', $_SESSION['old_url']) != 0)
 82             $_SESSION['login_url'] = $_SESSION['old_url'];
 83         else
 84             unset($_SESSION['login_url']);
 85 
 86         // Create a one time token.
 87         createToken('login');
 88     }
 89 
 90     /**
 91      * Actually logs you in.
 92      *
 93      * What it does:
 94      * - checks credentials and checks that login was successful.
 95      * - it employs protection against a specific IP or user trying to brute force
 96      *   a login to an account.
 97      * - upgrades password encryption on login, if necessary.
 98      * - after successful login, redirects you to $_SESSION['login_url'].
 99      * - accessed from ?action=login2, by forms.
100      *
101      * On error, uses the same templates action_login() uses.
102      */
103     public function action_login2()
104     {
105         global $txt, $scripturl, $user_info, $user_settings, $modSettings, $context, $sc;
106 
107         // Load cookie authentication and all stuff.
108         require_once(SUBSDIR . '/Auth.subs.php');
109 
110         // Beyond this point you are assumed to be a guest trying to login.
111         if (!$user_info['is_guest'])
112             redirectexit();
113 
114         // Are you guessing with a script?
115         checkSession('post');
116         validateToken('login');
117         spamProtection('login');
118 
119         // Set the login_url if it's not already set (but careful not to send us to an attachment).
120         if ((empty($_SESSION['login_url']) && isset($_SESSION['old_url']) && strpos($_SESSION['old_url'], 'dlattach') === false && preg_match('~(board|topic)[=,]~', $_SESSION['old_url']) != 0) || (isset($_GET['quicklogin']) && isset($_SESSION['old_url']) && strpos($_SESSION['old_url'], 'login') === false))
121             $_SESSION['login_url'] = $_SESSION['old_url'];
122 
123         // Been guessing a lot, haven't we?
124         if (isset($_SESSION['failed_login']) && $_SESSION['failed_login'] >= $modSettings['failed_login_threshold'] * 3)
125             fatal_lang_error('login_threshold_fail', 'critical');
126 
127         // Set up the cookie length.  (if it's invalid, just fall through and use the default.)
128         if (isset($_POST['cookieneverexp']) || (!empty($_POST['cookielength']) && $_POST['cookielength'] == -1))
129             $modSettings['cookieTime'] = 3153600;
130         elseif (!empty($_POST['cookielength']) && ($_POST['cookielength'] >= 1 || $_POST['cookielength'] <= 525600))
131             $modSettings['cookieTime'] = (int) $_POST['cookielength'];
132 
133         loadLanguage('Login');
134 
135         // Load the template stuff
136         loadTemplate('Login');
137         $context['sub_template'] = 'login';
138 
139         // Set up the default/fallback stuff.
140         $context['default_username'] = isset($_POST['user']) ? preg_replace('~&amp;#(\\d{1,7}|x[0-9a-fA-F]{1,6});~', '&#\\1;', htmlspecialchars($_POST['user'], ENT_COMPAT, 'UTF-8')) : '';
141         $context['default_password'] = '';
142         $context['never_expire'] = $modSettings['cookieTime'] == 525600 || $modSettings['cookieTime'] == 3153600;
143         $context['login_errors'] = array($txt['error_occurred']);
144         $context['page_title'] = $txt['login'];
145 
146         // Add the login chain to the link tree.
147         $context['linktree'][] = array(
148             'url' => $scripturl . '?action=login',
149             'name' => $txt['login'],
150         );
151 
152         // This is an OpenID login. Let's validate...
153         if (!empty($_POST['openid_identifier']) && !empty($modSettings['enableOpenID']))
154         {
155             require_once(SUBSDIR . '/OpenID.subs.php');
156             $open_id = new OpenID();
157 
158             if (($open_id->validate($_POST['openid_identifier'])) !== 'no_data')
159                 return $open_id;
160             else
161             {
162                 $context['login_errors'] = array($txt['openid_not_found']);
163                 return;
164             }
165         }
166 
167         // You forgot to type your username, dummy!
168         if (!isset($_POST['user']) || $_POST['user'] == '')
169         {
170             $context['login_errors'] = array($txt['need_username']);
171             return;
172         }
173 
174         // No one needs a username that long, plus we only support 80 chars in the db
175         if (Util::strlen($_POST['user']) > 80)
176             $_POST['user'] = Util::substr($_POST['user'], 0, 80);
177 
178         // Can't use a password > 64 characters sorry, to long and only good for a DoS attack
179         // Plus we expect a 64 character one from SHA-256
180         if ((isset($_POST['passwrd']) && strlen($_POST['passwrd']) > 64) || (isset($_POST['hash_passwrd']) && strlen($_POST['hash_passwrd']) > 64))
181         {
182             $context['login_errors'] = array($txt['improper_password']);
183             return;
184         }
185 
186         // Hmm... maybe 'admin' will login with no password. Uhh... NO!
187         if ((!isset($_POST['passwrd']) || $_POST['passwrd'] == '') && (!isset($_POST['hash_passwrd']) || strlen($_POST['hash_passwrd']) != 64))
188         {
189             $context['login_errors'] = array($txt['no_password']);
190             return;
191         }
192 
193         // No funky symbols either.
194         if (preg_match('~[<>&"\'=\\\]~', preg_replace('~(&#(\\d{1,7}|x[0-9a-fA-F]{1,6});)~', '', $_POST['user'])) != 0)
195         {
196             $context['login_errors'] = array($txt['error_invalid_characters_username']);
197             return;
198         }
199 
200         // Are we using any sort of integration to validate the login?
201         if (in_array('retry', call_integration_hook('integrate_validate_login', array($_POST['user'], isset($_POST['hash_passwrd']) && strlen($_POST['hash_passwrd']) == 40 ? $_POST['hash_passwrd'] : null, $modSettings['cookieTime'])), true))
202         {
203             $context['login_errors'] = array($txt['login_hash_error']);
204             $context['disable_login_hashing'] = true;
205             return;
206         }
207 
208         // Find them... if we can
209         $user_settings = loadExistingMember($_POST['user']);
210 
211         // Let them try again, it didn't match anything...
212         if (empty($user_settings))
213         {
214             $context['login_errors'] = array($txt['username_no_exist']);
215             return;
216         }
217 
218         // Figure out if the password is using Elk's encryption - if what they typed is right.
219         if (isset($_POST['hash_passwrd']) && strlen($_POST['hash_passwrd']) === 64)
220         {
221             // Challenge what was passed
222             $valid_password = validateLoginPassword($_POST['hash_passwrd'], $user_settings['passwd']);
223 
224             // Let them in
225             if ($valid_password)
226             {
227                 $sha_passwd = $_POST['hash_passwrd'];
228                 $valid_password = true;
229             }
230             // Maybe is an old SHA-1 and needs upgrading if the db string is an actual 40 hexchar SHA-1
231             elseif (preg_match('/^[0-9a-f]{40}$/i', $user_settings['passwd']) && isset($_POST['old_hash_passwrd']) && $_POST['old_hash_passwrd'] === hash('sha1', $user_settings['passwd'] . $sc))
232             {
233                 // Old password passed, turn off hashing and ask for it again so we can update the db to something more secure.
234                 $context['login_errors'] = array($txt['login_hash_error']);
235                 $context['disable_login_hashing'] = true;
236                 unset($user_settings);
237 
238                 return;
239             }
240             // Bad password entered
241             else
242             {
243                 // Don't allow this!
244                 validatePasswordFlood($user_settings['id_member'], $user_settings['passwd_flood']);
245 
246                 $_SESSION['failed_login'] = isset($_SESSION['failed_login']) ? ($_SESSION['failed_login'] + 1) : 1;
247 
248                 // To many tries, maybe they need a reminder
249                 if ($_SESSION['failed_login'] >= $modSettings['failed_login_threshold'])
250                     redirectexit('action=reminder');
251                 else
252                 {
253                     log_error($txt['incorrect_password'] . ' - <span class="remove">' . $user_settings['member_name'] . '</span>', 'user');
254 
255                     // Wrong password, lets enable plain text responses in case form hashing is causing problems
256                     $context['disable_login_hashing'] = true;
257                     $context['login_errors'] = array($txt['incorrect_password']);
258                     unset($user_settings);
259 
260                     return;
261                 }
262             }
263         }
264         // Plain text password, no JS or hashing has been turned off
265         else
266         {
267             // validateLoginPassword will hash this like the form normally would and check its valid
268             $sha_passwd = $_POST['passwrd'];
269             $valid_password = validateLoginPassword($sha_passwd, $user_settings['passwd'], $user_settings['member_name']);
270         }
271 
272         // Bad password!  Thought you could fool the database?!
273         if ($valid_password === false)
274         {
275             // Let's be cautious, no hacking please. thanx.
276             validatePasswordFlood($user_settings['id_member'], $user_settings['passwd_flood']);
277 
278             // Maybe we were too hasty... let's try some other authentication methods.
279             $other_passwords = $this->_other_passwords($user_settings);
280 
281             // Whichever encryption it was using, let's make it use ElkArte's now ;).
282             if (in_array($user_settings['passwd'], $other_passwords))
283             {
284                 $user_settings['passwd'] = validateLoginPassword($sha_passwd, '', '', true);
285                 $user_settings['password_salt'] = substr(md5(mt_rand()), 0, 4);
286 
287                 // Update the password hash and set up the salt.
288                 updateMemberData($user_settings['id_member'], array('passwd' => $user_settings['passwd'], 'password_salt' => $user_settings['password_salt'], 'passwd_flood' => ''));
289             }
290             // Okay, they for sure didn't enter the password!
291             else
292             {
293                 // They've messed up again - keep a count to see if they need a hand.
294                 $_SESSION['failed_login'] = isset($_SESSION['failed_login']) ? ($_SESSION['failed_login'] + 1) : 1;
295 
296                 // Hmm... don't remember it, do you?  Here, try the password reminder ;).
297                 if ($_SESSION['failed_login'] >= $modSettings['failed_login_threshold'])
298                     redirectexit('action=reminder');
299                 // We'll give you another chance...
300                 else
301                 {
302                     // Log an error so we know that it didn't go well in the error log.
303                     log_error($txt['incorrect_password'] . ' - <span class="remove">' . $user_settings['member_name'] . '</span>', 'user');
304 
305                     $context['login_errors'] = array($txt['incorrect_password']);
306                     return;
307                 }
308             }
309         }
310         elseif (!empty($user_settings['passwd_flood']))
311         {
312             // Let's be sure they weren't a little hacker.
313             validatePasswordFlood($user_settings['id_member'], $user_settings['passwd_flood'], true);
314 
315             // If we got here then we can reset the flood counter.
316             updateMemberData($user_settings['id_member'], array('passwd_flood' => ''));
317         }
318 
319         // Correct password, but they've got no salt; fix it!
320         if ($user_settings['password_salt'] == '')
321         {
322             $user_settings['password_salt'] = substr(md5(mt_rand()), 0, 4);
323             updateMemberData($user_settings['id_member'], array('password_salt' => $user_settings['password_salt']));
324         }
325 
326         // Check their activation status.
327         if (!checkActivation())
328             return;
329 
330         doLogin();
331     }

501     /**
502      * Loads other possible password hash / crypts using the post data
503      *
504      * What it does:
505      * - Used when a board is converted to see if the user credentials and a 3rd
506      * party hash satisfy whats in the db passwd field
507      *
508      * @param mixed[] $user_settings
509      */
510     private function _other_passwords($user_settings)
511     {
512         global $modSettings, $sc;
513 
514         // What kind of data are we dealing with
515         $pw_strlen = strlen($user_settings['passwd']);
516 
517         // Start off with none, thats safe
518         $other_passwords = array();
519 
520         // None of the below cases will be used most of the time (because the salt is normally set.)
521         if (!empty($modSettings['enable_password_conversion']) && $user_settings['password_salt'] == '')
522         {
523             // YaBB SE, Discus, MD5 (used a lot), SHA-1 (used some), SMF 1.0.x, IkonBoard, and none at all.
524             $other_passwords[] = crypt($_POST['passwrd'], substr($_POST['passwrd'], 0, 2));
525             $other_passwords[] = crypt($_POST['passwrd'], substr($user_settings['passwd'], 0, 2));
526             $other_passwords[] = md5($_POST['passwrd']);
527             $other_passwords[] = sha1($_POST['passwrd']);
528             $other_passwords[] = md5_hmac($_POST['passwrd'], strtolower($user_settings['member_name']));
529             $other_passwords[] = md5($_POST['passwrd'] . strtolower($user_settings['member_name']));
530             $other_passwords[] = md5(md5($_POST['passwrd']));
531             $other_passwords[] = $_POST['passwrd'];
532 
533             // This one is a strange one... MyPHP, crypt() on the MD5 hash.
534             $other_passwords[] = crypt(md5($_POST['passwrd']), md5($_POST['passwrd']));
535 
536             // SHA-256
537             if ($pw_strlen === 64)
538             {
539                 // Snitz style
540                 $other_passwords[] = bin2hex(hash('sha256', $_POST['passwrd']));
541 
542                 // Normal SHA-256
543                 $other_passwords[] = hash('sha256', $_POST['passwrd']);
544             }
545 
546             // phpBB3 users new hashing.  We now support it as well ;).
547             $other_passwords[] = phpBB3_password_check($_POST['passwrd'], $user_settings['passwd']);
548 
549             // APBoard 2 Login Method.
550             $other_passwords[] = md5(crypt($_POST['passwrd'], 'CRYPT_MD5'));
551         }
552         // The hash should be 40 if it's SHA-1, so we're safe with more here too.
553         elseif (!empty($modSettings['enable_password_conversion']) && $pw_strlen === 32)
554         {
555             // vBulletin 3 style hashing?  Let's welcome them with open arms \o/.
556             $other_passwords[] = md5(md5($_POST['passwrd']) . stripslashes($user_settings['password_salt']));
557 
558             // Hmm.. p'raps it's Invision 2 style?
559             $other_passwords[] = md5(md5($user_settings['password_salt']) . md5($_POST['passwrd']));
560 
561             // Some common md5 ones.
562             $other_passwords[] = md5($user_settings['password_salt'] . $_POST['passwrd']);
563             $other_passwords[] = md5($_POST['passwrd'] . $user_settings['password_salt']);
564         }
565         // The hash is 40 characters, lets try some SHA-1 style auth
566         elseif ($pw_strlen === 40)
567         {
568             // Maybe they are using a hash from before our password upgrade
569             $other_passwords[] = sha1(strtolower($user_settings['member_name']) . un_htmlspecialchars($_POST['passwrd']));
570             $other_passwords[] = sha1($user_settings['passwd'] . $sc);
571 
572             if (!empty($modSettings['enable_password_conversion']))
573             {
574                 // BurningBoard3 style of hashing.
575                 $other_passwords[] = sha1($user_settings['password_salt'] . sha1($user_settings['password_salt'] . sha1($_POST['passwrd'])));
576 
577                 // PunBB 1.4 and later
578                 $other_passwords[] = sha1($user_settings['password_salt'] . sha1($_POST['passwrd']));
579             }
580 
581             // Perhaps we converted from a non UTF-8 db and have a valid password being hashed differently.
582             if (!empty($modSettings['previousCharacterSet']) && $modSettings['previousCharacterSet'] != 'utf8')
583             {
584                 // Try iconv first, for no particular reason.
585                 if (function_exists('iconv'))
586                     $other_passwords['iconv'] = sha1(strtolower(iconv('UTF-8', $modSettings['previousCharacterSet'], $user_settings['member_name'])) . un_htmlspecialchars(iconv('UTF-8', $modSettings['previousCharacterSet'], $_POST['passwrd'])));
587 
588                 // Say it aint so, iconv failed!
589                 if (empty($other_passwords['iconv']) && function_exists('mb_convert_encoding'))
590                     $other_passwords[] = sha1(strtolower(mb_convert_encoding($user_settings['member_name'], 'UTF-8', $modSettings['previousCharacterSet'])) . un_htmlspecialchars(mb_convert_encoding($_POST['passwrd'], 'UTF-8', $modSettings['previousCharacterSet'])));
591             }
592         }
593         // SHA-256 will be 64 characters long, lets check some of these possibilities
594         elseif (!empty($modSettings['enable_password_conversion']) && $pw_strlen === 64)
595         {
596             // PHP-Fusion7
597             $other_passwords[] = hash_hmac('sha256', $_POST['passwrd'], $user_settings['password_salt']);
598 
599             // Plain SHA-256?
600             $other_passwords[] = hash('sha256', $_POST['passwrd'] . $user_settings['password_salt']);
601 
602             // Xenforo?
603             $other_passwords[] = sha1(sha1($_POST['passwrd']) . $user_settings['password_salt']);
604             $other_passwords[] = hash('sha256', (hash('sha256', ($_POST['passwrd']) . $user_settings['password_salt'])));
605         }
606 
607         // ElkArte's sha1 function can give a funny result on Linux (Not our fault!). If we've now got the real one let the old one be valid!
608         if (stripos(PHP_OS, 'win') !== 0)
609         {
610             require_once(SUBSDIR . '/Compat.subs.php');
611             $other_passwords[] = sha1_smf(strtolower($user_settings['member_name']) . un_htmlspecialchars($_POST['passwrd']));
612         }
613 
614         // Allows mods to easily extend the $other_passwords array
615         call_integration_hook('integrate_other_passwords', array(&$other_passwords));
616 
617         return $other_passwords;
618     }
619 }

621 /**
622  * Check activation status of the current user.
623  *
624  * What it does:
625  * - > 10 Banned with activation status as value - 10
626  * - 5 = Awaiting COPPA concent
627  * - 4 = Awaiting Deletion approval
628  * - 3 = Awaiting Admin approval
629  * - 2 = Awaiting reactivation from email change
630  * - 1 = Approved and active
631  * - 0 = Not active
632  *
633  * @package Authorization
634  */
635 function checkActivation()
636 {
637     global $context, $txt, $scripturl, $user_settings, $modSettings;
638 
639     if (!isset($context['login_errors']))
640         $context['login_errors'] = array();
641 
642     // What is the true activation status of this account?
643     $activation_status = $user_settings['is_activated'] > 10 ? $user_settings['is_activated'] - 10 : $user_settings['is_activated'];
644 
645     // Check if the account is activated - COPPA first...
646     if ($activation_status == 5)
647     {
648         $context['login_errors'][] = $txt['coppa_no_concent'] . ' <a href="' . $scripturl . '?action=coppa;member=' . $user_settings['id_member'] . '">' . $txt['coppa_need_more_details'] . '</a>';
649         return false;
650     }
651     // Awaiting approval still?
652     elseif ($activation_status == 3)
653         fatal_lang_error('still_awaiting_approval', 'user');
654     // Awaiting deletion, changed their mind?
655     elseif ($activation_status == 4)
656     {
657         if (isset($_REQUEST['undelete']))
658         {
659             updateMemberData($user_settings['id_member'], array('is_activated' => 1));
660             updateSettings(array('unapprovedMembers' => ($modSettings['unapprovedMembers'] > 0 ? $modSettings['unapprovedMembers'] - 1 : 0)));
661         }
662         else
663         {
664             $context['disable_login_hashing'] = true;
665             $context['login_errors'][] = $txt['awaiting_delete_account'];
666             $context['login_show_undelete'] = true;
667             return false;
668         }
669     }
670     // Standard activation?
671     elseif ($activation_status != 1)
672     {
673         log_error($txt['activate_not_completed1'] . ' - <span class="remove">' . $user_settings['member_name'] . '</span>', false);
674 
675         $context['login_errors'][] = $txt['activate_not_completed1'] . ' <a href="' . $scripturl . '?action=activate;sa=resend;u=' . $user_settings['id_member'] . '">' . $txt['activate_not_completed2'] . '</a>';
676         return false;
677     }
678     return true;
679 }

681 /**
682  * This function performs the logging in.
683  *
684  * What it does:
685  *  - It sets the cookie, it call hooks, updates runtime settings for the user.
686  *
687  * @package Authorization
688  */
689 function doLogin()
690 {
691     global $user_info, $user_settings, $maintenance, $modSettings, $context;
692 
693     // Load authentication stuffs.
694     require_once(SUBSDIR . '/Auth.subs.php');
695 
696     // Call login integration functions.
697     call_integration_hook('integrate_login', array($user_settings['member_name'], isset($_POST['hash_passwrd']) && strlen($_POST['hash_passwrd']) == 64 ? $_POST['hash_passwrd'] : null, $modSettings['cookieTime']));
698 
699     // Get ready to set the cookie...
700     $user_info['id'] = $user_settings['id_member'];
701 
702     // Bam!  Cookie set.  A session too, just in case.
703     setLoginCookie(60 * $modSettings['cookieTime'], $user_settings['id_member'], hash('sha256', ($user_settings['passwd'] . $user_settings['password_salt'])));
704 
705     // Reset the login threshold.
706     if (isset($_SESSION['failed_login']))
707         unset($_SESSION['failed_login']);
708 
709     $user_info['is_guest'] = false;
710     $user_settings['additional_groups'] = explode(',', $user_settings['additional_groups']);
711     $user_info['is_admin'] = $user_settings['id_group'] == 1 || in_array(1, $user_settings['additional_groups']);
712 
713     // Are you banned?
714     is_not_banned(true);
715 
716     // An administrator, set up the login so they don't have to type it again.
717     if ($user_info['is_admin'] && isset($user_settings['openid_uri']) && empty($user_settings['openid_uri']))
718     {
719         // Let's validate if they really want..
720         if (!empty($modSettings['auto_admin_session']) && $modSettings['auto_admin_session'] == 1)
721             $_SESSION['admin_time'] = time();
722 
723         unset($_SESSION['just_registered']);
724     }
725 
726     // Don't stick the language or theme after this point.
727     unset($_SESSION['language'], $_SESSION['id_theme']);
728 
729     // We want to know if this is first login
730     if (isFirstLogin($user_info['id']))
731         $_SESSION['first_login'] = true;
732     else
733         unset($_SESSION['first_login']);
734 
735     // You're one of us: need to know all about you now, IP, stuff.
736     $req = request();
737 
738     // You've logged in, haven't you?
739     updateMemberData($user_info['id'], array('last_login' => time(), 'member_ip' => $user_info['ip'], 'member_ip2' => $req->ban_ip()));
740 
741     // Get rid of the online entry for that old guest....
742     deleteOnline('ip' . $user_info['ip']);

764 function md5_hmac($data, $key)
765 {
766     $key = str_pad(strlen($key) <= 64 ? $key : pack('H*', md5($key)), 64, chr(0x00));

802
