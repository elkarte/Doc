 89     /**
 90      * Check whether the method can be performed by this API.
 91      *
 92      * @param string $methodName
 93      * @param mixed[]|null $query_params
 94      */
 95     public function supportsMethod($methodName, $query_params = null)
 96     {
 97         switch ($methodName)
 98         {
 99             case 'isValid':
100             case 'searchSort':
101             case 'prepareIndexes':
102                 return true;
103             break;
104             case 'searchQuery':
105                 // Search can be performed, but not for 'subject only' query.
106                 return !$query_params['subject_only'];
107             default:
108                 // All other methods, too bad dunno you.
109                 return false;
110         }
111     }

123     /**
124      * Callback function for usort used to sort the results.
125      * the order of sorting is: large words, small words, large words that
126      * are excluded from the search, small words that are excluded.
127      *
128      * @param string $a Word A
129      * @param string $b Word B
130      * @return int
131      */
132     public function searchSort($a, $b)
133     {
134         global $excludedWords;
135 
136         $x = strlen($a) - (in_array($a, $excludedWords) ? 1000 : 0);
137         $y = strlen($b) - (in_array($b, $excludedWords) ? 1000 : 0);
138 
139         return $x < $y ? 1 : ($x > $y ? -1 : 0);
140     }

142     /**
143      * Do we have to do some work with the words we are searching for to prepare them?

145      * @param mixed[] $word
146      * @param mixed[] $wordsSearch
147      * @param string[] $wordsExclude
148      * @param boolean $isExcluded
149      */
150     public function prepareIndexes($word, &$wordsSearch, &$wordsExclude, $isExcluded)
151     {
152         $subwords = text2words($word, null, false);
153 
154         $fulltextWord = count($subwords) === 1 ? $word : '"' . $word . '"';
155         $wordsSearch['indexed_words'][] = $fulltextWord;
156         if ($isExcluded)
157             $wordsExclude[] = $fulltextWord;
158     }

160     /**
161      * This has it's own custom search.
162      *
163      * @param mixed[] $search_params
164      * @param mixed[] $search_words
165      * @param string[] $excluded_words
166      * @param int[] $participants
167      * @param string[] $search_results
168      */
169     public function searchQuery($search_params, $search_words, $excluded_words, &$participants, &$search_results)
170     {
171         global $user_info, $context, $modSettings;
172 
173         // Only request the results if they haven't been cached yet.
174         if (($cached_results = cache_get_data('search_results_' . md5($user_info['query_see_board'] . '_' . $context['params']))) === null)
175         {
176             // The API communicating with the search daemon.
177             require_once(SOURCEDIR . '/sphinxapi.php');
178 
179             // Create an instance of the sphinx client and set a few options.
180             $mySphinx = new SphinxClient();
181             $mySphinx->SetServer($modSettings['sphinx_searchd_server'], (int) $modSettings['sphinx_searchd_port']);
182             $mySphinx->SetLimits(0, (int) $modSettings['sphinx_max_results'], (int) $modSettings['sphinx_max_results']);
183 
184             // Put together a sort string; besides the main column sort (relevance, id_topic, or num_replies),
185             $search_params['sort_dir'] = strtoupper($search_params['sort_dir']);
186             $sphinx_sort = $search_params['sort'] === 'id_msg' ? 'id_topic' : $search_params['sort'];
187 
188             // Add secondary sorting based on relevance value (if not the main sort method) and age
189             $sphinx_sort .= ' ' . $search_params['sort_dir'] . ($search_params['sort'] === 'relevance' ? '' : ', relevance DESC') . ', poster_time DESC';
190 
191             // Include the engines weight values in the group sort
192             $sphinx_sort = str_replace('relevance ', '@weight ' . $search_params['sort_dir'] . ', relevance ', $sphinx_sort);
193 
194             // Grouping by topic id makes it return only one result per topic, so don't set that for in-topic searches
195             if (empty($search_params['topic']) )
196                 $mySphinx->SetGroupBy('id_topic', SPH_GROUPBY_ATTR, $sphinx_sort);
197 
198             // Set up the sort expresssion
199             $mySphinx->SetSortMode(SPH_SORT_EXPR, '(@weight + (relevance / 1000))');
200 
201             // Update the field weights for subject vs body
202             $mySphinx->SetFieldWeights(array('subject' => !empty($modSettings['search_weight_subject']) ? $modSettings['search_weight_subject'] * 200 : 1000, 'body' => 1000));
203 
204             // Set the limits based on the search parameters.
205             if (!empty($search_params['min_msg_id']) || !empty($search_params['max_msg_id']))
206                 $mySphinx->SetIDRange($search_params['min_msg_id'], empty($search_params['max_msg_id']) ? (int) $modSettings['maxMsgID'] : $search_params['max_msg_id']);
207 
208             if (!empty($search_params['topic']))
209                 $mySphinx->SetFilter('id_topic', array((int) $search_params['topic']));
210 
211             if (!empty($search_params['brd']))
212                 $mySphinx->SetFilter('id_board', $search_params['brd']);
213 
214             if (!empty($search_params['memberlist']))
215                 $mySphinx->SetFilter('id_member', $search_params['memberlist']);
216 
217             // Construct the (binary mode & |) query while accounting for excluded words
218             $orResults = array();
219             $inc_words = array();
220             foreach ($search_words as $orIndex => $words)
221             {
222                 $inc_words = array_merge($inc_words, $words['indexed_words']);
223                 $andResult = '';
224                 foreach ($words['indexed_words'] as $sphinxWord)
225                     $andResult .= (in_array($sphinxWord, $excluded_words) ? '-' : '') . $this->_cleanWordSphinx($sphinxWord, $mySphinx) . ' & ';
226                 $orResults[] = substr($andResult, 0, -3);
227             }
228 
229             // If no search terms are left after comparing against excluded words (i.e. "test -test" or "test last -test -last"),
230             // sending that to Sphinx would result in a fatal error
231             if (!count(array_diff($inc_words, $excluded_words)))
232                 // Instead, fail gracefully (return "no results")
233                 return 0;
234 
235             $query = count($orResults) === 1 ? $orResults[0] : '(' . implode(') | (', $orResults) . ')';
236 
237             // Subject only searches need to be specified.
238             if ($search_params['subject_only'])
239                 $query = '@(subject) ' . $query;
240 
241             // Choose an appropriate matching mode
242             $mode = SPH_MATCH_ALL;
243 
244             // Over two words and searching for any (since we build a binary string, this will never get set)
245             if (substr_count($query, ' ') > 1 && (!empty($search_params['searchtype']) && $search_params['searchtype'] == 2))
246                    $mode = SPH_MATCH_ANY;
247             // Binary search?
248             if (preg_match('~[\|\(\)\^\$\?"\/=-]~', $query))
249                 $mode = SPH_MATCH_EXTENDED;
250 
251             // Set the matching mode
252             $mySphinx->SetMatchMode($mode);
253 
254             // Execute the search query.
255             $request = $mySphinx->Query($query, 'elkarte_index');
256 
257             // Can a connection to the daemon be made?
258             if ($request === false)
259             {
260                 // Just log the error.
261                 if ($mySphinx->GetLastError())
262                     log_error($mySphinx->GetLastError());
263 
264                 fatal_lang_error('error_no_search_daemon');
265             }
266 
267             // Get the relevant information from the search results.
268             $cached_results = array(
269                 'matches' => array(),
270                 'num_results' => $request['total'],
271             );
272 
273             if (isset($request['matches']))
274             {
275                 foreach ($request['matches'] as $msgID => $match)
276                     $cached_results['matches'][$msgID] = array(
277                         'id' => $match['attrs']['id_topic'],
278                         'relevance' => round($match['attrs']['@count'] + $match['attrs']['relevance'] / 10000, 1) . '%',
279                         'num_matches' => empty($search_params['topic']) ? $match['attrs']['@count'] : 0,
280                         'matches' => array(),
281                     );
282             }
283 
284             // Store the search results in the cache.
285             cache_put_data('search_results_' . md5($user_info['query_see_board'] . '_' . $context['params']), $cached_results, 600);
286         }
287 
288         $participants = array();
289         foreach (array_slice(array_keys($cached_results['matches']), $_REQUEST['start'], $modSettings['search_results_per_page']) as $msgID)
290         {
291             $context['topics'][$msgID] = $cached_results['matches'][$msgID];
292             $participants[$cached_results['matches'][$msgID]['id']] = false;
293         }
294 
295         // Sentences need to be broken up in words for proper highlighting.
296         $search_results = array();
297         foreach ($search_words as $orIndex => $words)
298             $search_results = array_merge($search_results, $search_words[$orIndex]['subject_words']);
299 
300         return $cached_results['num_results'];
301     }

303     /**
304      * Clean up a search word/phrase/term for Sphinx.
305      *
306      * @param string $sphinx_term
307      * @param SphinxClient $sphinx_client
308      */
309     private function _cleanWordSphinx($sphinx_term, $sphinx_client)
310     {
311         // Multiple quotation marks in a row can cause fatal errors, so handle them
312         $sphinx_term = preg_replace('/""+/', '"', $sphinx_term);
313 
314         // Unmatched (i.e. odd number of) quotation marks also cause fatal errors, so handle them
315         if (substr_count($sphinx_term, '"') % 2)
316 
317         // Using preg_replace since it supports limiting the number of replacements
318         $sphinx_term = preg_replace('/"/', '', $sphinx_term, 1);
319 
320         // Use the Sphinx API's built-in EscapeString function to escape special characters
321         $sphinx_term = $sphinx_client->EscapeString($sphinx_term);
322 
323         // Since it escapes quotation marks and we don't want that, unescape them
324         $sphinx_term = str_replace('\"', '"', $sphinx_term);
325 
326         return $sphinx_term;
327     }
