17 /**
18  * Fetches a list of package servers.
19  *
20  * @package Packages
21  * @param int|null $server
22  * @return array
23  */
24 function fetchPackageServers($server = null)
25 {
26     $db = database();
27 
28     $servers = array();
29 
30     // Load the list of servers.
31     $request = $db->query('', '
32         SELECT id_server, name, url
33         FROM {db_prefix}package_servers' .
34         (!empty($server) ? ' WHERE id_server = {int:current_server}' : ''),
35         array(
36             'current_server' => $server,
37         )
38     );
39     while ($row = $db->fetch_assoc($request))
40     {
41         $servers[] = array(
42             'name' => $row['name'],
43             'url' => $row['url'],
44             'id' => $row['id_server'],
45         );
46     }
47     $db->free_result($request);
48 
49     return $servers;
50 }
