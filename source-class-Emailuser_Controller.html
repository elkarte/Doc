 56      * Uses the Emailuser template, with the main sub template.
 57      * Requires the send_topic permission.
 58      * Redirects back to the first page of the topic when done.
 59      * Is accessed via ?action=emailuser;sa=sendtopic.
 60      */
 61     public function action_sendtopic()
 62     {
 63         global $topic, $txt, $context, $modSettings;
 64 
 65         // Check permissions...
 66         isAllowedTo('send_topic');
 67 
 68         // We need at least a topic... go away if you don't have one.
 69         if (empty($topic))
 70             fatal_lang_error('not_a_topic', false);
 71 
 72         require_once(SUBSDIR . '/Topic.subs.php');
 73         $row = getTopicInfo($topic, 'message');
 74         if (empty($row))
 75             fatal_lang_error('not_a_topic', false);
 76 
 77         // Can't send topic if its unapproved and using post moderation.
 78         if ($modSettings['postmod_active'] && !$row['approved'])
 79             fatal_lang_error('not_approved_topic', false);
 80 
 81         // Censor the subject....
 82         censorText($row['subject']);
 83 
 84         // Sending yet, or just getting prepped?
 85         if (empty($_POST['send']))
 86         {
 87             $context['page_title'] = sprintf($txt['sendtopic_title'], $row['subject']);
 88             $context['start'] = $_REQUEST['start'];
 89             $context['sub_template'] = 'send_topic';
 90 
 91             return;
 92         }
 93 
 94         // Actually send the message...
 95         checkSession();
 96         spamProtection('sendtopic');
 97 
 98         $result = $this->_sendTopic($row);
 99         if ($result !== true)
100             $context['sendtopic_error'] = $result;
101 
102         // Back to the topic!
103         redirectexit('topic=' . $topic . '.0');
104     }

196     /**
197      * Prepares the form data and database data for sending in an email format
198      * Does the actual sending of the email if everthing checks out as OK
199      *
200      * @param mixed[] $row
201      */
202     private function _sendTopic($row)
203     {
204         global $scripturl, $topic, $txt;
205 
206         // This is needed for sendmail().
207         require_once(SUBSDIR . '/Mail.subs.php');
208 
209         // Time to check and clean what was placed in the form
210         require_once(SUBSDIR . '/DataValidator.class.php');
211         $validator = new Data_Validator();
212         $validator->sanitation_rules(array(
213             'y_name' => 'trim',
214             'r_name' => 'trim'
215         ));
216         $validator->validation_rules(array(
217             'y_name' => 'required|notequal[_]',
218             'y_email' => 'required|valid_email',
219             'r_name' => 'required|notequal[_]',
220             'r_email' => 'required|valid_email'
221         ));
222         $validator->text_replacements(array(
223             'y_name' => $txt['sendtopic_sender_name'],
224             'y_email' => $txt['sendtopic_sender_email'],
225             'r_name' => $txt['sendtopic_receiver_name'],
226             'r_email' => $txt['sendtopic_receiver_email']
227         ));
228 
229         // Any errors or are we good to go?
230         if (!$validator->validate($_POST))
231         {
232             $errors = $validator->validation_errors();
233 
234             return array(
235                 'errors' => $errors,
236                 'type' => 'minor',
237                 'title' => $txt['validation_failure'],
238                 // And something for ajax
239                 'error' => 1,
240                 'text' => $errors[0],
241             );
242         }
243 
244         // Emails don't like entities...
245         $row['subject'] = un_htmlspecialchars($row['subject']);
246 
247         $replacements = array(
248             'TOPICSUBJECT' => $row['subject'],
249             'SENDERNAME' => $validator->y_name,
250             'RECPNAME' => $validator->r_name,
251             'TOPICLINK' => $scripturl . '?topic=' . $topic . '.0',
252         );
253 
254         $emailtemplate = 'send_topic';
255 
256         if (!empty($_POST['comment']))
257         {
258             $emailtemplate .= '_comment';
259             $replacements['COMMENT'] = $_POST['comment'];
260         }
261 
262         $emaildata = loadEmailTemplate($emailtemplate, $replacements);
263 
264         // And off we go!
265         sendmail($validator->r_email, $emaildata['subject'], $emaildata['body'], $validator->y_email);
266 
267         return true;
268     }
