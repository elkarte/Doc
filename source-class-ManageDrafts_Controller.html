 47     /**
 48      * Modify any setting related to drafts.
 49      *
 50      * - Requires the admin_forum permission.
 51      * - Accessed from ?action=admin;area=managedrafts
 52      *
 53      * @uses Admin template, edit_topic_settings sub-template.
 54      */
 55     public function action_draftSettings_display()
 56     {
 57         global $context, $txt, $scripturl;
 58 
 59         isAllowedTo('admin_forum');
 60         loadLanguage('Drafts');
 61 
 62         // Initialize the form
 63         $this->_initDraftSettingsForm();
 64 
 65         $config_vars = $this->_draftSettings->settings();
 66 
 67         // Setup the template.
 68         $context['page_title'] = $txt['managedrafts_settings'];
 69         $context['sub_template'] = 'show_settings';
 70         $context[$context['admin_menu_name']]['tab_data'] = array(
 71             'title' => $txt['drafts'],
 72             'help' => '',
 73             'description' => $txt['managedrafts_settings_description'],
 74         );
 75 
 76         // Saving them ?
 77         if (isset($_GET['save']))
 78         {
 79             checkSession();
 80 
 81             call_integration_hook('integrate_save_drafts_settings');
 82 
 83             // Protect them from themselves.
 84             $_POST['drafts_autosave_frequency'] = $_POST['drafts_autosave_frequency'] < 30 ? 30 : $_POST['drafts_autosave_frequency'];
 85 
 86             Settings_Form::save_db($config_vars);
 87             redirectexit('action=admin;area=managedrafts');
 88         }
 89 
 90         // Some javascript to enable / disable the frequency input box
 91         addInlineJavascript('
 92             var autosave = document.getElementById(\'drafts_autosave_enabled\');
 93 
 94             createEventListener(autosave);
 95             autosave.addEventListener(\'change\', toggle);
 96             toggle();
 97 
 98             function toggle()
 99             {
100                 var select_elem = document.getElementById(\'drafts_autosave_frequency\');
101 
102                 select_elem.disabled = !autosave.checked;
103             }', true);
104 
105         // Final settings...
106         $context['post_url'] = $scripturl . '?action=admin;area=managedrafts;save';
107         $context['settings_title'] = $txt['managedrafts_settings'];
108 
109         // Prepare the settings...
110         Settings_Form::prepare_db($config_vars);
111     }
112 
113     /**

116     private function _initDraftSettingsForm()

127     /**
128      * Returns all admin drafts settings in config_vars format.
129      */
130     private function _settings()
131     {
132         global $txt;
133 
134         loadLanguage('Drafts');
135 
136         // Here are all the draft settings, a bit lite for now, but we can add more :P
137         $config_vars = array(
138                 // Draft settings ...
139                 array('check', 'drafts_post_enabled'),
140                 array('check', 'drafts_pm_enabled'),
141                 array('int', 'drafts_keep_days', 'postinput' => $txt['days_word'], 'subtext' => $txt['drafts_keep_days_subnote']),
142             '',
143                 array('check', 'drafts_autosave_enabled', 'subtext' => $txt['drafts_autosave_enabled_subnote']),
144                 array('int', 'drafts_autosave_frequency', 'postinput' => $txt['manageposts_seconds'], 'subtext' => $txt['drafts_autosave_frequency_subnote']),
145         );
146 
147         call_integration_hook('integrate_modify_drafts_settings', array(&$config_vars));
148 
149         return $config_vars;
150     }
