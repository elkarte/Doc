  26 /**
  27  * Removes the passed id_topic's.
  28  * Permissions are NOT checked here because the function is used in a scheduled task
  29  *
  30  * @param int[]|int $topics The topics to remove (can be an id or an array of ids).
  31  * @param bool $decreasePostCount if true users' post count will be reduced
  32  * @param bool $ignoreRecycling if true topics are not moved to the recycle board (if it exists).
  33  */
  34 function removeTopics($topics, $decreasePostCount = true, $ignoreRecycling = false)
  35 {
  36     global $modSettings;
  37 
  38     $db = database();
  39 
  40     // Nothing to do?
  41     if (empty($topics))
  42         return;
  43 
  44     // Only a single topic.
  45     if (is_numeric($topics))
  46         $topics = array($topics);
  47 
  48     // Decrease the post counts for members.
  49     if ($decreasePostCount)
  50     {
  51         $requestMembers = $db->query('', '
  52             SELECT m.id_member, COUNT(*) AS posts
  53             FROM {db_prefix}messages AS m
  54                 INNER JOIN {db_prefix}boards AS b ON (b.id_board = m.id_board)
  55             WHERE m.id_topic IN ({array_int:topics})
  56                 AND m.icon != {string:recycled}
  57                 AND b.count_posts = {int:do_count_posts}
  58                 AND m.approved = {int:is_approved}
  59             GROUP BY m.id_member',
  60             array(
  61                 'do_count_posts' => 0,
  62                 'recycled' => 'recycled',
  63                 'topics' => $topics,
  64                 'is_approved' => 1,
  65             )
  66         );
  67         if ($db->num_rows($requestMembers) > 0)
  68         {
  69             while ($rowMembers = $db->fetch_assoc($requestMembers))
  70                 updateMemberData($rowMembers['id_member'], array('posts' => 'posts - ' . $rowMembers['posts']));
  71         }
  72         $db->free_result($requestMembers);
  73     }
  74 
  75     // Recycle topics that aren't in the recycle board...
  76     if (!empty($modSettings['recycle_enable']) && $modSettings['recycle_board'] > 0 && !$ignoreRecycling)
  77     {
  78         $request = $db->query('', '
  79             SELECT id_topic, id_board, unapproved_posts, approved
  80             FROM {db_prefix}topics
  81             WHERE id_topic IN ({array_int:topics})
  82                 AND id_board != {int:recycle_board}
  83             LIMIT ' . count($topics),
  84             array(
  85                 'recycle_board' => $modSettings['recycle_board'],
  86                 'topics' => $topics,
  87             )
  88         );
  89         if ($db->num_rows($request) > 0)
  90         {
  91             // Get topics that will be recycled.
  92             $recycleTopics = array();
  93             while ($row = $db->fetch_assoc($request))
  94             {
  95                 if (function_exists('apache_reset_timeout'))
  96                     @apache_reset_timeout();
  97 
  98                 $recycleTopics[] = $row['id_topic'];
  99 
 100                 // Set the id_previous_board for this topic - and make it not sticky.
 101                 $db->query('', '
 102                     UPDATE {db_prefix}topics
 103                     SET id_previous_board = {int:id_previous_board}, is_sticky = {int:not_sticky}
 104                     WHERE id_topic = {int:id_topic}',
 105                     array(
 106                         'id_previous_board' => $row['id_board'],
 107                         'id_topic' => $row['id_topic'],
 108                         'not_sticky' => 0,
 109                     )
 110                 );
 111             }
 112             $db->free_result($request);
 113 
 114             // Mark recycled topics as recycled.
 115             $db->query('', '
 116                 UPDATE {db_prefix}messages
 117                 SET icon = {string:recycled}
 118                 WHERE id_topic IN ({array_int:recycle_topics})',
 119                 array(
 120                     'recycle_topics' => $recycleTopics,
 121                     'recycled' => 'recycled',
 122                 )
 123             );
 124 
 125             // Move the topics to the recycle board.
 126             require_once(SUBSDIR . '/Topic.subs.php');
 127             moveTopics($recycleTopics, $modSettings['recycle_board']);
 128 
 129             // Close reports that are being recycled.
 130             require_once(SUBSDIR . '/Moderation.subs.php');
 131 
 132             $db->query('', '
 133                 UPDATE {db_prefix}log_reported
 134                 SET closed = {int:is_closed}
 135                 WHERE id_topic IN ({array_int:recycle_topics})',
 136                 array(
 137                     'recycle_topics' => $recycleTopics,
 138                     'is_closed' => 1,
 139                 )
 140             );
 141 
 142             updateSettings(array('last_mod_report_action' => time()));
 143             recountOpenReports();
 144 
 145             // Topics that were recycled don't need to be deleted, so subtract them.
 146             $topics = array_diff($topics, $recycleTopics);
 147         }
 148         else
 149             $db->free_result($request);
 150     }
 151 
 152     // Still topics left to delete?
 153     if (empty($topics))
 154         return;
 155 
 156     $adjustBoards = array();
 157 
 158     // Find out how many posts we are deleting.
 159     $request = $db->query('', '
 160         SELECT id_board, approved, COUNT(*) AS num_topics, SUM(unapproved_posts) AS unapproved_posts,
 161             SUM(num_replies) AS num_replies
 162         FROM {db_prefix}topics
 163         WHERE id_topic IN ({array_int:topics})
 164         GROUP BY id_board, approved',
 165         array(
 166             'topics' => $topics,
 167         )
 168     );
 169     while ($row = $db->fetch_assoc($request))
 170     {
 171         if (!isset($adjustBoards[$row['id_board']]['num_posts']))

 401 /**
 402  * Moves one or more topics to a specific board.
 403  * Determines the source boards for the supplied topics
 404  * Handles the moving of mark_read data
 405  * Updates the posts count of the affected boards
 406  * This function doesn't check permissions.
 407  *
 408  * @param int[]|int $topics
 409  * @param int $toBoard
 410  */
 411 function moveTopics($topics, $toBoard)
 412 {
 413     global $user_info, $modSettings;
 414 
 415     $db = database();
 416 
 417     // Empty array?
 418     if (empty($topics))
 419         return;
 420 
 421     // Only a single topic.
 422     if (is_numeric($topics))
 423         $topics = array($topics);
 424 
 425     $fromBoards = array();
 426 
 427     // Destination board empty or equal to 0?
 428     if (empty($toBoard))
 429         return;
 430

 432     $isRecycleDest = !empty($modSettings['recycle_enable']) && $modSettings['recycle_board'] == $toBoard;
 433 
 434     // Determine the source boards...
 435     $request = $db->query('', '
 436         SELECT id_board, approved, COUNT(*) AS num_topics, SUM(unapproved_posts) AS unapproved_posts,
 437             SUM(num_replies) AS num_replies
 438         FROM {db_prefix}topics
 439         WHERE id_topic IN ({array_int:topics})
 440         GROUP BY id_board, approved',
 441         array(
 442             'topics' => $topics,
 443         )
 444     );
 445     // Num of rows = 0 -> no topics found. Num of rows > 1 -> topics are on multiple boards.
 446     if ($db->num_rows($request) == 0)
 447         return;
 448     while ($row = $db->fetch_assoc($request))
 449     {
 450         if (!isset($fromBoards[$row['id_board']]['num_posts']))
 451         {
 452             $fromBoards[$row['id_board']] = array(
 453                 'num_posts' => 0,
 454                 'num_topics' => 0,
 455                 'unapproved_posts' => 0,
 456                 'unapproved_topics' => 0,
 457                 'id_board' => $row['id_board']
 458             );
 459         }
 460         // Posts = (num_replies + 1) for each approved topic.
 461         $fromBoards[$row['id_board']]['num_posts'] += $row['num_replies'] + ($row['approved'] ? $row['num_topics'] : 0);
 462         $fromBoards[$row['id_board']]['unapproved_posts'] += $row['unapproved_posts'];
 463 
 464         // Add the topics to the right type.
 465         if ($row['approved'])
 466             $fromBoards[$row['id_board']]['num_topics'] += $row['num_topics'];
 467         else
 468             $fromBoards[$row['id_board']]['unapproved_topics'] += $row['num_topics'];
 469     }
 470     $db->free_result($request);
 471 
 472     // Move over the mark_read data. (because it may be read and now not by some!)
 473     $SaveAServer = max(0, $modSettings['maxMsgID'] - 50000);
 474     $request = $db->query('', '
 475         SELECT lmr.id_member, lmr.id_msg, t.id_topic, IFNULL(lt.unwatched, 0) as unwatched
 476         FROM {db_prefix}topics AS t
 477             INNER JOIN {db_prefix}log_mark_read AS lmr ON (lmr.id_board = t.id_board
 478                 AND lmr.id_msg > t.id_first_msg AND lmr.id_msg > {int:protect_lmr_msg})
 479             LEFT JOIN {db_prefix}log_topics AS lt ON (lt.id_topic = t.id_topic AND lt.id_member = lmr.id_member)
 480         WHERE t.id_topic IN ({array_int:topics})
 481             AND lmr.id_msg > IFNULL(lt.id_msg, 0)',
 482         array(
 483             'protect_lmr_msg' => $SaveAServer,
 484             'topics' => $topics,
 485         )
 486     );
 487     $log_topics = array();
 488     while ($row = $db->fetch_assoc($request))
 489     {
 490         $log_topics[] = array($row['id_member'], $row['id_topic'], $row['id_msg'], $row['unwatched']);
 491 
 492         // Prevent queries from getting too big. Taking some steam off.
 493         if (count($log_topics) > 500)
 494         {
 495             markTopicsRead($log_topics, true);
 496             $log_topics = array();
 497         }
 498     }
 499     $db->free_result($request);
 500 
 501     // Now that we have all the topics that *should* be marked read, and by which members...
 502     if (!empty($log_topics))
 503     {
 504         // Insert that information into the database!
 505         markTopicsRead($log_topics, true);
 506     }
 507 
 508     // Update the number of posts on each board.
 509     $totalTopics = 0;
 510     $totalPosts = 0;
 511     $totalUnapprovedTopics = 0;
 512     $totalUnapprovedPosts = 0;
 513     foreach ($fromBoards as $stats)
 514     {
 515         $db->query('', '
 516             UPDATE {db_prefix}boards
 517             SET
 518                 num_posts = CASE WHEN {int:num_posts} > num_posts THEN 0 ELSE num_posts - {int:num_posts} END,
 519                 num_topics = CASE WHEN {int:num_topics} > num_topics THEN 0 ELSE num_topics - {int:num_topics} END,
 520                 unapproved_posts = CASE WHEN {int:unapproved_posts} > unapproved_posts THEN 0 ELSE unapproved_posts - {int:unapproved_posts} END,
 521                 unapproved_topics = CASE WHEN {int:unapproved_topics} > unapproved_topics THEN 0 ELSE unapproved_topics - {int:unapproved_topics} END
 522             WHERE id_board = {int:id_board}',
 523             array(
 524                 'id_board' => $stats['id_board'],
 525                 'num_posts' => $stats['num_posts'],
 526                 'num_topics' => $stats['num_topics'],
 527                 'unapproved_posts' => $stats['unapproved_posts'],
 528                 'unapproved_topics' => $stats['unapproved_topics'],
 529             )
 530         );
 531         $totalTopics += $stats['num_topics'];
 532         $totalPosts += $stats['num_posts'];
 533         $totalUnapprovedTopics += $stats['unapproved_topics'];
 534         $totalUnapprovedPosts += $stats['unapproved_posts'];
 535     }
 536     $db->query('', '
 537         UPDATE {db_prefix}boards
 538         SET
 539             num_topics = num_topics + {int:total_topics},
 540             num_posts = num_posts + {int:total_posts},' . ($isRecycleDest ? '
 541             unapproved_posts = {int:no_unapproved}, unapproved_topics = {int:no_unapproved}' : '
 542             unapproved_posts = unapproved_posts + {int:total_unapproved_posts},
 543             unapproved_topics = unapproved_topics + {int:total_unapproved_topics}') . '
 544         WHERE id_board = {int:id_board}',
 545         array(
 546             'id_board' => $toBoard,
 547             'total_topics' => $totalTopics,
 548             'total_posts' => $totalPosts,
 549             'total_unapproved_topics' => $totalUnapprovedTopics,
 550             'total_unapproved_posts' => $totalUnapprovedPosts,
 551             'no_unapproved' => 0,
 552         )
 553     );
 554 
 555     // Move the topic.  Done.  :P
 556     $db->query('', '
 557         UPDATE {db_prefix}topics
 558         SET id_board = {int:id_board}' . ($isRecycleDest ? ',
 559             unapproved_posts = {int:no_unapproved}, approved = {int:is_approved}' : '') . '
 560         WHERE id_topic IN ({array_int:topics})',
 561         array(
 562             'id_board' => $toBoard,
 563             'topics' => $topics,
 564             'is_approved' => 1,
 565             'no_unapproved' => 0,
 566         )
 567     );
 568 
 569     // If this was going to the recycle bin, check what messages are being recycled, and remove them from the queue.
 570     if ($isRecycleDest && ($totalUnapprovedTopics || $totalUnapprovedPosts))
 571     {
 572         $request = $db->query('', '
 573             SELECT id_msg
 574             FROM {db_prefix}messages
 575             WHERE id_topic IN ({array_int:topics})
 576                 and approved = {int:not_approved}',
 577             array(
 578                 'topics' => $topics,
 579                 'not_approved' => 0,
 580             )
 581         );
 582         $approval_msgs = array();
 583         while ($row = $db->fetch_assoc($request))
 584             $approval_msgs[] = $row['id_msg'];
 585         $db->free_result($request);
 586 
 587         // Empty the approval queue for these, as we're going to approve them next.
 588         if (!empty($approval_msgs))
 589             $db->query('', '
 590                 DELETE FROM {db_prefix}approval_queue
 591                 WHERE id_msg IN ({array_int:message_list})
 592                     AND id_attach = {int:id_attach}',
 593                 array(
 594                     'message_list' => $approval_msgs,
 595                     'id_attach' => 0,
 596                 )
 597             );
 598 
 599         // Get all the current max and mins.
 600         $request = $db->query('', '
 601             SELECT id_topic, id_first_msg, id_last_msg
 602             FROM {db_prefix}topics
 603             WHERE id_topic IN ({array_int:topics})',
 604             array(
 605                 'topics' => $topics,
 606             )
 607         );
 608         $topicMaxMin = array();
 609         while ($row = $db->fetch_assoc($request))
 610         {
 611             $topicMaxMin[$row['id_topic']] = array(
 612                 'min' => $row['id_first_msg'],
 613                 'max' => $row['id_last_msg'],
 614             );
 615         }
 616         $db->free_result($request);
 617 
 618         // Check the MAX and MIN are correct.
 619         $request = $db->query('', '
 620             SELECT id_topic, MIN(id_msg) AS first_msg, MAX(id_msg) AS last_msg
 621             FROM {db_prefix}messages
 622             WHERE id_topic IN ({array_int:topics})
 623             GROUP BY id_topic',
 624             array(
 625                 'topics' => $topics,
 626             )
 627         );
 628         while ($row = $db->fetch_assoc($request))
 629         {
 630             // If not, update.
 631             if ($row['first_msg'] != $topicMaxMin[$row['id_topic']]['min'] || $row['last_msg'] != $topicMaxMin[$row['id_topic']]['max'])
 632                 $db->query('', '
 633                     UPDATE {db_prefix}topics
 634                     SET id_first_msg = {int:first_msg}, id_last_msg = {int:last_msg}
 635                     WHERE id_topic = {int:selected_topic}',
 636                     array(
 637                         'first_msg' => $row['first_msg'],
 638                         'last_msg' => $row['last_msg'],
 639                         'selected_topic' => $row['id_topic'],
 640                     )
 641                 );
 642         }
 643         $db->free_result($request);
 644     }
 645 
 646     $db->query('', '
 647         UPDATE {db_prefix}messages
 648         SET id_board = {int:id_board}' . ($isRecycleDest ? ',approved = {int:is_approved}' : '') . '
 649         WHERE id_topic IN ({array_int:topics})',
 650         array(
 651             'id_board' => $toBoard,
 652             'topics' => $topics,
 653             'is_approved' => 1,
 654         )
 655     );
 656     $db->query('', '
 657         UPDATE {db_prefix}log_reported
 658         SET id_board = {int:id_board}
 659         WHERE id_topic IN ({array_int:topics})',
 660         array(
 661             'id_board' => $toBoard,
 662             'topics' => $topics,
 663         )
 664     );
 665     $db->query('', '
 666         UPDATE {db_prefix}calendar
 667         SET id_board = {int:id_board}
 668         WHERE id_topic IN ({array_int:topics})',
 669         array(
 670             'id_board' => $toBoard,
 671             'topics' => $topics,
 672         )
 673     );
 674 
 675     // Mark target board as seen, if it was already marked as seen before.
 676     $request = $db->query('', '
 677         SELECT (IFNULL(lb.id_msg, 0) >= b.id_msg_updated) AS isSeen
 678         FROM {db_prefix}boards AS b
 679             LEFT JOIN {db_prefix}log_boards AS lb ON (lb.id_board = b.id_board AND lb.id_member = {int:current_member})
 680         WHERE b.id_board = {int:id_board}',
 681         array(
 682             'current_member' => $user_info['id'],
 683             'id_board' => $toBoard,
 684         )
 685     );
 686     list ($isSeen) = $db->fetch_row($request);
 687     $db->free_result($request);
 688 
 689     if (!empty($isSeen) && !$user_info['is_guest'])
 690     {
 691         require_once(SUBSDIR . '/Boards.subs.php');
 692         markBoardsRead($toBoard);
 693     }
 694 
 695     // Update the cache?
 696     if (!empty($modSettings['cache_enable']) && $modSettings['cache_enable'] >= 3)
 697         foreach ($topics as $topic_id)
 698             cache_put_data('topic_board-' . $topic_id, null, 120);
 699 
 700     require_once(SUBSDIR . '/Post.subs.php');
 701 
 702     $updates = array_keys($fromBoards);
 703     $updates[] = $toBoard;
 704 
 705     updateLastMessages(array_unique($updates));
 706 
 707     // Update 'em pesky stats.
 708     updateTopicStats();
 709     updateStats('message');
 710     updateSettings(array(
 711         'calendar_updated' => time(),
 712     ));
 713 }

 978 /**
 979  * Advance topic pointer.
 980  * (in either direction)
 981  * This function is used by previousTopic() and nextTopic()
 982  * The boolean parameter $next determines direction.
 983  *
 984  * @param int $id_topic origin topic id
 985  * @param int $id_board board id
 986  * @param bool $next = true whether to increase or decrease the pointer
 987  * @param int $id_member = 0 member id
 988  * @param bool $includeUnapproved = false whether to include unapproved topics
 989  * @param bool $includeStickies = true whether to include sticky topics
 990  */
 991 function topicPointer($id_topic, $id_board, $next = true, $id_member = 0, $includeUnapproved = false, $includeStickies = true)
 992 {
 993     $db = database();
 994 
 995     $request = $db->query('', '
 996         SELECT t2.id_topic
 997         FROM {db_prefix}topics AS t
 998         INNER JOIN {db_prefix}topics AS t2 ON (' .
 999             (empty($includeStickies) ? '
1000                 t2.id_last_msg {raw:strictly} t.id_last_msg' : '
1001                 (t2.id_last_msg {raw:strictly} t.id_last_msg AND t2.is_sticky {raw:strictly_equal} t.is_sticky) OR t2.is_sticky {raw:strictly} t.is_sticky')
1002             . ')
1003         WHERE t.id_topic = {int:current_topic}
1004             AND t2.id_board = {int:current_board}' .
1005             ($includeUnapproved ? '' : '
1006                 AND (t2.approved = {int:is_approved} OR (t2.id_member_started != {int:id_member_started} AND t2.id_member_started = {int:current_member}))'
1007                 ) . '
1008         ORDER BY' . (
1009             $includeStickies ? '
1010                 t2.is_sticky {raw:sorting},' :
1011                 '') .
1012             ' t2.id_last_msg {raw:sorting}
1013         LIMIT 1',
1014         array(
1015             'strictly' => $next ? '<' : '>',
1016             'strictly_equal' => $next ? '<=' : '>=',
1017             'sorting' => $next ? 'DESC' : '',
1018             'current_board' => $id_board,
1019             'current_member' => $id_member,
1020             'current_topic' => $id_topic,
1021             'is_approved' => 1,
1022             'id_member_started' => 0,
1023         )
1024     );
1025 
1026     // Was there any?
1027     if ($db->num_rows($request) == 0)
1028     {
1029         $db->free_result($request);
1030 
1031         // Roll over - if we're going prev, get the last - otherwise the first.
1032         $request = $db->query('', '
1033             SELECT id_topic
1034             FROM {db_prefix}topics
1035             WHERE id_board = {int:current_board}' .
1036             ($includeUnapproved ? '' : '
1037                 AND (approved = {int:is_approved} OR (id_member_started != {int:id_member_started} AND id_member_started = {int:current_member}))') . '
1038             ORDER BY' . (
1039                 $includeStickies ? ' is_sticky {raw:sorting},' :
1040                 '').
1041                 ' id_last_msg {raw:sorting}
1042             LIMIT 1',
1043             array(
1044                 'sorting' => $next ? 'DESC' : '',
1045                 'current_board' => $id_board,
1046                 'current_member' => $id_member,
1047                 'is_approved' => 1,
1048                 'id_member_started' => 0,
1049             )
1050         );
1051     }
1052     // Now you can be sure $topic is the id_topic to view.
1053     list ($topic) = $db->fetch_row($request);
1054     $db->free_result($request);
1055 
1056     return $topic;
