  36 function messageDetails($id_msg, $id_topic = 0, $attachment_type = 0)
  37 {
  38     global $modSettings;
  39 
  40     $db = database();
  41 
  42     if (empty($id_msg))
  43         return false;
  44 
  45     $request = $db->query('', '
  46         SELECT
  47             m.id_member, m.modified_time, m.modified_name, m.smileys_enabled, m.body,
  48             m.poster_name, m.poster_email, m.subject, m.icon, m.approved,
  49             IFNULL(a.size, -1) AS filesize, a.filename, a.id_attach,
  50             a.approved AS attachment_approved, t.id_member_started AS id_member_poster,
  51             m.poster_time, log.id_action
  52         FROM {db_prefix}messages AS m
  53             INNER JOIN {db_prefix}topics AS t ON (t.id_topic = {int:current_topic})
  54             LEFT JOIN {db_prefix}attachments AS a ON (a.id_msg = m.id_msg AND a.attachment_type = {int:attachment_type})
  55             LEFT JOIN {db_prefix}log_actions AS log ON (m.id_topic = log.id_topic AND log.action = {string:announce_action})
  56         WHERE m.id_msg = {int:id_msg}
  57             AND m.id_topic = {int:current_topic}',
  58         array(
  59             'current_topic' => $id_topic,
  60             'attachment_type' => $attachment_type,
  61             'id_msg' => $id_msg,
  62             'announce_action' => 'announce_topic',
  63         )
  64     );
  65     // The message they were trying to edit was most likely deleted.
  66     if ($db->num_rows($request) == 0)
  67         return false;
  68     $row = $db->fetch_assoc($request);
  69 
  70     $attachment_stuff = array($row);
  71     while ($row2 = $db->fetch_assoc($request))
  72         $attachment_stuff[] = $row2;
  73     $db->free_result($request);
  74 
  75     $temp = array();
  76     foreach ($attachment_stuff as $attachment)
  77     {
  78         if ($attachment['filesize'] >= 0 && !empty($modSettings['attachmentEnable']))
  79             $temp[$attachment['id_attach']] = $attachment;

  81     }

  83

 456             FROM {db_prefix}topics
 457             WHERE id_previous_topic = {int:id_previous_topic}

 460                 'id_previous_topic' => $row['id_topic'],

 462             )
 463         );

 490                 SET
 491                     id_topic = {int:id_topic},

 515                     'id_msg' => $message,

 559                     )

 563             $recycle = true;
 564

 566             updateSubjectStats($topicID, $row['subject']);
 567         }
 568 
 569         // If it wasn't approved don't keep it in the queue.
 570         if (!$row['approved'])
 571             $db->query('', '

 573                 WHERE id_msg = {int:id_msg}

 587         WHERE id_board = {int:id_board}',

 589             'no_posts' => 0,
 590             'no_unapproved' => 0,
 591             'id_board' => $row['id_board'],
 592         )
 593     );
 594 
 595     // If the poster was registered and the board this message was on incremented
 596     // the member's posts when it was posted, decrease his or her post count.
 597     if (!empty($row['id_member']) && $decreasePostCount && empty($row['count_posts']) && $row['approved'])
 598         updateMemberData($row['id_member'], array('posts' => '-'));
 599 
 600     // Only remove posts if they're not recycled.
 601     if (!$recycle)
 602     {
 603         // Remove the likes!
 604         $db->query('', '
 605             DELETE FROM {db_prefix}message_likes
 606             WHERE id_msg = {int:id_msg}',
 607             array(
 608                 'id_msg' => $message,
 609             )
 610         );
 611 
 612         // Remove the mentions!
 613         $db->query('', '
 614             DELETE FROM {db_prefix}log_mentions
 615             WHERE id_msg = {int:id_msg}',
 616             array(
 617                 'id_msg' => $message,
 618             )
 619         );
 620 
 621         // Remove the message!
 622         $db->query('', '
 623             DELETE FROM {db_prefix}messages
 624             WHERE id_msg = {int:id_msg}',
 625             array(
 626                 'id_msg' => $message,
 627             )
 628         );
 629 
 630         if (!empty($modSettings['search_custom_index_config']))
 631         {
 632             $customIndexSettings = unserialize($modSettings['search_custom_index_config']);
 633             $words = text2words($row['body'], $customIndexSettings['bytes_per_word'], true);
 634             if (!empty($words))
 635                 $db->query('', '
 636                     DELETE FROM {db_prefix}log_search_words
 637                     WHERE id_word IN ({array_int:word_list})
 638                         AND id_msg = {int:id_msg}',
 639                     array(
 640                         'word_list' => $words,
 641                         'id_msg' => $message,
 642                     )
 643                 );
 644         }
 645 
 646         // Delete attachment(s) if they exist.
 647         require_once(SUBSDIR . '/ManageAttachments.subs.php');
 648         $attachmentQuery = array(
 649             'attachment_type' => 0,
 650             'id_msg' => $message,
 651         );
 652         removeAttachments($attachmentQuery);
 653 
 654         // Delete follow-ups too
 655         require_once(SUBSDIR . '/FollowUps.subs.php');
 656 
 657         // If it is an entire topic
 658         if ($row['id_first_msg'] == $message)
 659         {
 660             $db->query('', '
 661                 DELETE FROM {db_prefix}follow_ups
 662                 WHERE follow_ups IN ({array_int:topics})',
 663                 array(
 664                     'topics' => $row['id_topic'],
 665                 )
 666             );
 667         }
 668 
 669         // Allow mods to remove message related data of their own (likes, maybe?)
 670         call_integration_hook('integrate_remove_message', array($message));
 671     }
 672 
 673     // Update the pesky statistics.
 674     updateMessageStats();
 675     updateStats('topic');
 676     updateSettings(array(
 677         'calendar_updated' => time(),
 678     ));
 679 
 680     // And now to update the last message of each board we messed with.
 681     require_once(SUBSDIR . '/Post.subs.php');
 682     if ($recycle)
 683         updateLastMessages(array($row['id_board'], $modSettings['recycle_board']));
 684     else
 685         updateLastMessages($row['id_board']);
 686 
 687     // Close any moderation reports for this message.
 688     require_once(SUBSDIR . '/Moderation.subs.php');
 689     $updated_reports = updateReportsStatus($message, 'close', 1);
 690     if ($updated_reports != 0)
 691     {
 692         updateSettings(array('last_mod_report_action' => time()));
 693         recountOpenReports();
 694     }
 695 
 696     return false;
 697 }

 828 /**
 829  * Retrieve the message id/s at a certain position in a topic
 830  *
 831  * @param int $start the offset of the message/s
 832  * @param int $id_topic the id of the topic
 833  * @param mixed[] $params an (optional) array of params, includes:
 834  *      - 'not_in' => array - of messages to exclude
 835  *      - 'include' => array - of messages to explicitely include
 836  *      - 'only_approved' => true/false - include or exclude the unapproved messages
 837  *      - 'limit' => mixed - the number of values to return (if false, no limits applied)
 838  * @todo very similar to selectMessages in Topics.subs.php
 839  */
 840 function messageAt($start, $id_topic, $params = array())
 841 {
 842     $db = database();
 843 
 844     $params = array_merge(
 845         // Defaults
 846         array(
 847             'not_in' => false,
 848             'include' => false,
 849             'only_approved' => false,
 850             'limit' => 1,
 851         ),
 852         // Passed arguments
 853         $params,
 854         // Others
 855         array(
 856             'current_topic' => $id_topic,
 857             'start' => $start,
 858             'is_approved' => 1,

 860     );
 861 
 862     $result = $db->query('', '
 863         SELECT id_msg
 864         FROM {db_prefix}messages
 865         WHERE id_topic = {int:current_topic}' . (!$params['include'] ? '' : '
 866             AND id_msg IN ({array_int:include})') . (!$params['not_in'] ? '' : '

 870         LIMIT {int:start}, {int:limit}'),
 871         $params

 875         $msg[] = $row['id_msg'];
 876     $db->free_result($result);
