 24 /**
 25  * Loads the signature from 50 members per request
 26  * Used in ManageFeatures to apply signature settings to all members
 27  *
 28  * @param int $start_member
 29  * @return array
 30  */
 31 function getSignatureFromMembers($start_member)
 32 {
 33     $db = database();
 34 
 35     $members = array();
 36 
 37     $request = $db->query('', '
 38         SELECT id_member, signature
 39         FROM {db_prefix}members
 40         WHERE id_member BETWEEN ' . $start_member . ' AND ' . $start_member . ' + 49
 41             AND id_group != {int:admin_group}
 42             AND FIND_IN_SET({int:admin_group}, additional_groups) = 0',
 43         array(
 44             'admin_group' => 1,
 45         )
 46     );
 47     while ($result = $db->fetch_assoc($request))
 48     {
 49         $members[$result['id_member']]['id_member'] = $result['id_member'];
 50         $members[$result['id_member']]['signature'] = $result['signature'];
 51     }
 52 
 53     return $members;
 54 }

 67 /**
 68  * Callback for createList() in displaying profile fields
 69  * Can be used to load standard or custom fields by setting the $standardFields flag
 70  *
 71  * @param int $start
 72  * @param int $items_per_page
 73  * @param string $sort
 74  * @param boolean $standardFields
 75  */
 76 function list_getProfileFields($start, $items_per_page, $sort, $standardFields)
 77 {
 78     global $txt, $modSettings;
 79 
 80     $db = database();
 81 
 82     $list = array();
 83 
 84     if ($standardFields)
 85     {
 86         $standard_fields = array('location', 'gender', 'website', 'personal_text', 'posts', 'warning_status');
 87         $fields_no_registration = array('posts', 'warning_status');
 88         $disabled_fields = isset($modSettings['disabled_profile_fields']) ? explode(',', $modSettings['disabled_profile_fields']) : array();
 89         $registration_fields = isset($modSettings['registration_fields']) ? explode(',', $modSettings['registration_fields']) : array();
 90 
 91         foreach ($standard_fields as $field)
 92             $list[] = array(
 93                 'id' => $field,
 94                 'label' => isset($txt['standard_profile_field_' . $field]) ? $txt['standard_profile_field_' . $field] : (isset($txt[$field]) ? $txt[$field] : $field),
 95                 'disabled' => in_array($field, $disabled_fields),
 96                 'on_register' => in_array($field, $registration_fields) && !in_array($field, $fields_no_registration),
 97                 'can_show_register' => !in_array($field, $fields_no_registration),
 98             );
 99     }
100     else
101     {
102         // Load all the fields.
103         $request = $db->query('', '
104             SELECT id_field, col_name, field_name, field_desc, field_type, active, placement, vieworder
105             FROM {db_prefix}custom_fields
106             ORDER BY {raw:sort}, vieworder ASC
107             LIMIT {int:start}, {int:items_per_page}',
108             array(
109                 'sort' => $sort,
110                 'start' => $start,
111                 'items_per_page' => $items_per_page,
112             )
113         );
114         while ($row = $db->fetch_assoc($request))
115             $list[$row['id_field']] = $row;
116         $db->free_result($request);
117     }
118 
119     return $list;
120 }

142 /**
143  * Loads the profile field properties from a given field id
144  *
145  * @param int $id_field
146  * @return array $field
147  */
148 function getProfileField($id_field)
149 {
150     $db = database();
151 
152     $field = array();
153 
154     $request = $db->query('', '
155         SELECT
156             id_field, col_name, field_name, field_desc, field_type, field_length, field_options,
157             show_reg, show_display, show_memberlist, show_profile, private, active, default_value, can_search,
158             bbc, mask, enclose, placement, vieworder
159         FROM {db_prefix}custom_fields
160         WHERE id_field = {int:current_field}',
161         array(
162             'current_field' => $id_field,
163         )
164     );
165     while ($row = $db->fetch_assoc($request))
166     {
167         if ($row['field_type'] == 'textarea')
168             @list ($rows, $cols) = explode(',', $row['default_value']);
169         else
170         {
171             $rows = 3;
172             $cols = 30;
173         }
174
