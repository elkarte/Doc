 208  * Remove one or more members from one or more membergroups.
 209  * Requires the manage_membergroups permission.
 210  * Function includes a protection against removing from implicit groups.
 211  * Non-admins are not able to remove members from the admin group.

 214  * @param integer $groups

 219 function removeMembersFromGroups($members, $groups = null, $permissionCheckDone = false)
 220 {
 221     global $modSettings;
 222 
 223     $db = database();
 224 
 225     // You're getting nowhere without this permission, unless of course you are the group's moderator.
 226     if (!$permissionCheckDone)
 227         isAllowedTo('manage_membergroups');
 228 
 229     // Assume something will happen.
 230     updateSettings(array('settings_updated' => time()));
 231 
 232     // Cleaning the input.
 233     if (!is_array($members))
 234         $members = array((int) $members);
 235     else
 236     {
 237         $members = array_unique($members);
 238 
 239         // Cast the members to integer.
 240         foreach ($members as $key => $value)
 241             $members[$key] = (int) $value;
 242     }
 243 
 244     // Before we get started, let's check we won't leave the admin group empty!
 245     if ($groups === null || $groups == 1 || (is_array($groups) && in_array(1, $groups)))
 246     {
 247         $admins = array();
 248         listMembergroupMembers_Href($admins, 1);
 249 
 250         // Remove any admins if there are too many.
 251         $non_changing_admins = array_diff(array_keys($admins), $members);
 252 
 253         if (empty($non_changing_admins))
 254             $members = array_diff($members, array_keys($admins));
 255     }
 256 
 257     // Just in case.
 258     if (empty($members))
 259         return false;
 260 
 261     // Wanna remove all groups from these members? That's easy.
 262     if ($groups === null)
 263     {
 264         $db->query('', '
 265             UPDATE {db_prefix}members
 266             SET
 267                 id_group = {int:regular_member},
 268                 additional_groups = {string:blank_string}
 269             WHERE id_member IN ({array_int:member_list})' . (allowedTo('admin_forum') ? '' : '
 270                 AND id_group != {int:admin_group}
 271                 AND FIND_IN_SET({int:admin_group}, additional_groups) = 0'),
 272             array(
 273                 'member_list' => $members,
 274                 'regular_member' => 0,
 275                 'admin_group' => 1,
 276                 'blank_string' => '',
 277             )
 278         );
 279 
 280         updatePostgroupStats($members);
 281 
 282         // Log what just happened.
 283         foreach ($members as $member)
 284             logAction('removed_all_groups', array('member' => $member), 'admin');
 285 
 286         return true;
 287     }
 288     elseif (!is_array($groups))
 289         $groups = array((int) $groups);
 290     // Make sure all groups are integer.
 291     else
 292         $groups = array_unique(array_map('intval', $groups));
 293 
 294     // Fetch a list of groups members cannot be assigned to explicitely, and the group names of the ones we want.
 295     $implicitGroups = array(-1, 0, 3);
 296     $group_names = array();
 297     $group_details = membergroupsById($groups, 0, true);
 298     foreach ($group_details as $key => $row)
 299     {
 300         if ($row['min_posts'] != -1)
 301             $implicitGroups[] = $row['id_group'];
 302         else
 303             $group_names[$row['id_group']] = $row['group_name'];
 304     }
 305 
 306     // Now get rid of those groups.
 307     $groups = array_diff($groups, $implicitGroups);
 308 
 309     // Don't forget the protected groups.
 310     if (!allowedTo('admin_forum'))
 311     {
 312         $request = $db->query('', '
 313             SELECT id_group
 314             FROM {db_prefix}membergroups
 315             WHERE group_type = {int:is_protected}',
 316             array(
 317                 'is_protected' => 1,
 318             )
 319         );
 320         $protected_groups = array(1);
 321         while ($row = $db->fetch_assoc($request))
 322             $protected_groups[] = $row['id_group'];
 323         $db->free_result($request);
 324 
 325         // If you're not an admin yourself, you can't touch protected groups!
 326         $groups = array_diff($groups, array_unique($protected_groups));
 327     }
 328 
 329     // Only continue if there are still groups and members left.
 330     if (empty($groups) || empty($members))
 331         return false;
 332 
 333     // First, reset those who have this as their primary group - this is the easy one.
 334     $log_inserts = array();
 335     $request = $db->query('', '
 336         SELECT id_member, id_group
 337         FROM {db_prefix}members AS members
 338         WHERE id_group IN ({array_int:group_list})
 339             AND id_member IN ({array_int:member_list})',
 340         array(
 341             'group_list' => $groups,
 342             'member_list' => $members,
 343         )
 344     );
 345     while ($row = $db->fetch_assoc($request))
 346         $log_inserts[] = array('group' => $group_names[$row['id_group']], 'member' => $row['id_member']);
 347     $db->free_result($request);
 348 
 349     $db->query('', '
 350         UPDATE {db_prefix}members
 351         SET id_group = {int:regular_member}
 352         WHERE id_group IN ({array_int:group_list})
 353             AND id_member IN ({array_int:member_list})',
 354         array(
 355             'group_list' => $groups,
 356             'member_list' => $members,
 357             'regular_member' => 0,
 358         )
 359     );
 360 
 361     // Those who have it as part of their additional group must be updated the long way... sadly.
 362     $request = $db->query('', '
 363         SELECT id_member, additional_groups
 364         FROM {db_prefix}members
 365         WHERE (FIND_IN_SET({raw:additional_groups_implode}, additional_groups) != 0)
 366             AND id_member IN ({array_int:member_list})
 367         LIMIT ' . count($members),
 368         array(
 369             'member_list' => $members,
 370             'additional_groups_implode' => implode(', additional_groups) != 0 OR FIND_IN_SET(', $groups),

 380

 386         updateMemberData($memberArray, array('additional_groups' => array_diff(explode(',', $additional_groups), $groups)));

 400 /**
 401  * Add one or more members to a membergroup.
 402  *
 403  * Requires the manage_membergroups permission.
 404  * Function has protection against adding members to implicit groups.
 405  * Non-admins cannot add members to the admin group, or protected groups.
 406  *
 407  * @param int|int[] $members
 408  * @param int $group
 409  * @param string $type = 'auto' specifies whether the group is added as primary or as additional group.
 410  * Supported types:
 411  *  - only_primary    - Assigns a membergroup as primary membergroup, but only
 412  *                      if a member has not yet a primary membergroup assigned,
 413  *                      unless the member is already part of the membergroup.
 414  *  - only_additional - Assigns a membergroup to the additional membergroups,
 415  *                      unless the member is already part of the membergroup.
 416  *  - force_primary   - Assigns a membergroup as primary membergroup no matter
 417  *                      what the previous primary membergroup was.
 418  *  - auto            - Assigns a membergroup to the primary group if it's still
 419  *                      available. If not, assign it to the additional group.
 420  * @param bool $permissionCheckDone = false if true, it checks permission of the current user to add groups ('manage_membergroups')
 421  *
 422  * @return boolean success or failure
 423  */
 424 function addMembersToGroup($members, $group, $type = 'auto', $permissionCheckDone = false)
 425 {
 426     $db = database();
 427 
 428     // Show your licence, but only if it hasn't been done yet.
 429     if (!$permissionCheckDone)
 430         isAllowedTo('manage_membergroups');
 431 
 432     // Make sure we don't keep old stuff cached.
 433     updateSettings(array('settings_updated' => time()));
 434 
 435     if (!is_array($members))
 436         $members = array((int) $members);
 437     // Make sure all members are integer.
 438     else
 439         $members = array_unique(array_map('intval', $members));
 440 
 441     $group = (int) $group;
 442 
 443     // Some groups just don't like explicitly having members.
 444     $implicitGroups = array(-1, 0, 3);
 445     $group_names = array();
 446     $group_details = membergroupById($group, true);
 447     if ($group_details['min_posts'] != -1)
 448         $implicitGroups[] = $group_details['id_group'];
 449     else
 450         $group_names[$group_details['id_group']] = $group_details['group_name'];
 451 
 452     // Sorry, you can't join an implicit group.
 453     if (in_array($group, $implicitGroups) || empty($members))
 454         return false;
 455 
 456     // Only admins can add admins...
 457     if (!allowedTo('admin_forum') && $group == 1)
 458         return false;
 459     // ... and assign protected groups!
 460     elseif (!allowedTo('admin_forum') && $group_details['group_type'] == 1)
 461         return false;
 462 
 463     // Do the actual updates.
 464     if ($type == 'only_additional')
 465         $db->query('', '
 466             UPDATE {db_prefix}members
 467             SET additional_groups = CASE WHEN additional_groups = {string:blank_string} THEN {string:id_group_string} ELSE CONCAT(additional_groups, {string:id_group_string_extend}) END
 468             WHERE id_member IN ({array_int:member_list})
 469                 AND id_group != {int:id_group}
 470                 AND FIND_IN_SET({int:id_group}, additional_groups) = 0',
 471             array(
 472                 'member_list' => $members,
 473                 'id_group' => $group,
 474                 'id_group_string' => (string) $group,
 475                 'id_group_string_extend' => ',' . $group,
 476                 'blank_string' => '',
 477             )
 478         );
 479     elseif ($type == 'only_primary' || $type == 'force_primary')
 480         $db->query('', '
 481             UPDATE {db_prefix}members
 482             SET id_group = {int:id_group}
 483             WHERE id_member IN ({array_int:member_list})' . ($type == 'force_primary' ? '' : '
 484                 AND id_group = {int:regular_group}
 485                 AND FIND_IN_SET({int:id_group}, additional_groups) = 0'),
 486             array(
 487                 'member_list' => $members,
 488                 'id_group' => $group,
 489                 'regular_group' => 0,
 490             )
 491         );
 492     elseif ($type == 'auto')
 493         $db->query('', '
 494             UPDATE {db_prefix}members
 495             SET
 496                 id_group = CASE WHEN id_group = {int:regular_group} THEN {int:id_group} ELSE id_group END,
 497                 additional_groups = CASE WHEN id_group = {int:id_group} THEN additional_groups
 498                     WHEN additional_groups = {string:blank_string} THEN {string:id_group_string}
 499                     ELSE CONCAT(additional_groups, {string:id_group_string_extend}) END
 500             WHERE id_member IN ({array_int:member_list})
 501                 AND id_group != {int:id_group}
 502                 AND FIND_IN_SET({int:id_group}, additional_groups) = 0',
 503             array(
 504                 'member_list' => $members,
 505                 'regular_group' => 0,
 506                 'id_group' => $group,
 507                 'blank_string' => '',
 508                 'id_group_string' => (string) $group,
 509                 'id_group_string_extend' => ',' . $group,
 510             )
 511         );
 512     // Ack!!?  What happened?
 513     else
 514         trigger_error('addMembersToGroup(): Unknown type \'' . $type . '\'', E_USER_WARNING);
 515 
 516     call_integration_hook('integrate_add_members_to_group', array($members, $group_details, &$group_names));
 517 
 518     // Update their postgroup statistics.
 519     updatePostgroupStats($members);
 520 
 521     require_once(SOURCEDIR . '/Logging.php');
 522     foreach ($members as $member)
 523         logAction('added_to_group', array('group' => $group_names[$group], 'member' => $member), 'admin');
 524
