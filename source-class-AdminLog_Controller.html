 36      * Accessed by ?action=admin;area=logs

 38     public function action_index()
 39     {
 40         global $context, $txt, $scripturl, $modSettings;
 41 
 42         // We're working with them settings here.
 43         require_once(SUBSDIR . '/Settings.class.php');
 44 
 45         // These are the logs they can load.
 46         $log_functions = array(
 47             'errorlog' => array(
 48                 'file' => 'ManageErrors.controller.php',
 49                 'function' => 'action_index',
 50                 'controller' => 'ManageErrors_Controller'),
 51             'adminlog' => array(
 52                 'file' => 'Modlog.controller.php',
 53                 'function' => 'action_log',
 54                 'controller' => 'Modlog_Controller'),
 55             'modlog' => array(
 56                 'file' => 'Modlog.controller.php',
 57                 'function' => 'action_log',
 58                 'controller' => 'Modlog_Controller',
 59                 'disabled' => !in_array('ml', $context['admin_features'])),
 60             'badbehaviorlog' => array(
 61                 'file' => 'BadBehavior.controller.php',
 62                 'function' => 'action_log',
 63                 'disabled' => empty($modSettings['badbehavior_enabled']),
 64                 'controller' => 'BadBehavior_Controller'),
 65             'banlog' => array(
 66                 'file' => 'ManageBans.controller.php',
 67                 'function' => 'action_log',
 68                 'controller' => 'ManageBans_Controller'),
 69             'spiderlog' => array(
 70                 'file' => 'ManageSearchEngines.controller.php',
 71                 'function' => 'action_logs',
 72                 'controller' => 'ManageSearchEngines_Controller'),
 73             'tasklog' => array(
 74                 'file' => 'ManageScheduledTasks.controller.php',
 75                 'function' => 'action_log',
 76                 'controller' => 'ManageScheduledTasks_Controller'),
 77             'pruning' => array(
 78                 'init' => '_initPruningSettingsForm',
 79                 'display' => 'action_pruningSettings_display'),
 80         );
 81 
 82         call_integration_hook('integrate_manage_logs', array(&$log_functions));
 83 
 84         $sub_action = isset($_REQUEST['sa']) && isset($log_functions[$_REQUEST['sa']]) && empty($log_functions[$_REQUEST['sa']]['disabled']) ? $_REQUEST['sa'] : 'errorlog';
 85 
 86         // If it's not got a sa set it must have come here for first time, pretend error log should be reversed.
 87         if (!isset($_REQUEST['sa']))
 88             $_REQUEST['desc'] = true;
 89 
 90         // Setup some tab stuff.
 91         $context[$context['admin_menu_name']]['tab_data'] = array(
 92             'title' => $txt['logs'],
 93             'help' => '',
 94             'description' => $txt['maintain_info'],
 95             'tabs' => array(
 96                 'errorlog' => array(
 97                     'url' => $scripturl . '?action=admin;area=logs;sa=errorlog;desc',
 98                     'description' => sprintf($txt['errlog_desc'], $txt['remove']),
 99                 ),
100                 'adminlog' => array(
101                     'description' => $txt['admin_log_desc'],
102                 ),
103                 'modlog' => array(
104                     'description' => $txt['moderation_log_desc'],
105                 ),
106                 'banlog' => array(
107                     'description' => $txt['ban_log_description'],
108                 ),
109                 'spiderlog' => array(
110                     'description' => $txt['spider_log_desc'],
111                 ),
112                 'tasklog' => array(
113                     'description' => $txt['scheduled_log_desc'],
114                 ),
115                 'badbehaviorlog' => array(
116                     'description' => $txt['badbehavior_log_desc'],
117                 ),
118                 'pruning' => array(
119                     'description' => $txt['pruning_log_desc'],
120                 ),
121             ),
122         );
123 
124         // figure out what to call
125         if (isset($log_functions[$sub_action]['file']))
126         {
127             // different file
128             require_once(ADMINDIR . '/' . $log_functions[$sub_action]['file']);
129         }
130 
131         if (isset($log_functions[$sub_action]['controller']))
132         {
133             // if we have an object oriented controller, call its method
134             $controller = new $log_functions[$sub_action]['controller']();
135             $controller->{$log_functions[$sub_action]['function']}();
136         }
137         elseif (isset($log_functions[$sub_action]['function']))
138         {
139             // procedural: call the function
140             $log_functions[$sub_action]['function']();
141         }
142         else
143         {
144             // our own method then
145 
146             // initialize the form
147             $this->{$log_functions[$sub_action]['init']}();
148 
149             // call the action handler
150             // this is hardcoded now, to be fixed
151             $this->{$log_functions[$sub_action]['display']}();
152         }
153     }

155     /**
156      * Allow to edit the settings on the pruning screen.
157      *
158      * Uses the _pruningSettings form.
159      */
160     public function action_pruningSettings_display()
161     {
162         global $txt, $scripturl, $context, $modSettings;
163 
164         // Make sure we understand what's going on.
165         loadLanguage('ManageSettings');
166 
167         $context['page_title'] = $txt['pruning_title'];
168 
169         $config_vars = $this->_pruningSettings->settings();
170 
171         call_integration_hook('integrate_prune_settings');
172 
173         // Saving?
174         if (isset($_GET['save']))
175         {
176             checkSession();
177 
178             $savevar = array(
179                 array('text', 'pruningOptions')
180             );
181 
182             if (!empty($_POST['pruningOptions']))
183             {
184                 $vals = array();
185                 foreach ($config_vars as $index => $dummy)
186                 {
187                     if (!is_array($dummy) || $index == 'pruningOptions')
188                         continue;
189 
190                     $vals[] = empty($_POST[$dummy[1]]) || $_POST[$dummy[1]] < 0 ? 0 : (int) $_POST[$dummy[1]];
191                 }
192                 $_POST['pruningOptions'] = implode(',', $vals);
193             }
194             else
195                 $_POST['pruningOptions'] = '';
196 
197             Settings_Form::save_db($savevar);
198             redirectexit('action=admin;area=logs;sa=pruning');
199         }
200 
201         $context['post_url'] = $scripturl . '?action=admin;area=logs;save;sa=pruning';
202         $context['settings_title'] = $txt['pruning_title'];
203         $context['sub_template'] = 'show_settings';
204 
205         // Get the actual values
206         if (!empty($modSettings['pruningOptions']))
207             @list ($modSettings['pruneErrorLog'], $modSettings['pruneModLog'], $modSettings['pruneBanLog'], $modSettings['pruneReportLog'], $modSettings['pruneScheduledTaskLog'], $modSettings['pruneBadbehaviorLog'], $modSettings['pruneSpiderHitLog']) = explode(',', $modSettings['pruningOptions']);
208         else
209             $modSettings['pruneErrorLog'] = $modSettings['pruneModLog'] = $modSettings['pruneBanLog'] = $modSettings['pruneReportLog'] = $modSettings['pruneScheduledTaskLog'] = $modSettings['pruneBadbehaviorLog'] = $modSettings['pruneSpiderHitLog'] = 0;
210 
211         Settings_Form::prepare_db($config_vars);
212     }

228     /**
229      * Returns the configuration settings for pruning logs.
230      */
231     private function _settings()
232     {
233         global $txt;
234 
235         $config_vars = array(
236             // Even do the pruning?
237             // The array indexes are there so we can remove/change them before saving.
238             'pruningOptions' => array('check', 'pruningOptions'),
239         '',
240             // Various logs that could be pruned.

242             array('int', 'pruneModLog', 'postinput' => $txt['days_word'], 'subtext' => $txt['zero_to_disable']), // Moderation log.
243             array('int', 'pruneBanLog', 'postinput' => $txt['days_word'], 'subtext' => $txt['zero_to_disable']), // Ban hit log.
244             array('int', 'pruneReportLog', 'postinput' => $txt['days_word'], 'subtext' => $txt['zero_to_disable']), // Report to moderator log.
245             array('int', 'pruneScheduledTaskLog', 'postinput' => $txt['days_word'], 'subtext' => $txt['zero_to_disable']), // Log of the scheduled tasks and how long they ran.
246             array('int', 'pruneBadbehaviorLog', 'postinput' => $txt['days_word'], 'subtext' => $txt['zero_to_disable']), // Bad Behavior log.
247             array('int', 'pruneSpiderHitLog', 'postinput' => $txt['days_word'], 'subtext' => $txt['zero_to_disable']), // Log of the scheduled tasks and how long they ran.
248             // If you add any additional logs make sure to add them after this point.  Additionally, make sure you add them to the weekly scheduled task.
249         );
250 
251         return $config_vars;
252     }
