 17     die('No access...');

 19 /**
 20  * Similar in construction to saveDBSettings,
 21  *
 22  * - Saves the config_vars to a specified table instead of a file
 23  * - Var names are considered the table col names,
 24  * - Values are cast per config vars
 25  * - If editing a row, the primary col index and existing index value must be
 26  *   supplied (editid and editname), otherwise a new row will be added
 27  *
 28  * @package Maillist
 29  */
 30 class MaillistSettingsClass extends Settings_Form
 31 {
 32     /**
 33      * static function saveTableSettings, now part of the Settings Form class
 34      *
 35      * @param mixed[] $config_vars the key names of the vars are the table cols
 36      * @param string $tablename name of the table the values will be saved in
 37      * @param string[] $index for compatability
 38      * @param integer $editid -1 add a row, otherwise edit a row with the supplied key value
 39      * @param string $editname used when editing a row, needs to be the name of the col to find $editid key value
 40      */
 41     public static function saveTableSettings($config_vars, $tablename, $index = array(), $editid = -1, $editname = '')
 42     {
 43         $db = database();
 44 
 45         // Init
 46         $insert_type = array();
 47         $insert_value = array();
 48         $update = false;
 49 
 50         // Cast all the config vars as defined
 51         foreach ($config_vars as $var)
 52         {
 53             if (!isset($var[1]) || (!isset($_POST[$var[1]]) && $var[0] !== 'check'))
 54                 continue;
 55 
 56             // Checkboxes ...
 57             elseif ($var[0] === 'check')
 58             {
 59                 $insert_type[$var[1]] = 'int';
 60                 $insert_value[] = !empty($_POST[$var[1]]) ? 1 : 0;
 61             }
 62             // Or maybe even select boxes
 63             elseif ($var[0] === 'select' && in_array($_POST[$var[1]], array_keys($var[2])))
 64             {
 65                 $insert_type[$var[1]] = 'string';
 66                 $insert_value[] = $_POST[$var[1]];
 67             }
 68             elseif ($var[0] === 'select' && !empty($var['multiple']) && array_intersect($_POST[$var[1]], array_keys($var[2])) != array())
 69             {
 70                 // For security purposes we need to validate this line by line.
 71                 $options = array();
 72                 foreach ($_POST[$var[1]] as $invar)
 73                 {
 74                     if (in_array($invar, array_keys($var[2])))
 75                         $options[] = $invar;
 76                 }
 77 
 78                 $insert_type[$var[1]] = 'string';
 79                 $insert_value[] = serialize($options);
 80             }
 81             // Integers are fundamental
 82             elseif ($var[0] == 'int')
 83             {
 84                 $insert_type[$var[1]] = 'int';
 85                 $insert_value[] = (int) $_POST[$var[1]];
 86             }
 87             // Floating points are easy
 88             elseif ($var[0] === 'float')
 89             {
 90                 $insert_type[$var[1]] = 'float';
 91                 $insert_value[] = (float) $_POST[$var[1]];
 92             }
 93             // Text is fine too!
 94             elseif ($var[0] === 'text' || $var[0] === 'large_text')
 95             {
 96                 $insert_type[$var[1]] = 'string';
 97                 $insert_value[] = $_POST[$var[1]];
 98             }
 99         }
100 
101         // Everything is now set so is this a new row or an edit?
102         if ($editid !== -1 && !empty($editname))
103         {
104             // Time to edit, add in the id col name, assumed to be primary/unique!
105             $update = true;
106             $insert_type[$editname] = 'int';
107             $insert_value[] = $editid;
108         }
109 
110         // Do it !!
111         $db->insert($update ? 'replace' : 'insert',
112             '{db_prefix}' . $tablename,
113             $insert_type,
114             $insert_value,
115             $index
116         );
117     }
118 }
