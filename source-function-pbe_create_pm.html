 33     /**
 34      * Main email posting controller, reads, parses, checks and posts an email message or PM
 35      *
 36      * What it does:
 37      * - Allows a user to reply to a topic on the board by emailing a reply to a
 38      * notification message.
 39      * - It must have the security key in the email or it will be rejected
 40      * - It must be from the email of a registered user
 41      * - The key must have been sent to that user
 42      * - Keys are used once and then discarded
 43      * - Accessed by email imap cron script, and ManageMaillist.controller.php.
 44      *
 45      * @param string|null $data used to supply a full headers+body email
 46      * @param boolean $force used to override common failure errors
 47      * @param string|null $key used to supply a lost key
 48      */
 49     public function action_pbe_post($data = null, $force = false, $key = null)
 50     {
 51         global $txt, $modSettings, $language, $user_info, $maintenance;
 52 
 53         // The function is not even on ...
 54         if (empty($modSettings['maillist_enabled']))
 55             return;
 56 
 57         // Our mail parser and our main subs
 58         require_once(SUBSDIR . '/EmailParse.class.php');
 59         require_once(SUBSDIR . '/Emailpost.subs.php');
 60 
 61         // Init
 62         loadLanguage('Maillist');
 63         setMemoryLimit('256M');
 64 
 65         // Load the email parser and get some data to work with
 66         $email_message = new Email_Parse();
 67         $email_message->read_data($data, BOARDDIR);
 68         if (!$email_message->raw_message)
 69             return false;
 70 
 71         // Ask for an html version (if available) and some needed details
 72         $email_message->read_email(true, $email_message->raw_message);
 73         $email_message->load_address();
 74         $email_message->load_key($key);
 75 
 76         // If the feature is on but the post/pm function is not enabled, just log the message.
 77         if (empty($modSettings['pbe_post_enabled']) && empty($modSettings['pbe_pm_enabled']))
 78             return pbe_emailError('error_email_notenabled', $email_message);
 79 
 80         // Spam I am?
 81         if ($email_message->load_spam())
 82             return pbe_emailError('error_found_spam', $email_message);
 83 
 84         // Load the user from the database based on the sending email address
 85         $email_message->email['from'] = !empty($email_message->email['from']) ? strtolower($email_message->email['from']) : '';
 86         $pbe = query_load_user_info($email_message->email['from']);
 87 
 88         // Can't find this email in our database, a non-user, a spammer, a looser, a poser or even worse?
 89         if (empty($pbe))
 90             return pbe_emailError('error_not_find_member', $email_message);
 91 
 92         // Find the message security key, without it we are not going anywhere ever
 93         if (empty($email_message->message_key_id))
 94             return pbe_emailError('error_missing_key', $email_message);
 95 
 96         // Good we have a key, who was it sent to?
 97         $key_owner = query_key_owner($email_message->message_key_id);
 98 
 99         // Can't find this key in the database, either
100         // a) spam attempt or b) replying with an expired/consumed key
101         if (empty($key_owner))
102             return pbe_emailError('error_' . ($email_message->message_type === 'p' ? 'pm_' : '') . 'not_find_entry', $email_message);
103 
104         // The key received was not sent to this member ... how we love those email aggregators
105         if (strtolower($key_owner) !== $email_message->email['from'] && !$force)
106             return pbe_emailError('error_key_sender_match', $email_message);
107 
108         // In maintenance mode, just log it for now
109         if (!empty($maintenance) && $maintenance !== 2 && !$pbe['user_info']['is_admin'] && !$user_info['is_admin'])
110             return pbe_emailError('error_in_maintenance_mode', $email_message);
111 
112         // The email looks valid, now on to check the actual user trying to make the post/pm
113         // lets load the topic/message info and any additional permissions we need
114         if ($email_message->message_type === 't' || $email_message->message_type === 'm')
115         {
116             // Load the message/topic details
117             $topic_info = query_load_message($email_message->message_type, $email_message->message_id, $pbe);
118             if (empty($topic_info))
119                 return pbe_emailError('error_topic_gone', $email_message);
120 
121             // Load board permissions
122             query_load_permissions('board', $pbe, $topic_info);
123         }
124         else
125         {
126             // Load the PM details
127             $pm_info = query_load_message($email_message->message_type, $email_message->message_id, $pbe);
128             if (empty($pm_info))
129             {
130                 // Duh oh ... likely they deleted the PM on the site and are now
131                 // replying to the PM by email, the agony!
132                 return pbe_emailError('error_pm_not_found', $email_message);
133             }
134         }
135 
136         // Account for moderation actions
137         pbe_check_moderation($pbe);
138 
139         // Maybe they want to do additional spam / security checking
140         call_integration_hook('integrate_mailist_checks_before', array($email_message, $pbe));
141 
142         // Load in the correct Re: for the language
143         if ($language === $pbe['user_info']['language'])
144             $pbe['response_prefix'] = $txt['response_prefix'];
145         else
146         {
147             loadLanguage('index', $language, false);
148             $pbe['response_prefix'] = $txt['response_prefix'];
149             loadLanguage('index');
150         }
151 
152         // Allow for new topics to be started via a email subject change
153         if (!empty($modSettings['maillist_newtopic_change']) && ($email_message->message_type === 'm'))
154         {
155             $subject = str_replace($pbe['response_prefix'], '', pbe_clean_email_subject($email_message->subject));
156             $current_subject = str_replace($pbe['response_prefix'], '', $topic_info['subject']);
157 
158             // If it does not match, then we go to make a new topic instead
159             if (trim($subject) != trim($current_subject))
160             {
161                 $board_info = query_load_board_details($topic_info['id_board'], $pbe);
162                 return pbe_create_topic($pbe, $email_message, $board_info);
163             }
164         }
165 
166         // Time to make a Post or a PM, first up topic and message replies
167         if ($email_message->message_type === 't' || $email_message->message_type === 'm')
168             $result = pbe_create_post($pbe, $email_message, $topic_info);
169         // Must be a PM then
170         elseif ($email_message->message_type === 'p')
171             $result = pbe_create_pm($pbe, $email_message, $pm_info);
172 
173         if (!empty($result))
174         {
175             // We have now posted or PM'ed .. lets do some database maintenance cause maintenance is fun :'(
176             query_key_maintenance($email_message);
177 
178             // Update this user so the log shows they were/are active, no luking in the email ether
179             query_update_member_stats($pbe, $email_message, $topic_info);
180         }
181 
182         return !empty($result);
183     }

185     /**
186      * New Topic posting controller, reads, parses, checks and posts a new topic
187      *
188      * What it does:
189      * - New topics do not have security keys in them so they are subject to spoofing
190      * - It must be from the email of a registered user
191      * - It must have been sent to an email ID that has been set to post new topics
192      * - Accessed through emailtopic.
193      *
194      * @param string|null $data used to supply a full body+headers email
195      */
196     public function action_pbe_topic($data = null)
197     {
198         global $modSettings, $user_info, $maintenance;
199 
200         // The function is not even on ...
201         if (empty($modSettings['maillist_enabled']))
202             return;
203 
204         // Our mail parser and our main subs
205         require_once(SUBSDIR . '/EmailParse.class.php');
206         require_once(SUBSDIR . '/Emailpost.subs.php');
207 
208         // Init
209         loadLanguage('Maillist');
210         setMemoryLimit('256M');
211 
212         // Get the data from one of our sources
213         $email_message = new Email_Parse();
214         $email_message->read_data($data, BOARDDIR);
215         if (!$email_message->raw_message)
216             return false;
217 
218         // Parse the header and some needed details
219         $email_message->read_email(true, $email_message->raw_message);
220         $email_message->load_address();
221 
222         // No key for this, so set some blanks for the error function (if needed)
223         $email_message->message_type = 'x';
224         $email_message->message_key_id = '';
225         $email_message->message_id = 0;
226 
227         // If the feature is on but the post/pm function is not enabled, just log the message.
228         if (empty($modSettings['pbe_post_enabled']))
229             return pbe_emailError('error_email_notenabled', $email_message);
230 
231         // Load the user from the database based on the sending email address
232         $email_message->email['from'] = !empty($email_message->email['from']) ? strtolower($email_message->email['from']) : '';
233         $pbe = query_load_user_info($email_message->email['from']);
234 
235         // Can't find this email as one of our users?
236         if (empty($pbe))
237             return pbe_emailError('error_not_find_member', $email_message);
238 
239         // Getting hammy with it?
240         if ($email_message->load_spam())
241             return pbe_emailError('error_found_spam', $email_message);
242 
243         // The board that this email address corresponds to
244         $board_number = pbe_find_board_number($email_message);
245         if (empty($board_number))
246             return pbe_emailError('error_not_find_board', $email_message);
247 
248         // In maintenance mode so just save it for the moderators to deal with
249         if (!empty($maintenance) && $maintenance !== 2 && !$pbe['user_info']['is_admin'] && !$user_info['is_admin'])
250             return pbe_emailError('error_in_maintenance_mode', $email_message);
251 
252         // Any additional spam / security checking
253         call_integration_hook('integrate_mailist_checks_before', array($email_message, $pbe));
254 
255         // To post a NEW topic, we need some board details for where it goes
256         $board_info = query_load_board_details($board_number, $pbe);
257         if (empty($board_info))
258             return pbe_emailError('error_board_gone', $email_message);
259 
260         // Load up this users permissions for that board
261         query_load_permissions('board', $pbe, $board_info);
262 
263         // Account for any moderation they may be under
264         pbe_check_moderation($pbe);
265 
266         // Create the topic, send notifications
267         return pbe_create_topic($pbe, $email_message, $board_info);
268     }

326 /**
327  * Attempts to create a reply post on the forum
328  *
329  * What it does:
330  * - Checks if the user has permissions to post/reply/postby email
331  * - Calls pbe_load_text to prepare text for the post
332  * - returns true if successful or false for any number of failures
333  *
334  * @package Maillist
335  * @param mixed[] $pbe array of all pbe user_info values
336  * @param Email_Parse $email_message
337  * @param mixed[] $topic_info
338  */
339 function pbe_create_post($pbe, $email_message, $topic_info)
340 {
341     global $modSettings, $txt;
342 
343     // Validate they have permission to reply
344     $becomesApproved = true;
345     if (!in_array('postby_email', $pbe['user_info']['permissions']))
346         return pbe_emailError('error_permission', $email_message);
347     elseif ($topic_info['locked'] && !$pbe['user_info']['is_admin'] && !in_array('moderate_forum', $pbe['user_info']['permissions']))
348         return pbe_emailError('error_locked', $email_message);
349     elseif ($topic_info['id_member_started'] === $pbe['profile']['id_member'] && !$pbe['user_info']['is_admin'])
350     {
351         if ($modSettings['postmod_active'] && in_array('post_unapproved_replies_any', $pbe['user_info']['permissions']) && (!in_array('post_reply_any', $pbe['user_info']['permissions'])))
352             $becomesApproved = false;
353         elseif (!in_array('post_reply_own', $pbe['user_info']['permissions']))
354             return pbe_emailError('error_cant_reply', $email_message);
355     }
356     elseif (!$pbe['user_info']['is_admin'])
357     {
358         if ($modSettings['postmod_active'] && in_array('post_unapproved_replies_any', $pbe['user_info']['permissions']) && (!in_array('post_reply_any', $pbe['user_info']['permissions'])))
359             $becomesApproved = false;
360         elseif (!in_array('post_reply_any', $pbe['user_info']['permissions']))
361             return pbe_emailError('error_cant_reply', $email_message);
362     }
363 
364     // Convert to BBC and Format the message
365     $html = $email_message->html_found;
366     $text = pbe_load_text($html, $email_message, $pbe);
367     if (empty($text))
368         return pbe_emailError('error_no_message', $email_message);
369 
370     // Seriously? Attachments?
371     if (!empty($email_message->attachments) && !empty($modSettings['maillist_allow_attachments']) && !empty($modSettings['attachmentEnable']) && $modSettings['attachmentEnable'] == 1)
372     {
373         if (($modSettings['postmod_active'] && in_array('post_unapproved_attachments', $pbe['user_info']['permissions'])) || in_array('post_attachment', $pbe['user_info']['permissions']))
374             $attachIDs = pbe_email_attachments($pbe, $email_message);
375         else
376             $text .= "\n\n" . $txt['error_no_attach'] . "\n";
377     }
378 
379     // Setup the post variables.
380     $msgOptions = array(
381         'id' => 0,
382         'subject' => strpos($topic_info['subject'], trim($pbe['response_prefix'])) === 0 ? $topic_info['subject'] : $pbe['response_prefix'] . $topic_info['subject'],
383         'smileys_enabled' => true,
384         'body' => $text,
385         'attachments' => empty($attachIDs) ? array() : $attachIDs,
386         'approved' => $becomesApproved
387     );
388 
389     $topicOptions = array(
390         'id' => $topic_info['id_topic'],
391         'board' => $topic_info['id_board'],
392         'mark_as_read' => true,
393         'is_approved' => !$modSettings['postmod_active'] || empty($topic_info['id_topic']) || !empty($topic_info['approved'])
394     );
395 
396     $posterOptions = array(
397         'id' => $pbe['profile']['id_member'],
398         'name' => $pbe['profile']['real_name'],
399         'email' => $pbe['profile']['email_address'],
400         'update_post_count' => empty($topic_info['count_posts']),
401         'ip' => $email_message->load_ip() ? $email_message->ip : $pbe['profile']['member_ip']
402     );
403 
404     // Make the post.
405     createPost($msgOptions, $topicOptions, $posterOptions);
406 
407     // We need the auto_notify setting, it may be theme based so pass the theme in use
408     $theme_settings = query_get_theme($pbe['profile']['id_member'], $pbe['profile']['id_theme'], $topic_info);
409     $auto_notify = isset($theme_settings['auto_notify']) ? $theme_settings['auto_notify'] : 0;
410 
411     // Turn notifications on or off
412     query_notifications($pbe['profile']['id_member'], $topic_info['id_board'], $topic_info['id_topic'], $auto_notify, $pbe['user_info']['permissions']);
413 
414     // Notify members who have notification turned on for this,
415     // but only if it's going to be approved
416     if ($becomesApproved)
417     {
418         require_once(SUBSDIR . '/Notification.subs.php');
419         sendNotifications($topic_info['id_topic'], 'reply', array(), array(), $pbe);
420     }
421 
422     return true;
423 }

425 /**
426  * Attempts to create a PM (reply) on the forum
427  *
428  * What it does
429  * - Checks if the user has permissions
430  * - Calls pbe_load_text to prepare text for the pm
431  * - Calls query_mark_pms to mark things as read
432  * - Uses sendpm to do the actual "sending"
433  * - Returns true if successful or false for any number of failures
434  *
435  * @package Maillist
436  * @param mixed[] $pbe array of pbe 'user_info' values
437  * @param Email_Parse $email_message
438  * @param mixed[] $pm_info
439  */
440 function pbe_create_pm($pbe, $email_message, $pm_info)
441 {
442     global $modSettings, $txt;
443 
444     // Can they send?
445     if (!$pbe['user_info']['is_admin'] && !in_array('pm_send', $pbe['user_info']['permissions']))
446         return pbe_emailError('error_pm_not_allowed', $email_message);
447 
448     // Convert the PM to BBC and Format the message
449     $html = $email_message->html_found;
450     $text = pbe_load_text($html, $email_message, $pbe);
451     if (empty($text))
452         return pbe_emailError('error_no_message', $email_message);
453 
454     // If they tried to attach a file, just say sorry
455     if (!empty($email_message->attachments) && !empty($modSettings['maillist_allow_attachments']) && !empty($modSettings['attachmentEnable']) && $modSettings['attachmentEnable'] == 1)
456         $text .= "\n\n" . $txt['error_no_pm_attach'] . "\n";
457 
458     // For sending the message...
459     $from = array(
460         'id' => $pbe['profile']['id_member'],
461         'name' => $pbe['profile']['real_name'],
462         'username' => $pbe['profile']['member_name']
463     );
464 
465     $pm_info['subject'] = strpos($pm_info['subject'], trim($pbe['response_prefix'])) === 0 ? $pm_info['subject'] : $pbe['response_prefix'] . $pm_info['subject'];
466 
467     // send/save the actual PM.
468     require_once(SUBSDIR . '/PersonalMessage.subs.php');
469     $pm_result = sendpm(array('to' => array($pm_info['id_member_from']), 'bcc' => array()), $pm_info['subject'], $text, true, $from, $pm_info['id_pm_head']);
470 
471     // Assuming all went well, mark this as read, replied to and update the unread counter
472     if (!empty($pm_result))
473         query_mark_pms($email_message, $pbe);
474 
475     return !empty($pm_result);
476 }

478 /**
479  * Create a new topic by email
480  *
481  * What it does:
482  * - Called by pbe_topic to create a new topic or by pbe_main to create a new topic via a subject change
483  * - checks posting permissions, but requires all email validation checks are complete
484  * - Calls pbe_load_text to prepare text for the post
485  * - Uses createPost to do the actual "posting"
486  * - Calls sendNotifications to announce the new post
487  * - Calls query_update_member_stats to show they did something
488  * - Requires the pbe, email_message and board_info arrays to be populated.
489  *
490  * @package Maillist
491  * @param mixed[] $pbe array of pbe 'user_info' values
492  * @param Email_Parse $email_message
493  * @param mixed[] $board_info
494  */
495 function pbe_create_topic($pbe, $email_message, $board_info)
496 {
497     global $txt, $modSettings;
498 
499     // It does not work like that
500     if (empty($pbe) || empty($email_message))
501         return false;
502 
503     // We have the board info, and their permissions - do they have a right to start a new topic?
504     $becomesApproved = true;
505     if (!$pbe['user_info']['is_admin'])
506     {
507         if (!in_array('postby_email', $pbe['user_info']['permissions']))
508             return pbe_emailError('error_permission', $email_message);
509         elseif ($modSettings['postmod_active'] && in_array('post_unapproved_topics', $pbe['user_info']['permissions']) && (!in_array('post_new', $pbe['user_info']['permissions'])))
510             $becomesApproved = false;

621
