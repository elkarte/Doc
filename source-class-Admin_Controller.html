 974     /**
 975      * This file allows the user to search the wiki documentation
 976      * for a little help.
 977      */
 978     public function action_search_doc()
 979     {
 980         global $context;
 981 
 982         $context['doc_apiurl'] = 'https://github.com/elkarte/Elkarte/wiki/api.php';
 983         $context['doc_scripturl'] = 'https://github.com/elkarte/Elkarte/wiki/';
 984 
 985         // Set all the parameters search might expect.
 986         $postVars = explode(' ', $context['search_term']);
 987 
 988         // Encode the search data.
 989         foreach ($postVars as $k => $v)
 990             $postVars[$k] = urlencode($v);
 991 
 992         // This is what we will send.
 993         $postVars = implode('+', $postVars);
 994 
 995         // Get the results from the doc site.
 996         require_once(SUBSDIR . '/Package.subs.php');
 997         // Demo URL:
 998         // https://github.com/elkarte/Elkarte/wiki/api.php?action=query&list=search&srprop=timestamp|snippet&format=xml&srwhat=text&srsearch=template+eval
 999         $search_results = fetch_web_data($context['doc_apiurl'] . '?action=query&list=search&srprop=timestamp|snippet&format=xml&srwhat=text&srsearch=' . $postVars);
1000 
1001         // If we didn't get any xml back we are in trouble - perhaps the doc site is overloaded?
1002         if (!$search_results || preg_match('~<' . '\?xml\sversion="\d+\.\d+"\?>\s*(<api>.+?</api>)~is', $search_results, $matches) != true)
1003             fatal_lang_error('cannot_connect_doc_site');
1004 
1005         $search_results = $matches[1];
1006 
1007         // Otherwise we simply walk through the XML and stick it in context for display.
1008         $context['search_results'] = array();
1009         require_once(SUBSDIR . '/XmlArray.class.php');
1010 
1011         // Get the results loaded into an array for processing!
1012         $results = new Xml_Array($search_results, false);
1013 
1014         // Move through the api layer.
1015         if (!$results->exists('api'))
1016             fatal_lang_error('cannot_connect_doc_site');
1017 
1018         // Are there actually some results?
1019         if ($results->exists('api/query/search/p'))
1020         {
1021             $relevance = 0;
1022             foreach ($results->set('api/query/search/p') as $result)
1023             {
1024                 $context['search_results'][$result->fetch('@title')] = array(
1025                     'title' => $result->fetch('@title'),
1026                     'relevance' => $relevance++,
1027                     'snippet' => str_replace('class=\'searchmatch\'', 'class="highlight"', un_htmlspecialchars($result->fetch('@snippet'))),
1028                 );
1029             }
1030         }
1031     }
