  25  * Check if the user is who he/she says he is.
  26  *
  27  * What it does:
  28  * - This function makes sure the user is who they claim to be by requiring a
  29  * password to be typed in every hour.
  30  * - This check can be turned on and off by the securityDisable setting.
  31  * - Uses the adminLogin() function of subs/Auth.subs.php if they need to login,
  32  * which saves all request (POST and GET) data.
  33  *
  34  * @param string $type = admin
  35  */
  36 function validateSession($type = 'admin')
  37 {
  38     global $modSettings, $user_info, $user_settings;
  39 
  40     // Guests are not welcome here.
  41     is_not_guest();
  42 
  43     // Validate what type of session check this is.
  44     $types = array();
  45     call_integration_hook('integrate_validateSession', array(&$types));
  46     $type = in_array($type, $types) || $type == 'moderate' ? $type : 'admin';
  47 
  48     // Set the lifetime for our admin session. Default is ten minutes.
  49     $refreshTime = 600;
  50 
  51     if (isset($modSettings['admin_session_lifetime']))
  52     {
  53         // Maybe someone is paranoid or mistakenly misconfigured the param? Give them at least 5 minutes.
  54         if ($modSettings['admin_session_lifetime'] < 5)
  55             $refreshTime = 300;
  56 
  57         // A whole day should be more than enough..
  58         elseif ($modSettings['admin_session_lifetime'] > 14400)
  59             $refreshTime = 86400;
  60 
  61         // We are between our internal min and max. Let's keep the board owner's value.
  62         else
  63             $refreshTime = $modSettings['admin_session_lifetime'] * 60;
  64     }
  65 
  66     // If we're using XML give an additional ten minutes grace as an admin can't log on in XML mode.
  67     if (isset($_GET['xml']))
  68         $refreshTime += 600;
  69 
  70     // Is the security option off?
  71     if (!empty($modSettings['securityDisable' . ($type != 'admin' ? '_' . $type : '')]))
  72         return;
  73 
  74     // If their admin or moderator session hasn't expired yet, let it pass, let the admin session trump a moderation one as well
  75     if ((!empty($_SESSION[$type . '_time']) && $_SESSION[$type . '_time'] + $refreshTime >= time()) || (!empty($_SESSION['admin_time']) && $_SESSION['admin_time'] + $refreshTime >= time()))
  76         return;
  77 
  78     require_once(SUBSDIR . '/Auth.subs.php');
  79 
  80     // Comming from the login screen
  81     if (isset($_POST[$type. '_pass']) || isset($_POST[$type . '_hash_pass']))
  82     {
  83         checkSession();
  84         validateToken('admin-login');
  85 
  86         // Hashed password, ahoy!
  87         if (isset($_POST[$type . '_hash_pass']) && strlen($_POST[$type . '_hash_pass']) === 64)
  88         {
  89             // Allow integration to verify the password
  90             $good_password = in_array(true, call_integration_hook('integrate_verify_password', array($user_info['username'], $_POST[$type . '_hash_pass'], true)), true);
  91 
  92             $password = $_POST[$type . '_hash_pass'];
  93             if ($good_password || validateLoginPassword($password, $user_info['passwd']))
  94             {
  95                 $_SESSION[$type . '_time'] = time();
  96                 unset($_SESSION['request_referer']);
  97 
  98                 return;
  99             }
 100         }
 101 
 102         // Posting the password... check it.
 103         if (isset($_POST[$type. '_pass']) && str_replace('*', '', $_POST[$type. '_pass']) !== '')
 104         {
 105             // Give integrated systems a chance to verify this password
 106             $good_password = in_array(true, call_integration_hook('integrate_verify_password', array($user_info['username'], $_POST[$type . '_pass'], false)), true);
 107 
 108             // Password correct?
 109             $password = $_POST[$type . '_pass'];
 110             if ($good_password || validateLoginPassword($password, $user_info['passwd'], $user_info['username']))
 111             {
 112                 $_SESSION[$type . '_time'] = time();
 113                 unset($_SESSION['request_referer']);
 114 
 115                 return;
 116             }
 117         }
 118     }
 119 
 120     // OpenID?
 121     if (!empty($user_settings['openid_uri']))
 122     {
 123         require_once(SUBSDIR . '/OpenID.subs.php');
 124         $openID = new OpenID();
 125         $openID->revalidate();
 126 
 127         $_SESSION[$type . '_time'] = time();
 128         unset($_SESSION['request_referer']);
 129         return;
 130     }
 131 
 132     // Better be sure to remember the real referer
 133     if (empty($_SESSION['request_referer']))
 134         $_SESSION['request_referer'] = isset($_SERVER['HTTP_REFERER']) ? @parse_url($_SERVER['HTTP_REFERER']) : array();
 135     elseif (empty($_POST))
 136         unset($_SESSION['request_referer']);
 137 
 138     // Need to type in a password for that, man.
 139     if (!isset($_GET['xml']))
 140         adminLogin($type);
 141     else
 142         return 'session_verify_fail';

 657 /**
 658  * Make sure the user's correct session was passed, and they came from here.
 659  *
 660  * What it does:
 661  * - Checks the current session, verifying that the person is who he or she should be.
 662  * - Also checks the referrer to make sure they didn't get sent here.
 663  * - Depends on the disableCheckUA setting, which is usually missing.
 664  * - Will check GET, POST, or REQUEST depending on the passed type.
 665  * - Also optionally checks the referring action if passed. (note that the referring action must be by GET.)
 666  *
 667  * @param string $type = 'post' (post, get, request)
 668  * @param string $from_action = ''
 669  * @param bool $is_fatal = true
 670  * @return string the error message if is_fatal is false.
 671  */
 672 function checkSession($type = 'post', $from_action = '', $is_fatal = true)
 673 {
 674     global $sc, $modSettings, $boardurl;
 675 
 676     // We'll work out user agent checks
 677     $req = request();
 678 
 679     // Is it in as $_POST['sc']?
 680     if ($type == 'post')
 681     {
 682         $check = isset($_POST[$_SESSION['session_var']]) ? $_POST[$_SESSION['session_var']] : (empty($modSettings['strictSessionCheck']) && isset($_POST['sc']) ? $_POST['sc'] : null);
 683         if ($check !== $sc)
 684             $error = 'session_timeout';
 685     }
 686     // How about $_GET['sesc']?
 687     elseif ($type === 'get')
 688     {
 689         $check = isset($_GET[$_SESSION['session_var']]) ? $_GET[$_SESSION['session_var']] : (empty($modSettings['strictSessionCheck']) && isset($_GET['sesc']) ? $_GET['sesc'] : null);
 690         if ($check !== $sc)
 691             $error = 'session_verify_fail';
 692     }
 693     // Or can it be in either?
 694     elseif ($type == 'request')
 695     {
 696         $check = isset($_GET[$_SESSION['session_var']]) ? $_GET[$_SESSION['session_var']] : (empty($modSettings['strictSessionCheck']) && isset($_GET['sesc']) ? $_GET['sesc'] : (isset($_POST[$_SESSION['session_var']]) ? $_POST[$_SESSION['session_var']] : (empty($modSettings['strictSessionCheck']) && isset($_POST['sc']) ? $_POST['sc'] : null)));
 697 
 698         if ($check !== $sc)
 699             $error = 'session_verify_fail';
 700     }
 701 
 702     // Verify that they aren't changing user agents on us - that could be bad.
 703     if ((!isset($_SESSION['USER_AGENT']) || $_SESSION['USER_AGENT'] != $req->user_agent()) && empty($modSettings['disableCheckUA']))
 704         $error = 'session_verify_fail';
 705 
 706     // Make sure a page with session check requirement is not being prefetched.
 707     if (isset($_SERVER['HTTP_X_MOZ']) && $_SERVER['HTTP_X_MOZ'] == 'prefetch')
 708     {
 709         ob_end_clean();
 710         header('HTTP/1.1 403 Forbidden');
 711         die;
 712     }
 713 
 714     // Check the referring site - it should be the same server at least!
 715     if (isset($_SESSION['request_referer']))
 716         $referrer = $_SESSION['request_referer'];
 717     else
 718         $referrer = isset($_SERVER['HTTP_REFERER']) ? @parse_url($_SERVER['HTTP_REFERER']) : array();
 719 
 720     if (!empty($referrer['host']))
