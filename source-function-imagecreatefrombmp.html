  27 /**
  28  * Create a thumbnail of the given source.
  29  *
  30  * @uses resizeImageFile() function to achieve the resize.
  31  *
  32  * @package Graphics
  33  * @param string $source
  34  * @param int $max_width
  35  * @param int $max_height
  36  * @return boolean whether the thumbnail creation was successful.
  37  */
  38 function createThumbnail($source, $max_width, $max_height)
  39 {
  40     global $modSettings;
  41 
  42     $destName = $source . '_thumb.tmp';
  43 
  44     // Do the actual resize.
  45     if (!empty($modSettings['attachment_thumb_png']))
  46         $success = resizeImageFile($source, $destName, $max_width, $max_height, 3);
  47     else
  48         $success = resizeImageFile($source, $destName, $max_width, $max_height);

  51     $destName = substr($destName, 0, -4);

  58         @touch($destName);
  59         return false;
  60     }

  94 /**
  95  * Searches through the file to see if there's potentialy harmful non-binary content.
  96  *
  97  * - if extensiveCheck is true, searches for asp/php short tags as well.
  98  *
  99  * @package Graphics
 100  * @param string $fileName
 101  * @param bool $extensiveCheck = false
 102  */
 103 function checkImageContents($fileName, $extensiveCheck = false)
 104 {
 105     $fp = fopen($fileName, 'rb');
 106     if (!$fp)
 107         fatal_lang_error('attach_timeout');
 108 
 109     $prev_chunk = '';
 110     while (!feof($fp))
 111     {
 112         $cur_chunk = fread($fp, 8192);
 113 
 114         // Though not exhaustive lists, better safe than sorry.
 115         if (!empty($extensiveCheck))
 116         {
 117             // Paranoid check. Some like it that way.
 118             if (preg_match('~(iframe|\\<\\?|\\<%|html|eval|body|script\W|(?-i)[CFZ]WS[\x01-\x0E])~i', $prev_chunk . $cur_chunk) === 1)
 119             {
 120                 fclose($fp);
 121                 return false;
 122             }
 123         }
 124         else
 125         {
 126             // Check for potential infection
 127             if (preg_match('~(iframe|(?<!cellTextIs)html|eval|body|script\W|(?-i)[CFZ]WS[\x01-\x0E])~i', $prev_chunk . $cur_chunk) === 1)
 128             {
 129                 fclose($fp);
 130                 return false;
 131             }
 132         }
 133         $prev_chunk = $cur_chunk;
 134     }
 135     fclose($fp);
 136 
 137     return true;
 138 }

 170 /**
 171  * See if we have enough memory to thumbnail an image
 172  *
 173  * @package Graphics
 174  * @param int[] $sizes image size
 175  */
 176 function imageMemoryCheck($sizes)
 177 {
 178     global $modSettings;
 179 
 180     // Doing the old 'set it and hope' way?
 181     if (empty($modSettings['attachment_thumb_memory']))
 182     {
 183         setMemoryLimit('128M');
 184         return true;
 185     }
 186 
 187     // Determine the memory requirements for this image, note: if you want to use an image formula
 188     // W x H x bits/8 x channels x Overhead factor
 189     // You will need to account for single bit images as GD expands them to an 8 bit and will greatly
 190     // overun the calculated value.
 191     // The 5 below is simply a shortcut of 8bpp, 3 channels, 1.66 overhead
 192     $needed_memory = ($sizes[0] * $sizes[1] * 5);
 193 
 194     // If we need more, lets try to get it
 195     return setMemoryLimit($needed_memory, true);

 198 /**
 199  * Resizes an image from a remote location or a local file.
 200  *
 201  * - Puts the resized image at the destination location.
 202  * - The file would have the format preferred_format if possible,
 203  * otherwise the default format is jpeg.
 204  *
 205  * @package Graphics
 206  * @param string $source
 207  * @param string $destination
 208  * @param int $max_width
 209  * @param int $max_height
 210  * @param int $preferred_format = 0
 211  */
 212 function resizeImageFile($source, $destination, $max_width, $max_height, $preferred_format = 0)
 213 {
 214     // Nothing to do without GD or IM
 215     if (!checkGD() && !checkImagick())
 216         return false;
 217 
 218     static $default_formats = array(
 219         '1' => 'gif',
 220         '2' => 'jpeg',
 221         '3' => 'png',
 222         '6' => 'bmp',
 223         '15' => 'wbmp'
 224     );
 225 
 226     require_once(SUBSDIR . '/Package.subs.php');
 227 
 228     // Get the image file, we have to work with something after all
 229     $fp_destination = fopen($destination, 'wb');
 230     if ($fp_destination && (substr($source, 0, 7) == 'http://' || substr($source, 0, 8) == 'https://'))
 231     {
 232         $fileContents = fetch_web_data($source);
 233 
 234         fwrite($fp_destination, $fileContents);
 235         fclose($fp_destination);
 236 
 237         $sizes = @getimagesize($destination);
 238     }
 239     elseif ($fp_destination)
 240     {
 241         $sizes = @getimagesize($source);
 242 
 243         $fp_source = fopen($source, 'rb');
 244         if ($fp_source !== false)
 245         {
 246             while (!feof($fp_source))
 247                 fwrite($fp_destination, fread($fp_source, 8192));
 248             fclose($fp_source);
 249         }
 250         else
 251             $sizes = array(-1, -1, -1);
 252         fclose($fp_destination);
 253     }
 254     // We can't get to the file.
 255     else
 256         $sizes = array(-1, -1, -1);
 257 
 258     // See if we have -or- can get the needed memory for this operation
 259     if (checkGD() && !imageMemoryCheck($sizes))
 260         return false;
 261 
 262     // A known and supported format?
 263     if (checkImagick() && isset($default_formats[$sizes[2]]))
 264         return resizeImage(null, $destination, null, null, $max_width, $max_height, true, $preferred_format);
 265     elseif (checkGD() && isset($default_formats[$sizes[2]]) && function_exists('imagecreatefrom' . $default_formats[$sizes[2]]))
 266     {
 267         $imagecreatefrom = 'imagecreatefrom' . $default_formats[$sizes[2]];
 268         if ($src_img = @$imagecreatefrom($destination))
 269             return resizeImage($src_img, $destination, imagesx($src_img), imagesy($src_img), $max_width === null ? imagesx($src_img) : $max_width, $max_height === null ? imagesy($src_img) : $max_height, true, $preferred_format);
 270     }
 271 
 272     return false;
 273 }

 275 /**
 276  * Resizes an image proportionally to fit within the defined max_width and max_height limits
 277  *
 278  * - Will do nothing to the image if the file fits within the size limits
 279  * - If Image Magick is present it will use those function over any GD solutions
 280  * - If GD2 is present, it'll use it to achieve better quality (imagecopyresampled)
 281  * - Saves the new image to destination_filename, in the preferred_format
 282  * if possible, default is jpeg.
 283  *
 284  * @uses GD
 285  * @uses Imagick
 286  *
 287  * @package Graphics
 288  * @param resource|null $src_img null for Imagick images, resource form imagecreatefrom for GD
 289  * @param string $destName
 290  * @param int $src_width
 291  * @param int $src_height
 292  * @param int $max_width
 293  * @param int $max_height
 294  * @param bool $force_resize = false
 295  * @param int $preferred_format = 0
 296  */
 297 function resizeImage($src_img, $destName, $src_width, $src_height, $max_width, $max_height, $force_resize = false, $preferred_format = 0)
 298 {
 299     global $gd2;
 300 
 301     if (checkImagick())
 302     {
 303         // These are the file formats we know about
 304         static $default_formats = array(
 305             '1' => 'gif',
 306             '2' => 'jpeg',
 307             '3' => 'png',
 308             '6' => 'bmp',
 309             '15' => 'wbmp'
 310         );
 311         $preferred_format = empty($preferred_format) || !isset($default_formats[$preferred_format]) ? 2 : $preferred_format;
 312 
 313         // Since Imagick can throw exceptions, lets catch them
 314         try
 315         {
 316             // Get a new instance of Imagick for use
 317             $imagick = new Imagick($destName);
 318 
 319             // Set the input and output image size
 320             $src_width = empty($src_width) ? $imagick->getImageWidth() : $src_width;
 321             $src_height = empty($src_height) ? $imagick->getImageHeight() : $src_height;
 322             $dest_width = empty($max_width) ? $src_width : $max_width;
 323             $dest_height = empty($max_height) ? $src_height : $max_height;
 324 
 325             // Create a new image in our prefered format and resize it if needed
 326             $imagick->setImageFormat($default_formats[$preferred_format]);
 327             $imagick->resizeImage($dest_width, $dest_height, Imagick::FILTER_LANCZOS, 1, true);
 328 
 329             // Save the new image in the destination location
 330             $success = $imagick->writeImage($destName);
 331 
 332             // Free resources associated with the Imagick object
 333             $imagick->destroy();
 334         }
 335         catch(Exception $e)
 336         {
 337             // Not currently used, but here is the error
 338             $success = $e->getMessage();
 339             $success = false;
 340         }
 341 
 342         return !empty($success);
 343     }
 344     elseif (checkGD())
 345     {
 346         $success = false;
 347 
 348         // Determine whether to resize to max width or to max height (depending on the limits.)
 349         if (!empty($max_width) || !empty($max_height))
 350         {
 351             if (!empty($max_width) && (empty($max_height) || $src_height * $max_width / $src_width <= $max_height))
 352             {
 353                 $dst_width = $max_width;
 354                 $dst_height = floor($src_height * $max_width / $src_width);
 355             }
 356             elseif (!empty($max_height))
 357             {
 358                 $dst_width = floor($src_width * $max_height / $src_height);
 359                 $dst_height = $max_height;
 360             }
 361 
 362             // Don't bother resizing if it's already smaller...
 363             if (!empty($dst_width) && !empty($dst_height) && ($dst_width < $src_width || $dst_height < $src_height || $force_resize))
 364             {
 365                 // (make a true color image, because it just looks better for resizing.)
 366                 if ($gd2)
 367                 {
 368                     $dst_img = imagecreatetruecolor($dst_width, $dst_height);
 369 
 370                     // Deal nicely with a PNG - because we can.
 371                     if ((!empty($preferred_format)) && ($preferred_format == 3))
 372                     {
 373                         imagealphablending($dst_img, false);
 374                         if (function_exists('imagesavealpha'))
 375                             imagesavealpha($dst_img, true);
 376                     }
 377                 }
 378                 else
 379                     $dst_img = imagecreate($dst_width, $dst_height);
 380 
 381                 // Resize it!
 382                 if ($gd2)
 383                     imagecopyresampled($dst_img, $src_img, 0, 0, 0, 0, $dst_width, $dst_height, $src_width, $src_height);
 384                 else
 385                     imagecopyresamplebicubic($dst_img, $src_img, 0, 0, 0, 0, $dst_width, $dst_height, $src_width, $src_height);
 386             }
 387             else
 388                 $dst_img = $src_img;
 389         }
 390         else
 391             $dst_img = $src_img;
 392 
 393         // Save the image as ...
 394         if (!empty($preferred_format) && ($preferred_format == 3) && function_exists('imagepng'))
 395             $success = imagepng($dst_img, $destName);
 396         elseif (!empty($preferred_format) && ($preferred_format == 1) && function_exists('imagegif'))
 397             $success = imagegif($dst_img, $destName);
 398         elseif (function_exists('imagejpeg'))
 399             $success = imagejpeg($dst_img, $destName, 80);
 400 
 401         // Free the memory.
 402         imagedestroy($src_img);
 403         if ($dst_img != $src_img)
 404             imagedestroy($dst_img);
 405 
 406         return $success;
 407     }
 408     // Without Imagick or GD, no image resizing at all.
 409     else
 410         return false;
 411 }

 413 /**
 414  * Copy / resize an image using GD bicubic methods
 415  *
 416  * - Used when imagecopyresample() is not available
 417  * - Uses bicubic resizing methods which are lower quality then imagecopyresample
 418  *
 419  * @package Graphics
 420  * @param resource $dst_img
 421  * @param resource $src_img
 422  * @param int $dst_x
 423  * @param int $dst_y
 424  * @param int $src_x
 425  * @param int $src_y
 426  * @param int $dst_w
 427  * @param int $dst_h
 428  * @param int $src_w
 429  * @param int $src_h
 430  */
 431 function imagecopyresamplebicubic($dst_img, $src_img, $dst_x, $dst_y, $src_x, $src_y, $dst_w, $dst_h, $src_w, $src_h)
 432 {
 433     $palsize = imagecolorstotal($src_img);
 434     for ($i = 0; $i < $palsize; $i++)
 435     {
 436         $colors = imagecolorsforindex($src_img, $i);
 437         imagecolorallocate($dst_img, $colors['red'], $colors['green'], $colors['blue']);
 438     }
 439 
 440     $scaleX = ($src_w - 1) / $dst_w;
 441     $scaleY = ($src_h - 1) / $dst_h;
 442 
 443     $scaleX2 = (int) $scaleX / 2;
 444     $scaleY2 = (int) $scaleY / 2;
 445 
 446     for ($j = $src_y; $j < $dst_h; $j++)
 447     {
 448         $sY = (int) $j * $scaleY;
 449         $y13 = $sY + $scaleY2;
 450 
 451         for ($i = $src_x; $i < $dst_w; $i++)
 452         {
 453             $sX = (int) $i * $scaleX;
 454             $x34 = $sX + $scaleX2;
 455 
 456             $color1 = imagecolorsforindex($src_img, imagecolorat($src_img, $sX, $y13));
 457             $color2 = imagecolorsforindex($src_img, imagecolorat($src_img, $sX, $sY));
 458             $color3 = imagecolorsforindex($src_img, imagecolorat($src_img, $x34, $y13));
 459             $color4 = imagecolorsforindex($src_img, imagecolorat($src_img, $x34, $sY));
 460 
 461             $red = ($color1['red'] + $color2['red'] + $color3['red'] + $color4['red']) / 4;
 462             $green = ($color1['green'] + $color2['green'] + $color3['green'] + $color4['green']) / 4;
 463             $blue = ($color1['blue'] + $color2['blue'] + $color3['blue'] + $color4['blue']) / 4;
 464 
 465             $color = imagecolorresolve($dst_img, $red, $green, $blue);
 466             if ($color == -1)
 467             {
 468                 if ($palsize++ < 256)
 469                     imagecolorallocate($dst_img, $red, $green, $blue);
 470                 $color = imagecolorclosest($dst_img, $red, $green, $blue);
 471             }
 472 
 473             imagesetpixel($dst_img, $i + $dst_x - $src_x, $j + $dst_y - $src_y, $color);
 474         }
 475     }
 476 }

 478 if (!function_exists('imagecreatefrombmp'))
 479 {
 480     /**
 481      * It is set only if it doesn't already exist (for forwards compatiblity.)
 482      *
 483      * - It only supports uncompressed bitmaps.
 484      * - It only supports standard windows bitmaps (no os/2 varients)
 485      * - Returns an image identifier representing the bitmap image
 486      * obtained from the given filename.
 487      *
 488      * @package Graphics
 489      * @param string $filename
 490      * @return resource
 491      */
 492     function imagecreatefrombmp($filename)
 493     {
 494         global $gd2;
 495 
 496         $fp = fopen($filename, 'rb');
 497 
 498         $errors = error_reporting(0);
 499 
 500         // Unpack the general information about the Bitmap Image File, first 14 Bytes
 501         $header = unpack('vtype/Vsize/Vreserved/Voffset', fread($fp, 14));
 502 
 503         // Upack the DIB header, it stores detailed information about the bitmap image the pixel format, 40 Bytes long
 504         $info = unpack('Vsize/Vwidth/Vheight/vplanes/vbits/Vcompression/Vimagesize/Vxres/Vyres/Vncolor/Vcolorimportant', fread($fp, 40));
 505 
 506         // Not a standard bitmap, bail out
 507         if ($header['type'] != 0x4D42)
 508             return false;
 509 
 510         // Create our image canvas with the given WxH
 511         if ($gd2)
 512             $dst_img = imagecreatetruecolor($info['width'], $info['height']);
 513         else
 514             $dst_img = imagecreate($info['width'], $info['height']);
 515 
 516         // Color bitCounts 1,4,8 have palette information we use
 517         if ($info['bits'] == 1 || $info['bits'] == 4 || $info['bits'] == 8)
 518         {
 519             $palette_size = $header['offset'] - 54;
 520 
 521             // Read the palette data
 522             $palettedata = fread($fp, $palette_size);
 523 
 524             // Create the rgb color array
 525             $palette = array();
 526             $n = 0;
 527             for ($j = 0; $j < $palette_size; $j++)
 528             {
 529                 $b = ord($palettedata[$j++]);
 530                 $g = ord($palettedata[$j++]);
 531                 $r = ord($palettedata[$j++]);
 532 
 533                 $palette[$n++] = imagecolorallocate($dst_img, $r, $g, $b);
 534             }
 535         }
 536 
 537         $scan_line_size = ($info['bits'] * $info['width'] + 7) >> 3;
 538         $scan_line_align = $scan_line_size & 3 ? 4 - ($scan_line_size & 3) : 0;
 539 
 540         for ($y = 0, $l = $info['height'] - 1; $y < $info['height']; $y++, $l--)
 541         {
 542             fseek($fp, $header['offset'] + ($scan_line_size + $scan_line_align) * $l);
 543             $scan_line = fread($fp, $scan_line_size);
 544 
 545             if (strlen($scan_line) < $scan_line_size)
 546                 continue;
 547 
 548             // 32 bits per pixel
 549             if ($info['bits'] == 32)
 550             {
 551                 $x = 0;
 552                 for ($j = 0; $j < $scan_line_size; $x++)
 553                 {
 554                     $b = ord($scan_line[$j++]);
 555                     $g = ord($scan_line[$j++]);
 556                     $r = ord($scan_line[$j++]);
 557                     $j++;
 558 
 559                     $color = imagecolorexact($dst_img, $r, $g, $b);
 560                     if ($color == -1)
 561                     {
 562                         $color = imagecolorallocate($dst_img, $r, $g, $b);
 563 
 564                         // Gah!  Out of colors?  Stupid GD 1... try anyhow.
 565                         if ($color == -1)
 566                             $color = imagecolorclosest($dst_img, $r, $g, $b);
 567                     }
 568 
 569                     imagesetpixel($dst_img, $x, $y, $color);
 570                 }
 571             }
 572             // 24 bits per pixel
 573             elseif ($info['bits'] == 24)
 574             {
 575                 $x = 0;
 576                 for ($j = 0; $j < $scan_line_size; $x++)
 577                 {
 578                     $b = ord($scan_line[$j++]);
 579                     $g = ord($scan_line[$j++]);
 580                     $r = ord($scan_line[$j++]);
 581 
 582                     $color = imagecolorexact($dst_img, $r, $g, $b);
 583                     if ($color == -1)
 584                     {
 585                         $color = imagecolorallocate($dst_img, $r, $g, $b);
 586 
 587                         // Gah!  Out of colors?  Stupid GD 1... try anyhow.
 588                         if ($color == -1)
 589                             $color = imagecolorclosest($dst_img, $r, $g, $b);
 590                     }
 591 
 592                     imagesetpixel($dst_img, $x, $y, $color);
 593                 }
 594             }
 595             // 16 bits per pixel
 596             elseif ($info['bits'] == 16)
 597             {
 598                 $x = 0;
 599                 for ($j = 0; $j < $scan_line_size; $x++)
 600                 {
 601                     $b1 = ord($scan_line[$j++]);
 602                     $b2 = ord($scan_line[$j++]);
 603 
 604                     $word = $b2 * 256 + $b1;
 605 
 606                     $b = (($word & 31) * 255) / 31;
 607                     $g = ((($word >> 5) & 31) * 255) / 31;
 608                     $r = ((($word >> 10) & 31) * 255) / 31;
 609 
 610                     // Scale the image colors up properly.
 611                     $color = imagecolorexact($dst_img, $r, $g, $b);
 612                     if ($color == -1)
 613                     {
 614                         $color = imagecolorallocate($dst_img, $r, $g, $b);
 615 
 616                         // Gah!  Out of colors?  Stupid GD 1... try anyhow.
 617                         if ($color == -1)
 618                             $color = imagecolorclosest($dst_img, $r, $g, $b);
 619                     }
 620 
 621                     imagesetpixel($dst_img, $x, $y, $color);
 622                 }
 623             }
 624             // 8 bits per pixel
 625             elseif ($info['bits'] == 8)
 626             {
 627                 $x = 0;
 628                 for ($j = 0; $j < $scan_line_size; $x++)
 629                     imagesetpixel($dst_img, $x, $y, $palette[ord($scan_line[$j++])]);
 630             }
 631             // 4 bits per pixel
 632             elseif ($info['bits'] == 4)
 633             {
 634                 $x = 0;
 635                 for ($j = 0; $j < $scan_line_size; $x++)
 636                 {
 637                     $byte = ord($scan_line[$j++]);
 638 
 639                     imagesetpixel($dst_img, $x, $y, $palette[(int) ($byte / 16)]);
 640                     if (++$x < $info['width'])
 641                         imagesetpixel($dst_img, $x, $y, $palette[$byte & 15]);
 642                 }
 643             }
 644             // 1 bit
 645             elseif ($info['bits'] == 1)
 646             {
 647                 $x = 0;
 648                 for ($j = 0; $j < $scan_line_size; $x++)
 649                 {
 650                     $byte = ord($scan_line[$j++]);
 651 
 652                     imagesetpixel($dst_img, $x, $y, $palette[(($byte) & 128) != 0]);
 653                     for ($shift = 1; $shift < 8; $shift++)
 654                     {
 655                         if (++$x < $info['width'])
 656                             imagesetpixel($dst_img, $x, $y, $palette[(($byte << $shift) & 128) != 0]);
 657                     }
 658                 }
 659             }
 660         }
 661 
 662         fclose($fp);
 663 
 664         error_reporting($errors);
 665 
 666         return $dst_img;
 667     }
