 25 /**
 26  * Get a list of versions that are currently installed on the server.
 27  *
 28  * @package Admin
 29  * @param string[] $checkFor
 30  */
 31 function getServerVersions($checkFor)
 32 {
 33     global $txt, $memcached, $modSettings;
 34 
 35     $db = database();
 36 
 37     loadLanguage('Admin');
 38 
 39     $versions = array();
 40 
 41     // Is GD available?  If it is, we should show version information for it too.
 42     if (in_array('gd', $checkFor) && function_exists('gd_info'))
 43     {
 44         $temp = gd_info();
 45         $versions['gd'] = array('title' => $txt['support_versions_gd'], 'version' => $temp['GD Version']);
 46     }
 47 
 48     // Why not have a look at ImageMagick? If it is, we should show version information for it too.
 49     if (in_array('imagick', $checkFor) && class_exists('Imagick'))
 50     {
 51         $temp = new Imagick;
 52         $temp2 = $temp->getVersion();
 53         $versions['imagick'] = array('title' => $txt['support_versions_imagick'], 'version' => $temp2['versionString']);
 54     }
 55 
 56     // Now lets check for the Database.
 57     if (in_array('db_server', $checkFor))
 58     {
 59         $conn = $db->connection();
 60         if (empty($conn))
 61             trigger_error('getServerVersions(): you need to be connected to the database in order to get its server version', E_USER_NOTICE);
 62         else
 63         {
 64             $versions['db_server'] = array('title' => sprintf($txt['support_versions_db'], $db->db_title()), 'version' => '');
 65             $versions['db_server']['version'] = $db->db_server_version();
 66         }
 67     }
 68 
 69     // If we're using memcache we need the server info.
 70     if (empty($memcached) && function_exists('memcache_get') && isset($modSettings['cache_memcached']) && trim($modSettings['cache_memcached']) != '')
 71     {
 72         require_once(SUBSDIR . '/Cache.subs.php');
 73         get_memcached_server();
 74     }
 75     else
 76     {
 77         if (($key = array_search('memcache', $checkFor)) !== false)
 78             unset($checkFor[$key]);
 79     }
 80 
 81     // Check to see if we have any accelerators installed...
 82     if (in_array('mmcache', $checkFor) && defined('MMCACHE_VERSION'))
 83         $versions['mmcache'] = array('title' => 'Turck MMCache', 'version' => MMCACHE_VERSION);
 84     if (in_array('eaccelerator', $checkFor) && defined('EACCELERATOR_VERSION'))
 85         $versions['eaccelerator'] = array('title' => 'eAccelerator', 'version' => EACCELERATOR_VERSION);
 86     if (in_array('zend', $checkFor) && function_exists('zend_shm_cache_store'))
 87         $versions['zend'] = array('title' => 'Zend SHM Accelerator', 'version' => zend_version());
 88     if (in_array('apc', $checkFor) && extension_loaded('apc'))
 89         $versions['apc'] = array('title' => 'Alternative PHP Cache', 'version' => phpversion('apc'));
 90     if (in_array('memcache', $checkFor) && function_exists('memcache_set'))
 91         $versions['memcache'] = array('title' => 'Memcached', 'version' => empty($memcached) ? '???' : memcache_get_version($memcached));
 92     if (in_array('xcache', $checkFor) && function_exists('xcache_set'))
 93         $versions['xcache'] = array('title' => 'XCache', 'version' => XCACHE_VERSION);
 94     if (in_array('opcache', $checkFor) && extension_loaded('Zend OPcache'))
 95     {
 96         $opcache_config = @opcache_get_configuration();
 97         if (!empty($opcache_config['directives']['opcache.enable']))
 98             $versions['opcache'] = array('title' => $opcache_config['version']['opcache_product_name'], 'version' => $opcache_config['version']['version']);
 99     }
100 
101     // PHP Version
102     if (in_array('php', $checkFor))
103         $versions['php'] = array('title' => 'PHP', 'version' => PHP_VERSION . ' (' . php_sapi_name() . ')', 'more' => '?action=admin;area=serversettings;sa=phpinfo');
104 
105     // Server info
106     if (in_array('server', $checkFor))
107     {
108         $req = request();
109         $versions['server'] = array('title' => $txt['support_versions_server'], 'version' => $req->server_software());
110 
111         // Compute some system info, if we can
112         $versions['server_name'] = array('title' => $txt['support_versions'], 'version' => php_uname());
113         $loading = detectServerLoad();
114         if ($loading !== false)
115             $versions['server_load'] = array('title' => $txt['load_balancing_settings'], 'version' => $loading);
116     }
117 
118     return $versions;
119 }

196 /**
197  * Search through source, theme and language files to determine their version.
198  * Get detailed version information about the physical Elk files on the server.
199  *
200  * What it does:
201  * - the input parameter allows to set whether to include SSI.php and whether
202  *   the results should be sorted.
203  * - returns an array containing information on source files, templates and
204  *   language files found in the default theme directory (grouped by language).
205  * - options include include_ssi, include_subscriptions, sort_results
206  *
207  * @package Admin
208  * @param mixed[] $versionOptions associative array of options
209  * @return array
210  */
211 function getFileVersions(&$versionOptions)
212 {
213     global $settings;
214 
215     // Default place to find the languages would be the default theme dir.
216     $lang_dir = $settings['default_theme_dir'] . '/languages';
217 
218     $version_info = array(
219         'file_versions' => array(),
220         'file_versions_admin' => array(),
221         'file_versions_controllers' => array(),
222         'file_versions_database' => array(),
223         'file_versions_subs' => array(),
224         'default_template_versions' => array(),
225         'template_versions' => array(),
226         'default_language_versions' => array(),
227     );
228 
229     // The comment looks rougly like... that.
230     $version_regex = '~\*\s@version\s+(.+)[\s]{2}~i';
231     $unknown_version = '??';
232 
233     // Find the version in SSI.php's file header.
234     if (!empty($versionOptions['include_ssi']) && file_exists(BOARDDIR . '/SSI.php'))
235     {
236         $header = file_get_contents(BOARDDIR . '/SSI.php', false, null, 0, 768);
237         if (preg_match($version_regex, $header, $match) == 1)
238             $version_info['file_versions']['SSI.php'] = $match[1];
239         // Not found!  This is bad.
240         else
241             $version_info['file_versions']['SSI.php'] = $unknown_version;
242     }
243 
244     // Do the paid subscriptions handler?
245     if (!empty($versionOptions['include_subscriptions']) && file_exists(BOARDDIR . '/subscriptions.php'))
246     {
247         $header = file_get_contents(BOARDDIR . '/subscriptions.php', false, null, 0, 768);
248         if (preg_match($version_regex, $header, $match) == 1)
249             $version_info['file_versions']['subscriptions.php'] = $match[1];
250         // If we haven't how do we all get paid?
251         else
252             $version_info['file_versions']['subscriptions.php'] = $unknown_version;
253     }
254 
255     // Load all the files in the sources and its sub directorys
256     $directories = array(
257         'file_versions' => SOURCEDIR,
258         'file_versions_admin' => ADMINDIR,
259         'file_versions_controllers' => CONTROLLERDIR,
260         'file_versions_database' => SOURCEDIR . '/database',
261         'file_versions_subs' => SUBSDIR,
262         'file_versions_lib' => EXTDIR
263     );
264     foreach ($directories as $area => $dir)
265     {
266         $sources_dir = dir($dir);
267         while ($entry = $sources_dir->read())
268         {
269             if (substr($entry, -4) === '.php' && !is_dir($dir . '/' . $entry) && $entry !== 'index.php' && $entry !== 'sphinxapi.php')
270             {
271                 // Read the first 4k from the file.... enough for the header.
272                 $header = file_get_contents($dir . '/' . $entry, false, null, 0, 768);
273 
274                 // Look for the version comment in the file header.
275                 if (preg_match($version_regex, $header, $match))
276                     $version_info[$area][$entry] = $match[1];
277                 // It wasn't found, but the file was... show a $unknown_version.
278                 else
279                     $version_info[$area][$entry] = '??';
280             }
281         }
282         $sources_dir->close();
283     }
284 
285     // Load all the files in the default template directory - and the current theme if applicable.
286     $directories = array('default_template_versions' => $settings['default_theme_dir']);
287     if ($settings['theme_id'] != 1)
288         $directories += array('template_versions' => $settings['theme_dir']);
289 
290     foreach ($directories as $type => $dirname)
291     {
292         $this_dir = dir($dirname);
293         while ($entry = $this_dir->read())
294         {
295             if (substr($entry, -12) == 'template.php' && !is_dir($dirname . '/' . $entry))
296             {
297                 // Read the first 768 bytes from the file.... enough for the header.
298                 $header = file_get_contents($dirname . '/' . $entry, false, null, 0, 768);
299 
300                 // Look for the version comment in the file header.
301                 if (preg_match($version_regex, $header, $match) == 1)
302                     $version_info[$type][$entry] = $match[1];
303                 // It wasn't found, but the file was... show a '??'.
304                 else
305                     $version_info[$type][$entry] = $unknown_version;
306             }
307         }
308         $this_dir->close();
309     }
310 
311     // Load up all the files in the default language directory and sort by language.
312     $this_dir = dir($lang_dir);
313     while ($path = $this_dir->read())
314     {
315         if ($path == '.' || $path == '..')
316             continue;
317 
318         if (is_dir($lang_dir . '/' . $path))
319         {
320             $language = $path;
321             $this_lang_path = $lang_dir . '/' . $language;
322             $this_lang = dir($this_lang_path);
323             while ($entry = $this_lang->read())
324             {
325                 if (substr($entry, -4) == '.php' && $entry != 'index.php' && !is_dir($this_lang_path . '/' . $entry))
326                 {
327                     // Read the first 768 bytes from the file.... enough for the header.
328                     $header = file_get_contents($this_lang_path . '/' . $entry, false, null, 0, 768);
329 
330                     // Split the file name off into useful bits.
331                     list ($name, $language) = explode('.', $entry);
332 
333                     // Look for the version comment in the file header.
334                     if (preg_match('~(?://|/\*)\s*Version:\s+(.+?);\s*' . preg_quote($name, '~') . '(?:[\s]{2}|\*/)~i', $header, $match) == 1)
335                         $version_info['default_language_versions'][$language][$name] = $match[1];
336                     // It wasn't found, but the file was... show a '??'.
337                     else
338                         $version_info['default_language_versions'][$language][$name] = $unknown_version;
339                 }
340             }
341         }
342     }
343     $this_dir->close();
344 
345     // Sort the file versions by filename.
346     if (!empty($versionOptions['sort_results']))
347     {
348         ksort($version_info['file_versions']);
349         ksort($version_info['file_versions_admin']);
350         ksort($version_info['file_versions_controllers']);
351         ksort($version_info['file_versions_database']);
352         ksort($version_info['file_versions_subs']);
353         ksort($version_info['default_template_versions']);
354         ksort($version_info['template_versions']);
355         ksort($version_info['default_language_versions']);
356 
357         // For languages sort each language too.
358         foreach ($version_info['default_language_versions'] as $language => $dummy)
359             ksort($version_info['default_language_versions'][$language]);
360     }
361 
362     return $version_info;
363 }

410 /**
411  * Send all the administrators a lovely email.
412  *
413  * What it does:
414  * - It loads all users who are admins or have the admin forum permission.
415  * - It uses the email template and replacements passed in the parameters.
416  * - It sends them an email.
417  *
418  * @package Admin
419  * @param string $template
420  * @param mixed[] $replacements
421  * @param int[] $additional_recipients
422  */
423 function emailAdmins($template, $replacements = array(), $additional_recipients = array())
424 {
425     global $language, $modSettings;
426 
427     $db = database();
428 
429     // We certainly want this.
430     require_once(SUBSDIR . '/Mail.subs.php');
431 
432     // Load all groups which are effectively admins.
433     $request = $db->query('', '
434         SELECT id_group
435         FROM {db_prefix}permissions
436         WHERE permission = {string:admin_forum}
437             AND add_deny = {int:add_deny}
438             AND id_group != {int:id_group}',
439         array(
440             'add_deny' => 1,
441             'id_group' => 0,
442             'admin_forum' => 'admin_forum',
443         )
444     );
445     $groups = array(1);
446     while ($row = $db->fetch_assoc($request))
447         $groups[] = $row['id_group'];
448     $db->free_result($request);
449 
450     $request = $db->query('', '
451         SELECT id_member, member_name, real_name, lngfile, email_address
452         FROM {db_prefix}members
453         WHERE (id_group IN ({array_int:group_list}) OR FIND_IN_SET({raw:group_array_implode}, additional_groups) != 0)
454             AND notify_types != {int:notify_types}
455         ORDER BY lngfile',
456         array(
457             'group_list' => $groups,
458             'notify_types' => 4,
459             'group_array_implode' => implode(', additional_groups) != 0 OR FIND_IN_SET(', $groups),
460         )
461     );
462     $emails_sent = array();
463     while ($row = $db->fetch_assoc($request))
464     {
465         // Stick their particulars in the replacement data.
466         $replacements['IDMEMBER'] = $row['id_member'];
467         $replacements['REALNAME'] = $row['member_name'];
468         $replacements['USERNAME'] = $row['real_name'];
469 
470         // Load the data from the template.
471         $emaildata = loadEmailTemplate($template, $replacements, empty($row['lngfile']) || empty($modSettings['userLanguage']) ? $language : $row['lngfile']);
472 
473         // Then send the actual email.
474         sendmail($row['email_address'], $emaildata['subject'], $emaildata['body'], null, null, false, 1);
475 
476         // Track who we emailed so we don't do it twice.
477         $emails_sent[] = $row['email_address'];
478     }
479     $db->free_result($request);
480 
481     // Any additional users we must email this to?
482     if (!empty($additional_recipients))
483         foreach ($additional_recipients as $recipient)
484         {
485             if (in_array($recipient['email'], $emails_sent))
486                 continue;
487 
488             $replacements['IDMEMBER'] = $recipient['id'];
489             $replacements['REALNAME'] = $recipient['name'];
490             $replacements['USERNAME'] = $recipient['name'];
491 
492             // Load the template again.
493             $emaildata = loadEmailTemplate($template, $replacements, empty($recipient['lang']) || empty($modSettings['userLanguage']) ? $language : $recipient['lang']);
494 
495             // Send off the email.
496             sendmail($recipient['email'], $emaildata['subject'], $emaildata['body'], null, null, false, 1);
497         }
498 }
