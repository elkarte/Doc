 23 /**
 24  * Sets the login cookie and session based on the id_member and password passed.
 25  *
 26  * What it does:
 27  * - password should be already encrypted with the cookie salt.
 28  * - logs the user out if id_member is zero.
 29  * - sets the cookie and session to last the number of seconds specified by cookie_length.
 30  * - when logging out, if the globalCookies setting is enabled, attempts to clear the subdomain's cookie too.
 31  *
 32  * @pachage Authorization
 33  * @param int $cookie_length
 34  * @param int $id The id of the member
 35  * @param string $password = ''
 36  */
 37 function setLoginCookie($cookie_length, $id, $password = '')
 38 {
 39     global $cookiename, $boardurl, $modSettings;
 40 
 41     // If changing state force them to re-address some permission caching.
 42     $_SESSION['mc']['time'] = 0;
 43 
 44     // Let's be sure it is an int to simplify the regexp used to validate the cookie
 45     $id = (int) $id;
 46 
 47     // The cookie may already exist, and have been set with different options.
 48     $cookie_state = (empty($modSettings['localCookies']) ? 0 : 1) | (empty($modSettings['globalCookies']) ? 0 : 2);
 49     if (isset($_COOKIE[$cookiename]) && preg_match('~^a:[34]:\{i:0;i:\d{1,8};i:1;s:(0|40):"([a-fA-F0-9]{40})?";i:2;[id]:\d{1,14};(i:3;i:\d;)?\}$~', $_COOKIE[$cookiename]) === 1)
 50     {
 51         $array = @unserialize($_COOKIE[$cookiename]);
 52 
 53         // Out with the old, in with the new!
 54         if (isset($array[3]) && $array[3] != $cookie_state)
 55         {
 56             $cookie_url = url_parts($array[3] & 1 > 0, $array[3] & 2 > 0);
 57             elk_setcookie($cookiename, serialize(array(0, '', 0)), time() - 3600, $cookie_url[1], $cookie_url[0]);
 58         }
 59     }
 60 
 61     // Get the data and path to set it on.
 62     $data = serialize(empty($id) ? array(0, '', 0) : array($id, $password, time() + $cookie_length, $cookie_state));
 63     $cookie_url = url_parts(!empty($modSettings['localCookies']), !empty($modSettings['globalCookies']));
 64 
 65     // Set the cookie, $_COOKIE, and session variable.
 66     elk_setcookie($cookiename, $data, time() + $cookie_length, $cookie_url[1], $cookie_url[0]);
 67 
 68     // If subdomain-independent cookies are on, unset the subdomain-dependent cookie too.
 69     if (empty($id) && !empty($modSettings['globalCookies']))
 70         elk_setcookie($cookiename, $data, time() + $cookie_length, $cookie_url[1], '');
 71 
 72     // Any alias URLs?  This is mainly for use with frames, etc.
 73     if (!empty($modSettings['forum_alias_urls']))
 74     {
 75         $aliases = explode(',', $modSettings['forum_alias_urls']);
 76 
 77         $temp = $boardurl;
 78         foreach ($aliases as $alias)
 79         {
 80             // Fake the $boardurl so we can set a different cookie.
 81             $alias = strtr(trim($alias), array('http://' => '', 'https://' => ''));
 82             $boardurl = 'http://' . $alias;
 83 
 84             $cookie_url = url_parts(!empty($modSettings['localCookies']), !empty($modSettings['globalCookies']));
 85 
 86             if ($cookie_url[0] == '')
 87                 $cookie_url[0] = strtok($alias, '/');
 88 
 89             elk_setcookie($cookiename, $data, time() + $cookie_length, $cookie_url[1], $cookie_url[0]);
 90         }
 91 
 92         $boardurl = $temp;
 93     }
 94 
 95     $_COOKIE[$cookiename] = $data;
 96 
 97     // Make sure the user logs in with a new session ID.
 98     if (!isset($_SESSION['login_' . $cookiename]) || $_SESSION['login_' . $cookiename] !== $data)
 99     {
100         // We need to meddle with the session.
101         require_once(SOURCEDIR . '/Session.php');
102 
103         // Backup the old session.
104         $oldSessionData = $_SESSION;
105 
106         // Remove the old session data and file / db entry
107         $_SESSION = array();
108         session_destroy();
109 
110         // Recreate and restore the new session.
111         loadSession();
112 
113         // Get a new session id, and load it with the data
114         session_regenerate_id();
115         $_SESSION = $oldSessionData;
116 
117         $_SESSION['login_' . $cookiename] = $data;
118     }

122  * Get the domain and path for the cookie
123  *
124  * What it does:
125  * - normally, local and global should be the localCookies and globalCookies settings, respectively.
126  * - uses boardurl to determine these two things.

130  * @param bool $global

132 function url_parts($local, $global)
133 {
134     global $boardurl, $modSettings;
135 
136     // Parse the URL with PHP to make life easier.
137     $parsed_url = parse_url($boardurl);
138 
139     // Is local cookies off?
140     if (empty($parsed_url['path']) || !$local)
141         $parsed_url['path'] = '';
142 
143     if (!empty($modSettings['globalCookiesDomain']) && strpos($boardurl, $modSettings['globalCookiesDomain']) !== false)
144         $parsed_url['host'] = $modSettings['globalCookiesDomain'];
145 
146     // Globalize cookies across domains (filter out IP-addresses)?
147     elseif ($global && preg_match('~^\d{1,3}(\.\d{1,3}){3}$~', $parsed_url['host']) == 0 && preg_match('~(?:[^\.]+\.)?([^\.]{2,}\..+)\z~i', $parsed_url['host'], $parts) == 1)
148             $parsed_url['host'] = '.' . $parts[1];
149 
150     // We shouldn't use a host at all if both options are off.
151     elseif (!$local && !$global)

153 
154     // The host also shouldn't be set if there aren't any dots in it.
155     elseif (!isset($parsed_url['host']) || strpos($parsed_url['host'], '.') === false)
156         $parsed_url['host'] = '';
157 
158     return array($parsed_url['host'], $parsed_url['path'] . '/');

499 /**
500  * Checks whether a password meets the current forum rules
501  *
502  * What it does:
503  * - called when registering/choosing a password.
504  * - checks the password obeys the current forum settings for password strength.
505  * - if password checking is enabled, will check that none of the words in restrict_in appear in the password.
506  * - returns an error identifier if the password is invalid, or null.
507  *
508  * @pachage Authorization
509  * @param string $password
510  * @param string $username
511  * @param string[] $restrict_in = array()
512  * @return string an error identifier if the password is invalid
513  */
514 function validatePassword($password, $username, $restrict_in = array())
515 {
516     global $modSettings;
517 
518     // Perform basic requirements first.
519     if (Util::strlen($password) < (empty($modSettings['password_strength']) ? 4 : 8))
520         return 'short';
521 
522     // Is this enough?
523     if (empty($modSettings['password_strength']))
524         return null;
525 
526     // Otherwise, perform the medium strength test - checking if password appears in the restricted string.
527     if (preg_match('~\b' . preg_quote($password, '~') . '\b~', implode(' ', $restrict_in)) != 0)
528         return 'restricted_words';
529     elseif (Util::strpos($password, $username) !== false)
530         return 'restricted_words';
531 
532     // If just medium, we're done.
533     if ($modSettings['password_strength'] == 1)
534         return null;
535 
536     // Otherwise, hard test next, check for numbers and letters, uppercase too.
537     $good = preg_match('~(\D\d|\d\D)~', $password) != 0;
538     $good &= Util::strtolower($password) != $password;
539 
540     return $good ? null : 'chars';
541 }

546  * What it does:
547  * - called when logging in or whenever a password needs to be validated for a user
548  * - used to generate a new hash for the db, used during registration or any password changes
549  * - if a non SHA256 password is sent, will generate one with SHA256(user + password) and return it in password

551  * @pachage Authorization
552  * @param string $password user password if not already 64 characters long will be SHA256 with the user name

555  * @param boolean $returnhash flag to determine if we are returning a hash suitable for the database
556  */
557 function validateLoginPassword(&$password, $hash, $user = '', $returnhash = false)
558 {
559     // Our hashing controller
560     require_once(EXTDIR . '/PasswordHash.php');
561 
562     // Base-2 logarithm of the iteration count used for password stretching, the
563     // higher the number the more secure and CPU time consuming
564     $hash_cost_log2 = 10;
565 
566     // Do we require the hashes to be portable to older systems (less secure)?
567     $hash_portable = false;
568 
569     // Get an instance of the hasher
570     $hasher = new PasswordHash($hash_cost_log2, $hash_portable);
571 
572     // Guilty until we know otherwise
573     $passhash = false;
574 
575     // If the password is not 64 characters, lets make it a (SHA-256)
576     if (strlen($password) !== 64)
577         $password = hash('sha256', Util::strtolower($user) . un_htmlspecialchars($password));
578 
579     // They need a password hash, something to save in the db?
580     if ($returnhash)
581     {
582         $passhash = $hasher->HashPassword($password);
583 
584         // Something is not right, we can not generate a valid hash thats <20 characters
585         if (strlen($passhash) < 20)
586             $passhash = false;
587     }
588     // Or doing a password check?
589     else
590         $passhash = (bool) $hasher->CheckPassword($password, $hash);

592     unset($hasher);
