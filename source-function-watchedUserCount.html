 230     {
 231         $filter_board = array((int) $brd);
 232         $approve_boards = $approve_boards == array(0) ? $filter_board : array_intersect($approve_boards, $filter_board);
 233     }
 234 
 235     // Work out the query
 236     if ($approve_boards == array(0))
 237         $approve_query = '';
 238     elseif (!empty($approve_boards))
 239         $approve_query = ' AND m.id_board IN (' . implode(',', $approve_boards) . ')';
 240     else
 241         $approve_query = ' AND 1=0';
 242 
 243     // Set up the cache key for this permissions level
 244     $cache_key = md5($user_info['query_see_board'] . $approve_query . $user_info['mod_cache']['bq'] . $user_info['mod_cache']['gq'] . $user_info['mod_cache']['mq'] . (int) allowedTo('approve_emails') . '_' . (int) allowedTo('moderate_forum'));
 245 
 246     if (isset($menu_errors[$cache_key]))
 247         return $menu_errors[$cache_key];
 248 
 249     // If its been cached, guess what, thats right use it!
 250     $temp = cache_get_data('num_menu_errors', 900);
 251     if ($temp === null || !isset($temp[$cache_key]))
 252     {
 253         // Starting out with nothing is a good start
 254         $menu_errors[$cache_key] = array(
 255             'memberreq' => 0,
 256             'groupreq' => 0,
 257             'attachments' => 0,
 258             'reports' => 0,
 259             'emailmod' => 0,
 260             'postmod' => 0,
 261             'topics' => 0,
 262             'posts' => 0,
 263         );
 264 
 265         if ($modSettings['postmod_active'] && !empty($approve_boards))
 266         {
 267             $totals = recountUnapprovedPosts($approve_query);
 268             $menu_errors[$cache_key]['posts'] = $totals['posts'];
 269             $menu_errors[$cache_key]['topics'] = $totals['topics'];
 270 
 271             // Totals for the menu item unapproved posts and topics
 272             $menu_errors[$cache_key]['postmod'] = $menu_errors[$cache_key]['topics'] + $menu_errors[$cache_key]['posts'];
 273         }
 274 
 275         // Attachments
 276         if ($modSettings['postmod_active'] && !empty($approve_boards))
 277         {
 278             require_once(SUBSDIR . '/ManageAttachments.subs.php');
 279             $menu_errors[$cache_key]['attachments'] = list_getNumUnapprovedAttachments($approve_query);
 280         }
 281 
 282         // Reported posts
 283         if (!empty($user_info['mod_cache']) && $user_info['mod_cache']['bq'] != '0=1')
 284             $menu_errors[$cache_key]['reports'] = recountOpenReports(false);
 285 
 286         // Email failures that require attention
 287         if (!empty($modSettings['maillist_enabled']) && allowedTo('approve_emails'))
 288             $menu_errors[$cache_key]['emailmod'] = recountFailedEmails($approve_query);
 289 
 290         // Group requests
 291         if (!empty($user_info['mod_cache']) && $user_info['mod_cache']['gq'] != '0=1')
 292             $menu_errors[$cache_key]['groupreq'] = count(groupRequests());
 293 
 294         // Member requests
 295         if (allowedTo('moderate_forum') && ((!empty($modSettings['registration_method']) && $modSettings['registration_method'] == 2) || !empty($modSettings['approveAccountDeletion'])))
 296         {
 297             require_once(SUBSDIR . '/Members.subs.php');
 298             $awaiting_activation = 0;
 299             $activation_numbers = countInactiveMembers();
 300 
 301             // 5 = COPPA, 4 = Awaiting Deletion, 3 = Awaiting Approval
 302             foreach ($activation_numbers as $activation_type => $total_members)
 303             {
 304                 if (in_array($activation_type, array(3, 4, 5)))
 305                     $awaiting_activation += $total_members;
 306             }
 307             $menu_errors[$cache_key]['memberreq'] = $awaiting_activation;
 308         }
 309 
 310         // Grand Totals for the top most menus
 311         $menu_errors[$cache_key]['pt_total'] = $menu_errors[$cache_key]['emailmod'] + $menu_errors[$cache_key]['postmod'] + $menu_errors[$cache_key]['reports'] + $menu_errors[$cache_key]['attachments'];
 312         $menu_errors[$cache_key]['mg_total'] = $menu_errors[$cache_key]['memberreq'] + $menu_errors[$cache_key]['groupreq'];
 313         $menu_errors[$cache_key]['grand_total'] = $menu_errors[$cache_key]['pt_total'] + $menu_errors[$cache_key]['mg_total'];
 314 
 315         // Add this key in to the array, technically this resets the cache time for all keys
 316         // done this way as the entire thing needs to go null once *any* moderation action is taken
 317         $menu_errors = is_array($temp) ? array_merge($temp, $menu_errors) : $menu_errors;
 318 
 319         // Store it away for a while, not like this should change that often
 320         cache_put_data('num_menu_errors', $menu_errors, 900);
 321     }
 322     else
 323         $menu_errors = $temp === null ? array() : $temp;
 324 
 325     return $menu_errors[$cache_key];
 326 }
