107     /**
108      * Shows an interface to set and test censored words.
109      *
110      * - It uses the censor_vulgar, censor_proper, censorWholeWord, and
111      * censorIgnoreCase settings.
112      * - Requires the admin_forum permission.
113      * - Accessed from ?action=admin;area=postsettings;sa=censor.
114      *
115      * @uses the Admin template and the edit_censored sub template.
116      */
117     public function action_censor()
118     {
119         global $txt, $modSettings, $context;
120 
121         if (!empty($_POST['save_censor']))
122         {
123             // Make sure censoring is something they can do.
124             checkSession();
125             validateToken('admin-censor');
126 
127             $censored_vulgar = array();
128             $censored_proper = array();
129 
130             // Rip it apart, then split it into two arrays.
131             if (isset($_POST['censortext']))
132             {
133                 $_POST['censortext'] = explode("\n", strtr($_POST['censortext'], array("\r" => '')));
134 
135                 foreach ($_POST['censortext'] as $c)
136                     list ($censored_vulgar[], $censored_proper[]) = array_pad(explode('=', trim($c)), 2, '');
137             }
138             elseif (isset($_POST['censor_vulgar'], $_POST['censor_proper']))
139             {
140                 if (is_array($_POST['censor_vulgar']))
141                 {
142                     foreach ($_POST['censor_vulgar'] as $i => $value)
143                     {
144                         if (trim(strtr($value, '*', ' ')) == '')
145                             unset($_POST['censor_vulgar'][$i], $_POST['censor_proper'][$i]);
146                     }
147 
148                     $censored_vulgar = $_POST['censor_vulgar'];
149                     $censored_proper = $_POST['censor_proper'];
150                 }
151                 else
152                 {
153                     $censored_vulgar = explode("\n", strtr($_POST['censor_vulgar'], array("\r" => '')));
154                     $censored_proper = explode("\n", strtr($_POST['censor_proper'], array("\r" => '')));
155                 }
156             }
157 
158             // Set the new arrays and settings in the database.
159             $updates = array(
160                 'censor_vulgar' => implode("\n", $censored_vulgar),
161                 'censor_proper' => implode("\n", $censored_proper),
162                 'censorWholeWord' => empty($_POST['censorWholeWord']) ? '0' : '1',
163                 'censorIgnoreCase' => empty($_POST['censorIgnoreCase']) ? '0' : '1',
164             );
165 
166             call_integration_hook('integrate_save_censors', array(&$updates));
167 
168             updateSettings($updates);
169         }
170 
171         // Testing a word to see how it will be censored?
172         if (isset($_POST['censortest']))
173         {
174             require_once(SUBSDIR . '/Post.subs.php');
175             $censorText = htmlspecialchars($_POST['censortest'], ENT_QUOTES, 'UTF-8');
176             preparsecode($censorText);
177             $pre_censor = $censorText;
178             $context['censor_test'] = strtr(censorText($censorText), array('"' => '&quot;'));
179         }
180 
181         // Set everything up for the template to do its thang.
182         $censor_vulgar = explode("\n", $modSettings['censor_vulgar']);
183         $censor_proper = explode("\n", $modSettings['censor_proper']);
184 
185         $context['censored_words'] = array();
186         for ($i = 0, $n = count($censor_vulgar); $i < $n; $i++)
187         {
188             if (empty($censor_vulgar[$i]))
189                 continue;
190 
191             // Skip it, it's either spaces or stars only.
192             if (trim(strtr($censor_vulgar[$i], '*', ' ')) == '')
193                 continue;
194 
195             $context['censored_words'][htmlspecialchars(trim($censor_vulgar[$i]))] = isset($censor_proper[$i]) ? htmlspecialchars($censor_proper[$i], ENT_COMPAT, 'UTF-8') : '';
196         }
197 
198         call_integration_hook('integrate_censors');
199         createToken('admin-censor');
200 
201         // Using ajax?
202         if (isset($_REQUEST['xml'], $_POST['censortest']))
203         {
204             // Clear the templates
205             $template_layers = Template_Layers::getInstance();
206             $template_layers->removeAll();
207 
208             // Send back a response
209             loadTemplate('Json');
210             $context['sub_template'] = 'send_json';
211             $context['json_data'] = array(
212                 'result' => true,
213                 'censor' => $pre_censor . ' <i class="fa fa-arrow-circle-right"></i> ' . $context['censor_test'],
214                 'token_val' => $context['admin-censor_token_var'],
215                 'token' => $context['admin-censor_token'],
216             );
217         }
218         else
219         {
220             $context['sub_template'] = 'edit_censored';
221             $context['page_title'] = $txt['admin_censored_words'];
222         }
223     }

225     /**
226      * Modify any setting related to posts and posting.
227      *
228      * - Requires the admin_forum permission.
229      * - Accessed from ?action=admin;area=postsettings;sa=posts.
230      *
231      * @uses Admin template, edit_post_settings sub-template.
232      */
233     public function action_postSettings_display()
234     {
235         global $context, $txt, $modSettings, $scripturl, $db_type;
236 
237         // Initialize the form
238         $this->_initPostSettingsForm();
239 
240         $config_vars = $this->_postSettings->settings();
241 
242         // Setup the template.
243         $context['page_title'] = $txt['manageposts_settings'];
244         $context['sub_template'] = 'show_settings';
245 
246         // Are we saving them - are we??
247         if (isset($_GET['save']))
248         {
249             checkSession();
250 
251             // If we're changing the message length (and we are using MySQL) let's check the column is big enough.
252             if (isset($_POST['max_messageLength']) && $_POST['max_messageLength'] != $modSettings['max_messageLength'] && $db_type == 'mysql')
253             {
254                 require_once(SUBSDIR . '/Maintenance.subs.php');
255                 $colData = getMessageTableColumns();
256                 foreach ($colData as $column)
257                     if ($column['name'] == 'body')
258                         $body_type = $column['type'];
259 
260                 if (isset($body_type) && ($_POST['max_messageLength'] > 65535 || $_POST['max_messageLength'] == 0) && $body_type == 'text')
261                     fatal_lang_error('convert_to_mediumtext', false, array($scripturl . '?action=admin;area=maintain;sa=database'));
262 
263             }
264 
265             // If we're changing the post preview length let's check its valid
266             if (!empty($_POST['preview_characters']))
267                 $_POST['preview_characters'] = (int) min(max(0, $_POST['preview_characters']), 512);
268 
269             call_integration_hook('integrate_save_post_settings');
270 
271             Settings_Form::save_db($config_vars);
272             redirectexit('action=admin;area=postsettings;sa=posts');
273         }
274 
275         // Final settings...
276         $context['post_url'] = $scripturl . '?action=admin;area=postsettings;save;sa=posts';
277         $context['settings_title'] = $txt['manageposts_settings'];
278 
279         // Prepare the settings...
280         Settings_Form::prepare_db($config_vars);
281     }

297     /**
298      * Return admin configuration settings for posts.
299      */
300     private function _settings()
301     {
302         global $txt;
303 
304         // Initialize it with our settings
305         $config_vars = array(
306                 // Simple post options...
307                 array('check', 'removeNestedQuotes'),
308                 array('check', 'enableVideoEmbeding'),
309                 array('check', 'enableCodePrettify'),
310                 // Note show the warning as read if pspell not installed!
311                 array('check', 'enableSpellChecking', 'postinput' => (function_exists('pspell_new') ? $txt['enableSpellChecking_warning'] : '<span class="error">' . $txt['enableSpellChecking_error'] . '</span>')),
312             '',
313                 // Posting limits...
314                 array('int', 'max_messageLength', 'subtext' => $txt['max_messageLength_zero'], 'postinput' => $txt['manageposts_characters']),
315                 array('int', 'topicSummaryPosts', 'postinput' => $txt['manageposts_posts']),
316             '',
317                 // Posting time limits...
318                 array('int', 'spamWaitTime', 'postinput' => $txt['manageposts_seconds']),
319                 array('int', 'edit_wait_time', 'postinput' => $txt['manageposts_seconds']),
320                 array('int', 'edit_disable_time', 'subtext' => $txt['edit_disable_time_zero'], 'postinput' => $txt['manageposts_minutes']),
321             '',
322                 // First & Last message preview lengths
323                 array('select', 'message_index_preview', array($txt['message_index_preview_off'], $txt['message_index_preview_first'], $txt['message_index_preview_last'])),
324                 array('int', 'preview_characters', 'subtext' => $txt['preview_characters_zero'], 'postinput' => $txt['preview_characters_units']),
325         );
326 
327         // Add new settings with a nice hook, makes them available for admin settings search as well
328         call_integration_hook('integrate_modify_post_settings', array(&$config_vars));
329 
330         return $config_vars;
331     }
