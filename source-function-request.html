  4  * This parses PHP server variables, and initializes its own checking variables for use
  5  *

  7  * @copyright ElkArte Forum contributors
  8  * @license   BSD http://opensource.org/licenses/BSD-3-Clause
  9  *

 14 /**
 15  * Class to parse $_REQUEST for always necessary data, such as 'action', 'board', 'topic', 'start'.
 16  * Sanitizes the necessary data
 17  * Determines the origin of $_REQUEST for use in security checks
 18  */
 19 class Request
 20 {
 21     /**
 22      * Remote IP, if we can know it easily (as found in $_SERVER['REMOTE_ADDR'])
 23      * @var string
 24      */
 25     private $_client_ip;
 26 
 27     /**
 28      * Secondary IP, a double-check for the most accurate client IP we can get
 29      * @var string
 30      */
 31     private $_ban_ip;
 32 
 33     /**
 34      * HTTP or HTTPS scheme
 35      * @var string
 36      */
 37     private $_scheme;
 38 
 39     /**
 40      * User agent
 41      * @var string
 42      */
 43     private $_user_agent;
 44 
 45     /**
 46      * Whether the request is an XmlHttpRequest
 47      * @var bool
 48      */
 49     private $_xml;
 50 
 51     /**
 52      * Web server software
 53      * @var string
 54      */
 55     private $_server_software;
 56 
 57     /**
 58      * Sole private Request instance
 59      * @var Request
 60      */
 61     private static $_req = null;
 62 
 63     /**
 64      * Retrieves client ip
 65      */
 66     public function client_ip()
 67     {
 68         return $this->_client_ip;
 69     }
 70 
 71     /**
 72      * Return a secondary IP, result of a deeper check for the IP
 73      * It can be identical with client IP (and many times it will be).
 74      */
 75     public function ban_ip()
 76     {
 77         return $this->_ban_ip;
 78     }
 79 
 80     /**
 81      * Return the HTTP scheme
 82      */
 83     public function scheme()
 84     {
 85         return $this->_scheme;
 86     }
 87 
 88     /**
 89      * Return the user agent
 90      */
 91     public function user_agent()
 92     {
 93         return $this->_user_agent;
 94     }
 95 
 96     /**
 97      * Returns whether the request is XML
 98      */
 99     public function is_xml()
100     {
101         return $this->_xml;
102     }
103 
104     /**
105      * Returns server software (or empty string if it wasn't set for PHP)
106      */
107     public function server_software()
108     {
109         return $this->_server_software;
110     }
111 
112     /**
113      * Private constructor.
114      * It parses PHP server variables, and initializes its variables.
115      */
116     private function __construct()
117     {
118         // Client IP: REMOTE_ADDR, unless missing
119         if (!isset($_SERVER['REMOTE_ADDR']))
120         {
121             // Command line, or else.
122             $this->_client_ip = '';
123         }
124         // Perhaps we have a IPv6 address.
125         elseif (!isValidIPv6($_SERVER['REMOTE_ADDR']) || preg_match('~::ffff:\d+\.\d+\.\d+\.\d+~', $_SERVER['REMOTE_ADDR']) !== 0)
126         {
127             $this->_client_ip = preg_replace('~^::ffff:(\d+\.\d+\.\d+\.\d+)~', '\1', $_SERVER['REMOTE_ADDR']);
128 
129             // Just incase we have a legacy IPv4 address.
130             // @ TODO: Convert to IPv6.
131             if (preg_match('~^((([1]?\d)?\d|2[0-4]\d|25[0-5])\.){3}(([1]?\d)?\d|2[0-4]\d|25[0-5])$~', $this->_client_ip) === 0)
132                 $this->_client_ip = 'unknown';
133         }
134         else
135             $this->_client_ip = $_SERVER['REMOTE_ADDR'];
136 
137         // Second IP, guesswork it is, try to get the best IP we can, when using proxies or such
138         $this->_ban_ip = $this->_client_ip;
139 
140         // Forwarded, maybe?
141         if (!empty($_SERVER['HTTP_X_FORWARDED_FOR']) && !empty($_SERVER['HTTP_CLIENT_IP']) && (preg_match('~^((0|10|172\.(1[6-9]|2[0-9]|3[01])|192\.168|255|127)\.|unknown|::1|fe80::|fc00::)~', $_SERVER['HTTP_CLIENT_IP']) == 0 || preg_match('~^((0|10|172\.(1[6-9]|2[0-9]|3[01])|192\.168|255|127)\.|unknown|::1|fe80::|fc00::)~', $this->_client_ip) != 0))
142         {
143             // check the first forwarded for as the block - only switch if it's better that way.
144             if (strtok($_SERVER['HTTP_X_FORWARDED_FOR'], '.') != strtok($_SERVER['HTTP_CLIENT_IP'], '.') && '.' . strtok($_SERVER['HTTP_X_FORWARDED_FOR'], '.') == strrchr($_SERVER['HTTP_CLIENT_IP'], '.') && (preg_match('~^((0|10|172\.(1[6-9]|2[0-9]|3[01])|192\.168|255|127)\.|unknown)~', $_SERVER['HTTP_X_FORWARDED_FOR']) == 0 || preg_match('~^((0|10|172\.(1[6-9]|2[0-9]|3[01])|192\.168|255|127)\.|unknown)~', $this->_client_ip) != 0))
145                 $this->_ban_ip = implode('.', array_reverse(explode('.', $_SERVER['HTTP_CLIENT_IP'])));
146             else
147                 $this->_ban_ip = $_SERVER['HTTP_CLIENT_IP'];
148         }
149 
150         if (!empty($_SERVER['HTTP_CLIENT_IP']) && (preg_match('~^((0|10|172\.(1[6-9]|2[0-9]|3[01])|192\.168|255|127)\.|unknown|::1|fe80::|fc00::)~', $_SERVER['HTTP_CLIENT_IP']) == 0 || preg_match('~^((0|10|172\.(1[6-9]|2[0-9]|3[01])|192\.168|255|127)\.|unknown|::1|fe80::|fc00::)~', $this->_client_ip) != 0))
151         {

205     /**
206      * Parse the $_REQUEST, for always necessary data, such as 'action', 'board', 'topic', 'start'.
207      * Also figures out if this is an xml request.
208      */
209     public function parseRequest()
210     {
211         global $board, $topic;
212 
213         // Parse the request for our dear globals, I know
214         // they're in there somewhere...
215 
216         // Look for $board first
217         if (isset($_REQUEST['board']))
218         {
219             // Make sure it's a string (not an array, say)
220             $_REQUEST['board'] = (string) $_REQUEST['board'];
221 
222             // If we have ?board=3/10, that's... board=3, start=10! (old, compatible links.)
223             if (strpos($_REQUEST['board'], '/') !== false)
224                 list ($_REQUEST['board'], $_REQUEST['start']) = explode('/', $_REQUEST['board']);
225             // Or perhaps we have... ?board=1.0...
226             elseif (strpos($_REQUEST['board'], '.') !== false)
227                 list ($_REQUEST['board'], $_REQUEST['start']) = explode('.', $_REQUEST['board']);
228 
229             // $board and $_REQUEST['start'] are always numbers.
230             $board = (int) $_REQUEST['board'];
231             $_REQUEST['start'] = isset($_REQUEST['start']) ? (int) $_REQUEST['start'] : 0;
232 
233             // This is for "Who's Online" because it might come via POST - and it should be an int here.
234             $_GET['board'] = $board;
235         }
236         // None? We still need *something*, and it'd better be a number
237         else
238             $board = 0;
239 
240         // Look for threadid, old YaBB SE links have those. Just read it as a topic.
241         if (isset($_REQUEST['threadid']) && !isset($_REQUEST['topic']))
242             $_REQUEST['topic'] = $_REQUEST['threadid'];
243 
244         // Look for $topic
245         if (isset($_REQUEST['topic']))
246         {
247             // Make sure it's a string (not an array, say)
248             $_REQUEST['topic'] = (string) $_REQUEST['topic'];
249 
250             // It might come as ?topic=1/15, from an old, SMF beta style link
251             if (strpos($_REQUEST['topic'], '/') !== false)
252                 list ($_REQUEST['topic'], $_REQUEST['start']) = explode('/', $_REQUEST['topic']);
253             // Or it might come as ?topic=1.15.
254             elseif (strpos($_REQUEST['topic'], '.') !== false)
255                 list ($_REQUEST['topic'], $_REQUEST['start']) = explode('.', $_REQUEST['topic']);
256 
257             // $topic and $_REQUEST['start'] are numbers, numbers I say.
258             $topic = (int) $_REQUEST['topic'];
259             // @todo in Display $_REQUEST['start'] is not always a number
260             $_REQUEST['start'] = isset($_REQUEST['start']) && preg_match('~^(:?(:?from|msg)?\d+|new)$~', $_REQUEST['start']) ?  $_REQUEST['start'] : 0;
261 
262             // Now make sure the online log gets the right number.
263             $_GET['topic'] = $topic;
264         }
265         // No topic? Well, set something, and that something is 0.
266         else
267             $topic = 0;
268 
269         // There should be a $_REQUEST['start'], some at least.  If you need to default to other than 0, use $_GET['start'].
270         if (empty($_REQUEST['start']) || $_REQUEST['start'] < 0 || (int) $_REQUEST['start'] > 2147473647)
271             $_REQUEST['start'] = 0;
272 
273         // The action needs to be a string, too.
274         if (isset($_REQUEST['action']))
275             $_REQUEST['action'] = (string) $_REQUEST['action'];
276 
277         if (isset($_GET['action']))
278             $_GET['action'] = (string) $_GET['action'];
279 
280         $this->_xml = (isset($_SERVER['X_REQUESTED_WITH']) && $_SERVER['X_REQUESTED_WITH'] == 'XMLHttpRequest') || isset($_REQUEST['xml']);
281     }
