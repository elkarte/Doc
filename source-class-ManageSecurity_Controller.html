199                 $_POST['warning_moderate'] = $modSettings['postmod_active'] ? min($_POST['warning_moderate'], 100) : 0;
200                 $_POST['warning_mute'] = min($_POST['warning_mute'], 100);
201             }
202 
203             // Fix the warning setting array!
204             $_POST['warning_settings'] = '1,' . min(100, (int) $_POST['user_limit']) . ',' . min(100, (int) $_POST['warning_decrement']);

206             $save_vars[] = array('text', 'warning_settings');
207             unset($save_vars['rem1'], $save_vars['rem2']);
208 
209             call_integration_hook('integrate_save_moderation_settings', array(&$save_vars));
210 
211             Settings_Form::save_db($save_vars);
212             redirectexit('action=admin;area=securitysettings;sa=moderation');
213         }
214 
215         // We actually store lots of these together - for efficiency.
216         list ($modSettings['warning_enable'], $modSettings['user_limit'], $modSettings['warning_decrement']) = explode(',', $modSettings['warning_settings']);
217 
218         $context['post_url'] = $scripturl . '?action=admin;area=securitysettings;save;sa=moderation';
219         $context['settings_title'] = $txt['moderation_settings'];

245      * Handles admin security spam settings.
246      *
247      * - Displays a page with settings and eventually allows the admin to change them.
248      */
249     public function action_spamSettings_display()
250     {
251         global $txt, $scripturl, $context, $modSettings;
252 
253         // Initialize the form
254         $this->_initSpamSettingsForm();
255 
256         // Retrieve the current config settings
257         $config_vars = $this->_spamSettings->settings();
258 
259         // Saving?
260         if (isset($_GET['save']))
261         {
262             checkSession();
263 
264             // Fix PM settings.
265             $_POST['pm_spam_settings'] = (int) $_POST['max_pm_recipients'] . ',' . (int) $_POST['pm_posts_verification'] . ',' . (int) $_POST['pm_posts_per_hour'];
266 
267             // Guest requiring verification!
268             if (empty($_POST['posts_require_captcha']) && !empty($_POST['guests_require_captcha']))
269                 $_POST['posts_require_captcha'] = -1;
270 
271             $save_vars = $config_vars;
272             unset($save_vars['pm1'], $save_vars['pm2'], $save_vars['pm3'], $save_vars['guest_verify']);
273 
274             $save_vars[] = array('text', 'pm_spam_settings');
275 
276             call_integration_hook('integrate_save_spam_settings', array(&$save_vars));
277 
278             // Now save.
279             Settings_Form::save_db($save_vars);
280             cache_put_data('verificationQuestionIds', null, 300);
281             redirectexit('action=admin;area=securitysettings;sa=spam');
282         }
283 
284         // Add in PM spam settings on the fly
285         list ($modSettings['max_pm_recipients'], $modSettings['pm_posts_verification'], $modSettings['pm_posts_per_hour']) = explode(',', $modSettings['pm_spam_settings']);
286 
287         // And the same for guests requiring verification.
288         $modSettings['guests_require_captcha'] = !empty($modSettings['posts_require_captcha']);
289         $modSettings['posts_require_captcha'] = !isset($modSettings['posts_require_captcha']) || $modSettings['posts_require_captcha'] == -1 ? 0 : $modSettings['posts_require_captcha'];
290 
291         // Some minor javascript for the guest post setting.
292         if ($modSettings['posts_require_captcha'])
293             addInlineJavascript('document.getElementById(\'guests_require_captcha\').disabled = true;', true);
294 
295         $context['post_url'] = $scripturl . '?action=admin;area=securitysettings;save;sa=spam';
296         $context['settings_title'] = $txt['antispam_Settings'];
297         Settings_Form::prepare_db($config_vars);
298     }

322     {
323         global $txt, $scripturl, $context, $modSettings, $boardurl;

325         // Initialize the form
326         $this->_initBBSettingsForm();
327 
328         // Our callback templates are here
329         loadTemplate('BadBehavior');
330 
331         // Any errors to display?
332         if ($context['invalid_badbehavior_httpbl_key'])
333         {
334             $context['settings_message'][] = $txt['badbehavior_httpbl_key_invalid'];
335             $context['error_type'] = 'warning';
336         }
337 
338         // Have we blocked anything in the last 7 days?
339         if (!empty($modSettings['badbehavior_enabled']))
340             $context['settings_message'][] = bb2_insert_stats(true) . '<a href="' . $boardurl . '/index.php?action=admin;area=logs;sa=badbehaviorlog;desc" /> [' . $txt['badbehavior_details'] . ']</a>';
341 
342         // Current whitelist data
343         $whitelist = array('badbehavior_ip_wl', 'badbehavior_useragent_wl', 'badbehavior_url_wl');
344         foreach ($whitelist as $list)
345         {
346             $context[$list] = array();
347             $context[$list . '_desc'] = array();
348 
349             if (!empty($modSettings[$list]))
350                 $context[$list] = unserialize($modSettings[$list]);
351 
352             if (!empty($modSettings[$list . '_desc']))
353                 $context[$list . '_desc'] = unserialize($modSettings[$list . '_desc']);
354         }
355 
356         $config_vars = $this->_bbSettings->settings();
357 
358         // Saving?
359         if (isset($_GET['save']))
360         {
361             checkSession();
362 
363             // Make sure Bad Behavior defaults are set if nothing was specified
364             $_POST['badbehavior_httpbl_threat'] = empty($_POST['badbehavior_httpbl_threat']) ? 25 : $_POST['badbehavior_httpbl_threat'];
365             $_POST['badbehavior_httpbl_maxage'] = empty($_POST['badbehavior_httpbl_maxage']) ? 30 : $_POST['badbehavior_httpbl_maxage'];
366             $_POST['badbehavior_reverse_proxy_header'] = empty($_POST['badbehavior_reverse_proxy_header']) ? 'X-Forwarded-For' : $_POST['badbehavior_reverse_proxy_header'];
367 
368             // Build up the whitelist options
369             foreach ($whitelist as $list)
370             {
371                 $this_list = array();
372                 $this_desc = array();
373 
374                 if (isset($_POST[$list]))
375                 {
376                     // Clear blanks from the data field, only grab the comments that don't have blank data value
377                     $this_list = array_map('trim', array_filter($_POST[$list]));
378                     $this_desc = array_intersect_key($_POST[$list . '_desc'], $this_list);
379                 }
380                 updateSettings(array($list => serialize($this_list), $list . '_desc' => serialize($this_desc)));
381             }
382 
383             Settings_Form::save_db($config_vars);
384             redirectexit('action=admin;area=securitysettings;sa=badbehavior');
385         }
386 
387         $context['post_url'] = $scripturl . '?action=admin;area=securitysettings;save;sa=badbehavior';
388 
389         // Javascript vars for the "add more xyz" buttons in the callback forms
390         addJavascriptVar(array(
391             'sUrlParent' => '\'add_more_url_placeholder\'',
392             'oUrlOptionsdt' => '{name: \'badbehavior_url_wl_desc[]\', class: \'input_text\'}',
393             'oUrlOptionsdd' => '{name: \'badbehavior_url_wl[]\', class: \'input_text\'}',
