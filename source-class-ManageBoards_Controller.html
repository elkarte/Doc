133     public function action_main()
134     {
135         global $txt, $context, $cat_tree, $boards, $boardList, $scripturl;
136 
137         loadTemplate('ManageBoards');
138 
139         require_once(SUBSDIR . '/Boards.subs.php');
140 
141         // Moving a board, child of, before, after, top
142         if (isset($_REQUEST['sa']) && $_REQUEST['sa'] == 'move' && in_array($_REQUEST['move_to'], array('child', 'before', 'after', 'top')))
143         {
144             checkSession('get');
145             validateToken('admin-bm-' . (int) $_REQUEST['src_board'], 'request');
146 
147             // Top is special, its the top!
148             if ($_REQUEST['move_to'] === 'top')
149                 $boardOptions = array(
150                     'move_to' => $_REQUEST['move_to'],
151                     'target_category' => (int) $_REQUEST['target_cat'],
152                     'move_first_child' => true,
153                 );
154             // Moving it after another board
155             else
156                 $boardOptions = array(
157                     'move_to' => $_REQUEST['move_to'],
158                     'target_board' => (int) $_REQUEST['target_board'],
159                     'move_first_child' => true,
160                 );
161 
162             // Use modifyBoard to perform the action
163             modifyBoard((int) $_REQUEST['src_board'], $boardOptions);
164         }
165 
166         getBoardTree();
167 
168         createToken('admin-sort');
169         $context['move_board'] = !empty($_REQUEST['move']) && isset($boards[(int) $_REQUEST['move']]) ? (int) $_REQUEST['move'] : 0;
170 
171         $context['categories'] = array();
172         foreach ($cat_tree as $catid => $tree)
173         {
174             $context['categories'][$catid] = array(
175                 'name' => &$tree['node']['name'],
176                 'id' => &$tree['node']['id'],
177                 'boards' => array()
178             );
179             $move_cat = !empty($context['move_board']) && $boards[$context['move_board']]['category'] == $catid;
180             foreach ($boardList[$catid] as $boardid)
181             {
182                 $context['categories'][$catid]['boards'][$boardid] = array(
183                     'id' => &$boards[$boardid]['id'],
184                     'name' => &$boards[$boardid]['name'],
185                     'description' => &$boards[$boardid]['description'],
186                     'child_level' => &$boards[$boardid]['level'],
187                     'move' => $move_cat && ($boardid == $context['move_board'] || isChildOf($boardid, $context['move_board'])),
188                     'permission_profile' => &$boards[$boardid]['profile'],
189                 );
190             }
191         }
192 
193         if (!empty($context['move_board']))
194         {
195             createToken('admin-bm-' . $context['move_board'], 'request');
196 
197             $context['move_title'] = sprintf($txt['mboards_select_destination'], htmlspecialchars($boards[$context['move_board']]['name'], ENT_COMPAT, 'UTF-8'));
198             foreach ($cat_tree as $catid => $tree)
199             {
200                 $prev_child_level = 0;
201                 $prev_board = 0;
202                 $stack = array();
203 
204                 // Just a shortcut, this is the same for all the urls
205                 $security = $context['session_var'] . '=' . $context['session_id'] . ';' . $context['admin-bm-' . $context['move_board'] . '_token_var'] . '=' . $context['admin-bm-' . $context['move_board'] . '_token'];
206                 foreach ($boardList[$catid] as $boardid)
207                 {
208                     if (!isset($context['categories'][$catid]['move_link']))
209                         $context['categories'][$catid]['move_link'] = array(
210                             'child_level' => 0,
211                             'label' => $txt['mboards_order_before'] . ' \'' . htmlspecialchars($boards[$boardid]['name'], ENT_COMPAT, 'UTF-8') . '\'',
212                             'href' => $scripturl . '?action=admin;area=manageboards;sa=move;src_board=' . $context['move_board'] . ';target_board=' . $boardid . ';move_to=before;' . $security,
213                         );
214 
215                     if (!$context['categories'][$catid]['boards'][$boardid]['move'])
216                     $context['categories'][$catid]['boards'][$boardid]['move_links'] = array(
217                         array(
218                             'child_level' => $boards[$boardid]['level'],
219                             'label' => $txt['mboards_order_after'] . '\'' . htmlspecialchars($boards[$boardid]['name'], ENT_COMPAT, 'UTF-8') . '\'',
220                             'href' => $scripturl . '?action=admin;area=manageboards;sa=move;src_board=' . $context['move_board'] . ';target_board=' . $boardid . ';move_to=after;' . $security,
221                         ),
222                         array(
223                             'child_level' => $boards[$boardid]['level'] + 1,
224                             'label' => $txt['mboards_order_child_of'] . ' \'' . htmlspecialchars($boards[$boardid]['name'], ENT_COMPAT, 'UTF-8') . '\'',
225                             'href' => $scripturl . '?action=admin;area=manageboards;sa=move;src_board=' . $context['move_board'] . ';target_board=' . $boardid . ';move_to=child;' . $security,
226                         ),
227                     );
228 
229                     $difference = $boards[$boardid]['level'] - $prev_child_level;
230                     if ($difference == 1)
231                         array_push($stack, !empty($context['categories'][$catid]['boards'][$prev_board]['move_links']) ? array_shift($context['categories'][$catid]['boards'][$prev_board]['move_links']) : null);
232                     elseif ($difference < 0)
233                     {
234                         if (empty($context['categories'][$catid]['boards'][$prev_board]['move_links']))
235                             $context['categories'][$catid]['boards'][$prev_board]['move_links'] = array();
236 
237                         for ($i = 0; $i < -$difference; $i++)
238                             if (($temp = array_pop($stack)) != null)
239                                 array_unshift($context['categories'][$catid]['boards'][$prev_board]['move_links'], $temp);
240                     }
241 
242                     $prev_board = $boardid;
243                     $prev_child_level = $boards[$boardid]['level'];
244 
245                 }
246 
247                 if (!empty($stack) && !empty($context['categories'][$catid]['boards'][$prev_board]['move_links']))
248                     $context['categories'][$catid]['boards'][$prev_board]['move_links'] = array_merge($stack, $context['categories'][$catid]['boards'][$prev_board]['move_links']);
249                 elseif (!empty($stack))
250                     $context['categories'][$catid]['boards'][$prev_board]['move_links'] = $stack;
251 
252                 if (empty($boardList[$catid]))
253                     $context['categories'][$catid]['move_link'] = array(
254                         'child_level' => 0,
255                         'label' => $txt['mboards_order_before'] . ' \'' . htmlspecialchars($tree['node']['name'], ENT_COMPAT, 'UTF-8') . '\'',
256                         'href' => $scripturl . '?action=admin;area=manageboards;sa=move;src_board=' . $context['move_board'] . ';target_cat=' . $catid . ';move_to=top;' . $security,
257                     );
258             }
259         }
260 
261         call_integration_hook('integrate_boards_main');
262 
263         $context['page_title'] = $txt['boards_and_cats'];
264         $context['sub_template'] = 'manage_boards';
265         $context['can_manage_permissions'] = allowedTo('manage_permissions');
266     }

344         }
345         if (!isset($_REQUEST['delete']))
346         {

349         }

351         {

353             $context['page_title'] = $txt['mboards_delete_cat'];
354         }
355 
356         // Create a special token.
357         createToken('admin-bc-' . $_REQUEST['cat']);
358         $context['token_check'] = 'admin-bc-' . $_REQUEST['cat'];
359 
360         call_integration_hook('integrate_edit_category');

363     /**
364      * Function for handling a submitted form saving the category.
365      *
366      * What it does:
367      * - complete the modifications to a specific category.
368      * - It also handles deletion of a category.
369      * - It requires manage_boards permission.
370      * - Called by ?action=admin;area=manageboards;sa=cat2
371      * - Redirects to ?action=admin;area=manageboards.
372      */
373     public function action_cat2()
374     {
375         checkSession();
376         validateToken('admin-bc-' . $_REQUEST['cat']);
377 
378         require_once(SUBSDIR . '/Categories.subs.php');
379 
380         $_POST['cat'] = (int) $_POST['cat'];
381 
382         // Add a new category or modify an existing one..
383         if (isset($_POST['edit']) || isset($_POST['add']))
384         {
385             $catOptions = array();
386 
387             if (isset($_POST['cat_order']))
388                 $catOptions['move_after'] = (int) $_POST['cat_order'];
389 
390             // Change "This & That" to "This &amp; That" but don't change "&cent" to "&amp;cent;"...
391             $catOptions['cat_name'] = preg_replace('~[&]([^;]{8}|[^;]{0,8}$)~', '&amp;$1', $_POST['cat_name']);
392 
393             $catOptions['is_collapsible'] = isset($_POST['collapse']);
394 
395             if (isset($_POST['add']))
396                 createCategory($catOptions);
397             else
398                 modifyCategory($_POST['cat'], $catOptions);
399         }
400         // If they want to delete - first give them confirmation.
401         elseif (isset($_POST['delete']) && !isset($_POST['confirmation']) && !isset($_POST['empty']))
402         {
403             // We need a new token.
404             validateToken('admin-bc-' . $_REQUEST['cat']);
405 
406             $this->action_cat();
407             return;
408         }
409         // Delete the category!
410         elseif (isset($_POST['delete']))
411         {
412             // First off - check if we are moving all the current boards first - before we start deleting!
413             if (isset($_POST['delete_action']) && $_POST['delete_action'] == 1)
414             {
415                 if (empty($_POST['cat_to']))
416                     fatal_lang_error('mboards_delete_error');
417 
418                 deleteCategories(array($_POST['cat']), (int) $_POST['cat_to']);
419             }
420             else
421                 deleteCategories(array($_POST['cat']));
422         }
423 
424         redirectexit('action=admin;area=manageboards');
425     }
